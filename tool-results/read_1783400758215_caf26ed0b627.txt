    36→// ---------- Typo-tolerance helper ----------
    37→function levenshteinDistance(a: string, b: string): number {
    38→  const matrix: number[][] = [];
    39→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    40→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    41→  for (let i = 1; i <= b.length; i++) {
    42→    for (let j = 1; j <= a.length; j++) {
    43→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    44→      matrix[i][j] = Math.min(
    45→        matrix[i - 1][j] + 1,
    46→        matrix[i][j - 1] + 1,
    47→        matrix[i - 1][j - 1] + cost,
    48→      );
    49→    }
    50→  }
    51→  return matrix[b.length][a.length];
    52→}

    54→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    55→  const ua = userAnswer.trim().toLowerCase();
    56→  const ca = correctAnswer.trim().toLowerCase();
    57→  if (ua === ca) return true;
    58→  if (ua.includes(ca) || ca.includes(ua)) return true;
    59→  // Levenshtein distance for typo tolerance
    60→  const dist = levenshteinDistance(ua, ca);
    61→  // Allow 1 typo for short answers, 2 for longer ones
    62→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    63→  if (dist <= threshold) return true;
    64→  // Word-level check: all words of correct answer present in user answer (in any order)
    65→  const correctWords = ca.split(/\s+/);
    66→  const userWords = ua.split(/\s+/);
    67→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    68→  return false;
    69→}

   236→function updateDailyStreak(): { current: number; best: number; lastDate: string } {
   237→  const streak = loadDailyStreak();
   238→  const today = getTodayStr();
   239→
   240→  if (streak.lastDate === today) return streak;
   241→
   242→  const yesterday = new Date();
   243→  yesterday.setDate(yesterday.getDate() - 1);
   244→  const yesterdayStr = yesterday.toISOString().split('T')[0];
   245→
   246→  let newCurrent: number;
   247→  if (streak.lastDate === yesterdayStr) {
   248→    newCurrent = streak.current + 1;
   249→  } else {
   250→    newCurrent = 1;
   251→  }
   252→
   253→  const newBest = Math.max(streak.best, newCurrent);
   254→  const updated = { current: newCurrent, best: newBest, lastDate: today };
   255→  saveDailyStreak(updated);
   256→  return updated;
   257→}

   259→function getTimeUntilMidnight(): { hours: number; minutes: number; seconds: number } {
   260→  const now = new Date();
   261→  const midnight = new Date(now);
   262→  midnight.setHours(24, 0, 0, 0);
   263→  const diff = midnight.getTime() - now.getTime();
   264→  return {
   265→    hours: Math.floor(diff / (1000 * 60 * 60)),
   266→    minutes: Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60)),
   267→    seconds: Math.floor((diff % (1000 * 60)) / 1000),
   268→  };
   269→}

   308→// ---------- Animated counter hook ----------
   309→function useAnimatedCounter(target: number, duration: number = 1200) {
   310→  const [count, setCount] = useState(0);
   311→  const rafRef = useRef<number>(0);
   312→  useEffect(() => {
   313→    const start = performance.now();
   314→    const tick = (now: number) => {
   315→      const elapsed = now - start;
   316→      const progress = Math.min(elapsed / duration, 1);
   317→      const eased = 1 - Math.pow(1 - progress, 3);
   318→      setCount(Math.round(eased * target));
   319→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   320→    };
   321→    rafRef.current = requestAnimationFrame(tick);
   322→    return () => cancelAnimationFrame(rafRef.current);
   323→  }, [target, duration]);
   324→  return count;
   325→}

   378→// ---------- Matching question component (drag-and-drop with SVG connectors) ----------
   379→function MatchingInput({
   380→  pairs,
   381→  onAnswer,
   382→}: {
   383→  pairs: Array<{ left: string; right: string }>;
   384→  onAnswer: (answer: string) => void;
   385→}) {
   386→  const shuffledRight = useMemo(() => {
   387→    const arr = [...pairs.map((p) => p.right)];
   388→    for (let i = arr.length - 1; i > 0; i--) {
   389→      const j = Math.floor(Math.random() * (i + 1));
   390→      [arr[i], arr[j]] = [arr[j], arr[i]];
   391→    }
   392→    return arr;
   393→  }, [pairs]);
   394→
   395→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   396→  const [matches, setMatches] = useState<Record<string, string>>({});
   397→  const [dragItem, setDragItem] = useState<string | null>(null);
   398→  const [dragOverTarget, setDragOverTarget] = useState<string | null>(null);
   399→  const [linePositions, setLinePositions] = useState<Array<{ id: string; x1: number; y1: number; x2: number; y2: number }>>([]);
   400→  const containerRef = useRef<HTMLDivElement>(null);
   401→
   402→  // Build index maps for safe DOM queries
   403→  const leftIdxMap = useMemo(() => {
   404→    const map: Record<string, number> = {};
   405→    pairs.forEach((p, i) => { map[p.left] = i; });
   406→    return map;
   407→  }, [pairs]);
   408→
   409→  const rightIdxMap = useMemo(() => {
   410→    const map: Record<string, number> = {};
   411→    shuffledRight.forEach((r, i) => { map[r] = i; });
   412→    return map;
   413→  }, [shuffledRight]);
   414→
   415→  // Compute SVG connector line positions
   416→  const updateLinePositions = useCallback(() => {
   417→    const container = containerRef.current;
   418→    if (!container) return;
   419→
   420→    requestAnimationFrame(() => {
   421→      if (!container) return;
   422→      const containerRect = container.getBoundingClientRect();
   423→      const positions: Array<{ id: string; x1: number; y1: number; x2: number; y2: number }> = [];
   424→
   425→      for (const [left, right] of Object.entries(matches)) {
   426→        const lIdx = leftIdxMap[left];
   427→        const rIdx = rightIdxMap[right];
   428→        if (lIdx === undefined || rIdx === undefined) continue;
   429→
   430→        const leftEl = container.querySelector(`[data-left-idx="${lIdx}"]`);
   431→        const rightEl = container.querySelector(`[data-right-idx="${rIdx}"]`);
   432→
   433→        if (leftEl && rightEl) {
   434→          const leftRect = leftEl.getBoundingClientRect();
   435→          const rightRect = rightEl.getBoundingClientRect();
   436→
   437→          positions.push({
   438→            id: `${left}-${right}`,
   439→            x1: leftRect.right - containerRect.left,
   440→            y1: leftRect.top + leftRect.height / 2 - containerRect.top,
   441→            x2: rightRect.left - containerRect.left,
   442→            y2: rightRect.top + rightRect.height / 2 - containerRect.top,
   443→          });
   444→        }
   445→      }
   446→
   447→      setLinePositions(positions);
   448→    });
   449→  }, [matches, leftIdxMap, rightIdxMap]);
   450→
   451→  // Recalculate on matches change
   452→  useEffect(() => {
   453→    updateLinePositions();
   454→  }, [updateLinePositions]);
   455→
   456→  // Recalculate on resize
   457→  useEffect(() => {
   458→    const container = containerRef.current;
   459→    if (!container) return;
   460→
   461→    const observer = new ResizeObserver(updateLinePositions);
   462→    observer.observe(container);
   463→
   464→    return () => observer.disconnect();
   465→  }, [updateLinePositions]);
   466→
   467→  const handleLeftClick = (left: string) => {
   468→    if (matches[left]) {
   469→      // Remove match by clicking connected left item
   470→      setMatches((prev) => {
   471→        const next = { ...prev };
   472→        delete next[left];
   473→        return next;
   474→      });
   475→      return;
   476→    }
   477→    setSelectedLeft(left);
   478→  };
   479→
   480→  const handleRightClick = (right: string) => {
   481→    if (!selectedLeft) return;
   482→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   483→    setSelectedLeft(null);
   484→  };
   485→
   486→  const handleDragStart = (e: React.DragEvent, left: string) => {
   487→    setDragItem(left);
   488→    e.dataTransfer.setData('text/plain', left);
   489→    e.dataTransfer.effectAllowed = 'link';
   490→  };
   491→
   492→  const handleDragOver = (e: React.DragEvent, right: string) => {
   493→    e.preventDefault();
   494→    e.dataTransfer.dropEffect = 'link';
   495→    const isUsedByOther = Object.entries(matches).some(
   496→      ([k, v]) => v === right && k !== dragItem,
   497→    );
   498→    if (dragItem && !isUsedByOther) {
   499→      setDragOverTarget(right);
   500→    }
   501→  };
   502→
   503→  const handleDragLeave = () => {
   504→    setDragOverTarget(null);
   505→  };
   506→
   507→  const handleDrop = (e: React.DragEvent, right: string) => {
   508→    e.preventDefault();
   509→    const left = e.dataTransfer.getData('text/plain') || dragItem;
   510→    if (left) {
   511→      // Remove any existing match for this left item first
   512→      setMatches((prev) => {
   513→        const next = { ...prev };
   514→        delete next[left];
   515→        next[left] = right;
   516→        return next;
   517→      });
   518→    }
   519→    setDragItem(null);
   520→    setDragOverTarget(null);
   521→  };
   522→
   523→  const handleDragEnd = () => {
   524→    setDragItem(null);
   525→    setDragOverTarget(null);
   526→  };
   527→
   528→  const handleSubmit = () => {
   529→    const answer = Object.entries(matches)
   530→      .map(([l, r]) => `${l}-${r}`)
   531→      .join(', ');
   532→    onAnswer(answer);
   533→  };
   534→
   535→  const allMatched = pairs.every((p) => matches[p.left]);
   536→  const usedRights = new Set(Object.values(matches));
   537→
   538→  return (
   539→    <div className="space-y-4">
   540→      <p className="text-sm text-muted-foreground">
   541→        Drag left items to their match on the right, or click to select. Click a connected left item to remove it.
   542→      </p>
   543→      <div ref={containerRef} className="relative grid grid-cols-2 gap-4">
   544→        {/* SVG Connector Overlay */}
