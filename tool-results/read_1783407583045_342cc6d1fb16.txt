    47→// ---------- Typo-tolerance helper ----------
    48→function levenshteinDistance(a: string, b: string): number {
    49→  const matrix: number[][] = [];
    50→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    51→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    52→  for (let i = 1; i <= b.length; i++) {
    53→    for (let j = 1; j <= a.length; j++) {
    54→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    55→      matrix[i][j] = Math.min(
    56→        matrix[i - 1][j] + 1,
    57→        matrix[i][j - 1] + 1,
    58→        matrix[i - 1][j - 1] + cost,
    59→      );
    60→    }
    61→  }
    62→  return matrix[b.length][a.length];
    63→}

    65→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    66→  const ua = userAnswer.trim().toLowerCase();
    67→  const ca = correctAnswer.trim().toLowerCase();
    68→  if (ua === ca) return true;
    69→  if (ua.includes(ca) || ca.includes(ua)) return true;
    70→  // Levenshtein distance for typo tolerance
    71→  const dist = levenshteinDistance(ua, ca);
    72→  // Allow 1 typo for short answers, 2 for longer ones
    73→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    74→  if (dist <= threshold) return true;
    75→  // Word-level check: all words of correct answer present in user answer (in any order)
    76→  const correctWords = ca.split(/\s+/);
    77→  const userWords = ua.split(/\s+/);
    78→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    79→  return false;
    80→}

   247→function updateDailyStreak(): { current: number; best: number; lastDate: string } {
   248→  const streak = loadDailyStreak();
   249→  const today = getTodayStr();
   250→
   251→  if (streak.lastDate === today) return streak;
   252→
   253→  const yesterday = new Date();
   254→  yesterday.setDate(yesterday.getDate() - 1);
   255→  const yesterdayStr = yesterday.toISOString().split('T')[0];
   256→
   257→  let newCurrent: number;
   258→  if (streak.lastDate === yesterdayStr) {
   259→    newCurrent = streak.current + 1;
   260→  } else {
   261→    newCurrent = 1;
   262→  }
   263→
   264→  const newBest = Math.max(streak.best, newCurrent);
   265→  const updated = { current: newCurrent, best: newBest, lastDate: today };
   266→  saveDailyStreak(updated);
   267→  return updated;
   268→}

   270→function getTimeUntilMidnight(): { hours: number; minutes: number; seconds: number } {
   271→  const now = new Date();
   272→  const midnight = new Date(now);
   273→  midnight.setHours(24, 0, 0, 0);
   274→  const diff = midnight.getTime() - now.getTime();
   275→  return {
   276→    hours: Math.floor(diff / (1000 * 60 * 60)),
   277→    minutes: Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60)),
   278→    seconds: Math.floor((diff % (1000 * 60)) / 1000),
   279→  };
   280→}

   332→function scoreQuestionForAdaptive(input: AdaptiveScoringInput): ScoredQuestion {
   333→  const { question, masteryMap, adaptiveResults, dueConcepts } = input;
   334→  let score = 0;
   335→  const reasons: string[] = [];
   336→  const concept = question.concept || 'Unknown';
   337→  const mastery = masteryMap[concept];
   338→
   339→  // 1. Low mastery bonus (level 1-2)
   340→  if (mastery && mastery.level < 3) {
   341→    score += 3;
   342→    reasons.push('low mastery');
   343→  }
   344→
   345→  // 2. Spaced repetition due for review
   346→  if (dueConcepts.has(concept)) {
   347→    score += 2;
   348→    reasons.push('needs review');
   349→  }
   350→
   351→  // 3. Recent performance: last 5 results on concept
   352→  const conceptResults = adaptiveResults
   353→    .filter((r) => r.concept === concept)
   354→    .slice(-5);
   355→  const last3 = conceptResults.slice(-3);
   356→  const wrongCount = last3.filter((r) => !r.correct).length;
   357→  const correctCount = conceptResults.filter((r) => r.correct).length;
   358→
   359→  if (wrongCount >= 2) {
   360→    score += 2;
   361→    reasons.push('recent struggles');
   362→  }
   363→
   364→  // 4. Difficulty match bonus
   365→  const targetLevel = mastery ? Math.min(mastery.level + 1, 3) : 1;
   366→  const targetDifficulty = targetLevel === 1 ? 'easy' : targetLevel === 2 ? 'medium' : 'hard';
   367→  if (question.difficulty === targetDifficulty) {
   368→    score += 1;
   369→  } else if (
   370→    (targetDifficulty === 'medium' && question.difficulty === 'easy') ||
   371→    (targetDifficulty === 'hard' && question.difficulty === 'medium')
   372→  ) {
   373→    score += 0.5;
   374→  }
   375→
   376→  // 5. Boost for concepts with many wrong answers (weaker areas)
   377→  if (correctCount >= 3) {
   378→    score += 1;
   379→    reasons.push('ready for harder questions');
   380→  }
   381→
   382→  return { question, score, reasons };
   383→}

   385→function getAdaptiveReasoning(scored: ScoredQuestion[]): string {
   386→  if (scored.length === 0) return '';
   387→
   388→  const conceptReasons = new Map<string, string[]>();
   389→  for (const sq of scored.slice(0, 5)) {
   390→    const concept = sq.question.concept || 'Unknown';
   391→    const existing = conceptReasons.get(concept) || [];
   392→    conceptReasons.set(concept, [...existing, ...sq.reasons]);
   393→  }
   394→
   395→  const parts: string[] = [];
   396→  for (const [concept, reasons] of conceptReasons) {
   397→    const uniqueReasons = [...new Set(reasons)];
   398→    const desc = uniqueReasons.map((r) => {
   399→      switch (r) {
   400→        case 'low mastery': return 'low mastery';
   401→        case 'needs review': return 'needs review';
   402→        case 'recent struggles': return 'struggling recently';
   403→        case 'ready for harder questions': return 'ready for challenge';
   404→        default: return r;
   405→      }
   406→    }).join(', ');
   407→    parts.push(`${concept} (${desc})`);
   408→  }
   409→
   410→  return 'Focusing on: ' + parts.join(', ');
   411→}

   477→// ---------- Animated counter hook ----------
   478→function useAnimatedCounter(target: number, duration: number = 1200) {
   479→  const [count, setCount] = useState(0);
   480→  const rafRef = useRef<number>(0);
   481→  useEffect(() => {
   482→    const start = performance.now();
   483→    const tick = (now: number) => {
   484→      const elapsed = now - start;
   485→      const progress = Math.min(elapsed / duration, 1);
   486→      const eased = 1 - Math.pow(1 - progress, 3);
   487→      setCount(Math.round(eased * target));
   488→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   489→    };
   490→    rafRef.current = requestAnimationFrame(tick);
   491→    return () => cancelAnimationFrame(rafRef.current);
   492→  }, [target, duration]);
   493→  return count;
   494→}

   547→// ---------- Matching question component (drag-and-drop with SVG connectors) ----------
   548→function MatchingInput({
   549→  pairs,
   550→  onAnswer,
   551→}: {
   552→  pairs: Array<{ left: string; right: string }>;
   553→  onAnswer: (answer: string) => void;
   554→}) {
   555→  const shuffledRight = useMemo(() => {
   556→    const arr = [...pairs.map((p) => p.right)];
   557→    for (let i = arr.length - 1; i > 0; i--) {
   558→      const j = Math.floor(Math.random() * (i + 1));
   559→      [arr[i], arr[j]] = [arr[j], arr[i]];
   560→    }
   561→    return arr;
   562→  }, [pairs]);
   563→
   564→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   565→  const [matches, setMatches] = useState<Record<string, string>>({});
   566→  const [dragItem, setDragItem] = useState<string | null>(null);
   567→  const [dragOverTarget, setDragOverTarget] = useState<string | null>(null);
   568→  const [linePositions, setLinePositions] = useState<Array<{ id: string; x1: number; y1: number; x2: number; y2: number }>>([]);
   569→  const containerRef = useRef<HTMLDivElement>(null);
   570→
   571→  // Build index maps for safe DOM queries
   572→  const leftIdxMap = useMemo(() => {
   573→    const map: Record<string, number> = {};
   574→    pairs.forEach((p, i) => { map[p.left] = i; });
   575→    return map;
   576→  }, [pairs]);
   577→
   578→  const rightIdxMap = useMemo(() => {
   579→    const map: Record<string, number> = {};
   580→    shuffledRight.forEach((r, i) => { map[r] = i; });
   581→    return map;
   582→  }, [shuffledRight]);
   583→
   584→  // Compute SVG connector line positions
   585→  const updateLinePositions = useCallback(() => {
   586→    const container = containerRef.current;
   587→    if (!container) return;
   588→
   589→    requestAnimationFrame(() => {
   590→      if (!container) return;
   591→      const containerRect = container.getBoundingClientRect();
   592→      const positions: Array<{ id: string; x1: number; y1: number; x2: number; y2: number }> = [];
   593→
   594→      for (const [left, right] of Object.entries(matches)) {
   595→        const lIdx = leftIdxMap[left];
   596→        const rIdx = rightIdxMap[right];
   597→        if (lIdx === undefined || rIdx === undefined) continue;
   598→
   599→        const leftEl = container.querySelector(`[data-left-idx="${lIdx}"]`);
   600→        const rightEl = container.querySelector(`[data-right-idx="${rIdx}"]`);
   601→
   602→        if (leftEl && rightEl) {
   603→          const leftRect = leftEl.getBoundingClientRect();
   604→          const rightRect = rightEl.getBoundingClientRect();
   605→
   606→          positions.push({
   607→            id: `${left}-${right}`,
   608→            x1: leftRect.right - containerRect.left,
   609→            y1: leftRect.top + leftRect.height / 2 - containerRect.top,
   610→            x2: rightRect.left - containerRect.left,
   611→            y2: rightRect.top + rightRect.height / 2 - containerRect.top,
   612→          });
   613→        }
   614→      }
   615→
   616→      setLinePositions(positions);
   617→    });
   618→  }, [matches, leftIdxMap, rightIdxMap]);
   619→
   620→  // Recalculate on matches change
   621→  useEffect(() => {
   622→    updateLinePositions();
   623→  }, [updateLinePositions]);
   624→
   625→  // Recalculate on resize
   626→  useEffect(() => {
   627→    const container = containerRef.current;
   628→    if (!container) return;
   629→
   630→    const observer = new ResizeObserver(updateLinePositions);
   631→    observer.observe(container);
   632→
   633→    return () => observer.disconnect();
   634→  }, [updateLinePositions]);
   635→
   636→  const handleLeftClick = (left: string) => {
   637→    if (matches[left]) {
   638→      // Remove match by clicking connected left item
   639→      setMatches((prev) => {
   640→        const next = { ...prev };
   641→        delete next[left];
   642→        return next;
   643→      });
   644→      return;
   645→    }
   646→    setSelectedLeft(left);
   647→  };
   648→
   649→  const handleRightClick = (right: string) => {
   650→    if (!selectedLeft) return;
   651→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   652→    setSelectedLeft(null);
   653→  };
   654→
   655→  const handleDragStart = (e: React.DragEvent, left: string) => {
   656→    setDragItem(left);
   657→    e.dataTransfer.setData('text/plain', left);
   658→    e.dataTransfer.effectAllowed = 'link';
   659→  };
   660→
   661→  const handleDragOver = (e: React.DragEvent, right: string) => {
   662→    e.preventDefault();
   663→    e.dataTransfer.dropEffect = 'link';
   664→    const isUsedByOther = Object.entries(matches).some(
   665→      ([k, v]) => v === right && k !== dragItem,
   666→    );
   667→    if (dragItem && !isUsedByOther) {
   668→      setDragOverTarget(right);
   669→    }
   670→  };
   671→
   672→  const handleDragLeave = () => {
   673→    setDragOverTarget(null);
   674→  };
   675→
   676→  const handleDrop = (e: React.DragEvent, right: string) => {
   677→    e.preventDefault();
   678→    const left = e.dataTransfer.getData('text/plain') || dragItem;
   679→    if (left) {
   680→      // Remove any existing match for this left item first
   681→      setMatches((prev) => {
   682→        const next = { ...prev };
   683→        delete next[left];
   684→        next[left] = right;
   685→        return next;
   686→      });
   687→    }
   688→    setDragItem(null);
   689→    setDragOverTarget(null);
   690→  };
   691→
   692→  const handleDragEnd = () => {
   693→    setDragItem(null);
   694→    setDragOverTarget(null);
   695→  };
   696→
   697→  const handleSubmit = () => {
   698→    const answer = Object.entries(matches)
   699→      .map(([l, r]) => `${l}-${r}`)
   700→      .join(', ');
   701→    onAnswer(answer);
   702→  };
   703→
   704→  const allMatched = pairs.every((p) => matches[p.left]);
   705→  const usedRights = new Set(Object.values(matches));
   706→
   707→  return (
   708→    <div className="space-y-4">
   709→      <p className="text-sm text-muted-foreground">
   710→        Drag left items to their match on the right, or click to select. Click a connected left item to remove it.
   711→      </p>
   712→      <div ref={containerRef} className="relative grid grid-cols-2 gap-4">
   713→        {/* SVG Connector Overlay */}
   714→        <svg className="absolute inset-0 pointer-events-none z-10" style={{ width: '100%', height: '100%', overflow: 'visible' }}>
   715→          <defs>
   716→            <linearGradient id="connectorGrad" x1="0%" y1="0%" x2="100%" y2="0%">
   717→              <stop offset="0%" stopColor="#10b981" />
   718→              <stop offset="50%" stopColor="#14b8a6" />
   719→              <stop offset="100%" stopColor="#0d9488" />
   720→            </linearGradient>
   721→          </defs>
   722→          <AnimatePresence>
   723→            {linePositions.map((pos) => {
   724→              const midX = (pos.x1 + pos.x2) / 2;
   725→              return (
   726→                <motion.path
   727→                  key={pos.id}
   728→                  d={`M ${pos.x1} ${pos.y1} C ${midX} ${pos.y1}, ${midX} ${pos.y2}, ${pos.x2} ${pos.y2}`}
   729→                  fill="none"
   730→                  stroke="url(#connectorGrad)"
   731→                  strokeWidth="2.5"
   732→                  strokeLinecap="round"
   733→                  initial={{ pathLength: 0, opacity: 0 }}
   734→                  animate={{ pathLength: 1, opacity: 0.85 }}
   735→                  exit={{ pathLength: 0, opacity: 0 }}
   736→                  transition={{ duration: 0.5, ease: 'easeOut' }}
   737→                />
   738→              );
   739→            })}
   740→          </AnimatePresence>
   741→        </svg>
   742→
   743→        {/* Left column */}
   744→        <div className="flex flex-col gap-2">
   745→          {pairs.map((p) => {
   746→            const isMatched = !!matches[p.left];
   747→            const isDragging = dragItem === p.left;
   748→            return (
   749→              <motion.button
   750→                key={p.left}
   751→                data-left-idx={leftIdxMap[p.left]}
   752→                draggable={!isMatched}
   753→                whileTap={{ scale: 0.98 }}
   754→                onClick={() => handleLeftClick(p.left)}
   755→                onDragStart={(e) => handleDragStart(e, p.left)}
   756→                onDragEnd={handleDragEnd}
   757→                className={`rounded-lg border p-3 text-left text-sm font-medium transition-all ${
   758→                  isDragging
   759→                    ? 'border-primary/50 bg-primary/5 opacity-50 scale-95'
   760→                    : selectedLeft === p.left
   761→                      ? 'border-primary bg-primary/10 text-primary ring-1 ring-primary/30 shadow-sm shadow-primary/10'
   762→                      : isMatched
   763→                        ? 'border-emerald-500/50 bg-emerald-500/5 text-emerald-700 dark:text-emerald-400 shadow-sm shadow-emerald-500/5'
   764→                        : 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-grab active:cursor-grabbing'
   765→                }`}
   766→              >
   767→                <div className="flex items-center gap-2">
   768→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40 shrink-0" />
   769→                  <span className="flex-1">{p.left}</span>
   770→                  {isMatched && (
   771→                    <motion.span
   772→                      initial={{ opacity: 0, x: 10 }}
   773→                      animate={{ opacity: 1, x: 0 }}
   774→                      className="text-xs opacity-70 truncate max-w-[120px] text-emerald-600 dark:text-emerald-400"
   775→                    >
   776→                      {matches[p.left]}
   777→                    </motion.span>
   778→                  )}
   779→                </div>
   780→              </motion.button>
   781→            );
   782→          })}
   783→        </div>
   784→        {/* Right column */}
   785→        <div className="flex flex-col gap-2">
   786→          {shuffledRight.map((r) => {
   787→            const isUsed = usedRights.has(r);
   788→            const isDragOver = dragOverTarget === r && !isUsed;
   789→            return (
   790→              <motion.button
   791→                key={r}
   792→                data-right-idx={rightIdxMap[r]}
   793→                whileTap={{ scale: 0.98 }}
   794→                onClick={() => !isUsed && handleRightClick(r)}
   795→                onDragOver={(e) => handleDragOver(e, r)}
   796→                onDragLeave={handleDragLeave}
   797→                onDrop={(e) => handleDrop(e, r)}
   798→                disabled={isUsed}
   799→                className={`rounded-lg border p-3 text-left text-sm transition-all ${
   800→                  isDragOver
   801→                    ? 'border-emerald-500/60 bg-emerald-500/10 ring-2 ring-emerald-500/30 shadow-lg shadow-emerald-500/20 scale-[1.02]'
   802→                    : isUsed
   803→                      ? 'border-border bg-muted/50 text-muted-foreground line-through opacity-50'
   804→                      : selectedLeft
   805→                        ? 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-pointer hover:shadow-sm hover:shadow-primary/5'
   806→                        : 'border-border opacity-60'
   807→                }`}
   808→              >
   809→                <div className="flex items-center gap-2">
   810→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40 shrink-0" />
   811→                  <span className="flex-1">{r}</span>
   812→                </div>
   813→              </motion.button>
   814→            );
   815→          })}
   816→        </div>
   817→      </div>
   818→      <div className="flex items-center justify-between">
   819→        <div className="flex items-center gap-2">
   820→          <div className="relative h-1.5 w-24 rounded-full bg-muted/50 overflow-hidden">
   821→            <motion.div
   822→              className="absolute inset-y-0 left-0 rounded-full"
   823→              style={{ background: 'linear-gradient(90deg, oklch(0.627 0.194 149.214), oklch(0.687 0.159 177.89))' }}
   824→              animate={{ width: `${(Object.keys(matches).length / pairs.length) * 100}%` }}
   825→              transition={{ duration: 0.3 }}
   826→            />
   827→          </div>
   828→          <span className="text-xs text-muted-foreground">
   829→            {Object.keys(matches).length}/{pairs.length} matched
   830→          </span>
   831→        </div>
   832→        <motion.div
   833→          whileHover={{ scale: 1.05 }}
   834→          whileTap={{ scale: 0.95 }}
   835→        >
   836→          <Button onClick={handleSubmit} disabled={!allMatched} size="sm" className={allMatched ? 'glow-emerald' : ''}>
   837→            Submit Matches
   838→          </Button>
   839→        </motion.div>
   840→      </div>
   841→    </div>
   842→  );
