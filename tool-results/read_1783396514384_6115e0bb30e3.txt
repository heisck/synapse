    34→    34→function levenshteinDistance(a: string, b: string): number {
    35→    35→  const matrix: number[][] = [];
    36→    36→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    37→    37→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    38→    38→  for (let i = 1; i <= b.length; i++) {
    39→    39→    for (let j = 1; j <= a.length; j++) {
    40→    40→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    41→    41→      matrix[i][j] = Math.min(
    42→    42→        matrix[i - 1][j] + 1,
    43→    43→        matrix[i][j - 1] + 1,
    44→    44→        matrix[i - 1][j - 1] + cost,
    45→    45→      );
    46→    46→    }
    47→    47→  }
    48→    48→  return matrix[b.length][a.length];
    49→    49→}

    51→    51→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    52→    52→  const ua = userAnswer.trim().toLowerCase();
    53→    53→  const ca = correctAnswer.trim().toLowerCase();
    54→    54→  if (ua === ca) return true;
    55→    55→  if (ua.includes(ca) || ca.includes(ua)) return true;
    56→    56→  // Levenshtein distance for typo tolerance
    57→    57→  const dist = levenshteinDistance(ua, ca);
    58→    58→  // Allow 1 typo for short answers, 2 for longer ones
    59→    59→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    60→    60→  if (dist <= threshold) return true;
    61→    61→  // Word-level check: all words of correct answer present in user answer (in any order)
    62→    62→  const correctWords = ca.split(/\s+/);
    63→    63→  const userWords = ua.split(/\s+/);
    64→    64→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    65→    65→  return false;
    66→    66→}

   204→   204→// ---------- Animated counter hook ----------
   205→   205→function useAnimatedCounter(target: number, duration: number = 1200) {
   206→   206→  const [count, setCount] = useState(0);
   207→   207→  const rafRef = useRef<number>(0);
   208→   208→  useEffect(() => {
   209→   209→    const start = performance.now();
   210→   210→    const tick = (now: number) => {
   211→   211→      const elapsed = now - start;
   212→   212→      const progress = Math.min(elapsed / duration, 1);
   213→   213→      const eased = 1 - Math.pow(1 - progress, 3);
   214→   214→      setCount(Math.round(eased * target));
   215→   215→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   216→   216→    };
   217→   217→    rafRef.current = requestAnimationFrame(tick);
   218→   218→    return () => cancelAnimationFrame(rafRef.current);
   219→   219→  }, [target, duration]);
   220→   220→  return count;
   221→   221→}

   280→   280→  onAnswer: (answer: string) => void;

   283→   283→    const arr = [...pairs.map((p) => p.right)];
   284→   284→    for (let i = arr.length - 1; i > 0; i--) {
   285→   285→      const j = Math.floor(Math.random() * (i + 1));
   286→   286→      [arr[i], arr[j]] = [arr[j], arr[i]];
   287→   287→    }
   288→   288→    return arr;
   289→   289→  }, [pairs]);

   291→   291→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);

   294→   294→  const handleLeftClick = (left: string) => {
   295→   295→    setSelectedLeft(left);
   296→   296→  };
   297→   297→
   298→   298→  const handleRightClick = (right: string) => {
   299→   299→    if (!selectedLeft) return;
   300→   300→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   301→   301→    setSelectedLeft(null);

   306→   306→      const next = { ...prev };
