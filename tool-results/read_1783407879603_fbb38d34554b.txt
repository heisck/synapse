    48тЖТ    48тЖТfunction levenshteinDistance(a: string, b: string): number {
    49тЖТ    49тЖТ  const matrix: number[][] = [];
    50тЖТ    50тЖТ  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    51тЖТ    51тЖТ  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    52тЖТ    52тЖТ  for (let i = 1; i <= b.length; i++) {
    53тЖТ    53тЖТ    for (let j = 1; j <= a.length; j++) {
    54тЖТ    54тЖТ      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    55тЖТ    55тЖТ      matrix[i][j] = Math.min(
    56тЖТ    56тЖТ        matrix[i - 1][j] + 1,
    57тЖТ    57тЖТ        matrix[i][j - 1] + 1,
    58тЖТ    58тЖТ        matrix[i - 1][j - 1] + cost,
    59тЖТ    59тЖТ      );
    60тЖТ    60тЖТ    }
    61тЖТ    61тЖТ  }
    62тЖТ    62тЖТ  return matrix[b.length][a.length];
    63тЖТ    63тЖТ}

    65тЖТ    65тЖТfunction isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    66тЖТ    66тЖТ  const ua = userAnswer.trim().toLowerCase();
    67тЖТ    67тЖТ  const ca = correctAnswer.trim().toLowerCase();
    68тЖТ    68тЖТ  if (ua === ca) return true;
    69тЖТ    69тЖТ  if (ua.includes(ca) || ca.includes(ua)) return true;
    70тЖТ    70тЖТ  // Levenshtein distance for typo tolerance
    71тЖТ    71тЖТ  const dist = levenshteinDistance(ua, ca);
    72тЖТ    72тЖТ  // Allow 1 typo for short answers, 2 for longer ones
    73тЖТ    73тЖТ  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    74тЖТ    74тЖТ  if (dist <= threshold) return true;
    75тЖТ    75тЖТ  // Word-level check: all words of correct answer present in user answer (in any order)
    76тЖТ    76тЖТ  const correctWords = ca.split(/\s+/);
    77тЖТ    77тЖТ  const userWords = ua.split(/\s+/);
    78тЖТ    78тЖТ  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    79тЖТ    79тЖТ  return false;
    80тЖТ    80тЖТ}

   344тЖТ   344тЖТ
   345тЖТ   345тЖТ  // 2. Spaced repetition due for review
   346тЖТ   346тЖТ  if (dueConcepts.has(concept)) {
   347тЖТ   347тЖТ    score += 2;
   348тЖТ   348тЖТ    reasons.push('needs review');
   349тЖТ   349тЖТ  }
   350тЖТ   350тЖТ
   351тЖТ   351тЖТ  // 3. Recent performance: last 5 results on concept
   352тЖТ   352тЖТ  const conceptResults = adaptiveResults
   353тЖТ   353тЖТ    .filter((r) => r.concept === concept)
   354тЖТ   354тЖТ    .slice(-5);
   355тЖТ   355тЖТ  const last3 = conceptResults.slice(-3);
   356тЖТ   356тЖТ  const wrongCount = last3.filter((r) => !r.correct).length;
   357тЖТ   357тЖТ  const correctCount = conceptResults.filter((r) => r.correct).length;
   358тЖТ   358тЖТ
   359тЖТ   359тЖТ  if (wrongCount >= 2) {
   360тЖТ   360тЖТ    score += 2;
   361тЖТ   361тЖТ    reasons.push('recent struggles');
   362тЖТ   362тЖТ  }
   363тЖТ   363тЖТ
   364тЖТ   364тЖТ  // 4. Difficulty match bonus
   365тЖТ   365тЖТ  const targetLevel = mastery ? Math.min(mastery.level + 1, 3) : 1;
   366тЖТ   366тЖТ  const targetDifficulty = targetLevel === 1 ? 'easy' : targetLevel === 2 ? 'medium' : 'hard';
   367тЖТ   367тЖТ  if (question.difficulty === targetDifficulty) {
   368тЖТ   368тЖТ    score += 1;
   369тЖТ   369тЖТ  } else if (
   370тЖТ   370тЖТ    (targetDifficulty === 'medium' && question.difficulty === 'easy') ||
   371тЖТ   371тЖТ    (targetDifficulty === 'hard' && question.difficulty === 'medium')
   372тЖТ   372тЖТ  ) {
   373тЖТ   373тЖТ    score += 0.5;
   374тЖТ   374тЖТ  }
   375тЖТ   375тЖТ
   376тЖТ   376тЖТ  // 5. Boost for concepts with many wrong answers (weaker areas)
   377тЖТ   377тЖТ  if (correctCount >= 3) {
   378тЖТ   378тЖТ    score += 1;
   379тЖТ   379тЖТ    reasons.push('ready for harder questions');
   380тЖТ   380тЖТ  }
   381тЖТ   381тЖТ
   382тЖТ   382тЖТ  return { question, score, reasons };
   383тЖТ   383тЖТ}

   385тЖТ   385тЖТfunction getAdaptiveReasoning(scored: ScoredQuestion[]): string {
   386тЖТ   386тЖТ  if (scored.length === 0) return '';
   387тЖТ   387тЖТ
   388тЖТ   388тЖТ  const conceptReasons = new Map<string, string[]>();
   389тЖТ   389тЖТ  for (const sq of scored.slice(0, 5)) {
   390тЖТ   390тЖТ    const concept = sq.question.concept || 'Unknown';
   391тЖТ   391тЖТ    const existing = conceptReasons.get(concept) || [];
   392тЖТ   392тЖТ    conceptReasons.set(concept, [...existing, ...sq.reasons]);
   393тЖТ   393тЖТ  }
   394тЖТ   394тЖТ
   395тЖТ   395тЖТ  const parts: string[] = [];
   396тЖТ   396тЖТ  for (const [concept, reasons] of conceptReasons) {
   397тЖТ   397тЖТ    const uniqueReasons = [...new Set(reasons)];
   398тЖТ   398тЖТ    const desc = uniqueReasons.map((r) => {
   399тЖТ   399тЖТ      switch (r) {
   400тЖТ   400тЖТ        case 'low mastery': return 'low mastery';
   401тЖТ   401тЖТ        case 'needs review': return 'needs review';
   402тЖТ   402тЖТ        case 'recent struggles': return 'struggling recently';
   403тЖТ   403тЖТ        case 'ready for harder questions': return 'ready for challenge';
   404тЖТ   404тЖТ        default: return r;
   405тЖТ   405тЖТ      }
   406тЖТ   406тЖТ    }).join(', ');
   407тЖТ   407тЖТ    parts.push(`${concept} (${desc})`);
   408тЖТ   408тЖТ  }
   409тЖТ   409тЖТ
   410тЖТ   410тЖТ  return 'Focusing on: ' + parts.join(', ');
   411тЖТ   411тЖТ}

   478тЖТ   478тЖТfunction useAnimatedCounter(target: number, duration: number = 1200) {
   479тЖТ   479тЖТ  const [count, setCount] = useState(0);
   480тЖТ   480тЖТ  const rafRef = useRef<number>(0);
   481тЖТ   481тЖТ  useEffect(() => {
   482тЖТ   482тЖТ    const start = performance.now();
   483тЖТ   483тЖТ    const tick = (now: number) => {
   484тЖТ   484тЖТ      const elapsed = now - start;
   485тЖТ   485тЖТ      const progress = Math.min(elapsed / duration, 1);
   486тЖТ   486тЖТ      const eased = 1 - Math.pow(1 - progress, 3);
   487тЖТ   487тЖТ      setCount(Math.round(eased * target));
   488тЖТ   488тЖТ      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   489тЖТ   489тЖТ    };
   490тЖТ   490тЖТ    rafRef.current = requestAnimationFrame(tick);
   491тЖТ   491тЖТ    return () => cancelAnimationFrame(rafRef.current);
   492тЖТ   492тЖТ  }, [target, duration]);
   493тЖТ   493тЖТ  return count;
   494тЖТ   494тЖТ}

   548тЖТ   548тЖТfunction MatchingInput({
   549тЖТ   549тЖТ  pairs,
   550тЖТ   550тЖТ  onAnswer,
   551тЖТ   551тЖТ}: {
   552тЖТ   552тЖТ  pairs: Array<{ left: string; right: string }>;
   553тЖТ   553тЖТ  onAnswer: (answer: string) => void;
   554тЖТ   554тЖТ}) {
   555тЖТ   555тЖТ  const shuffledRight = useMemo(() => {
   556тЖТ   556тЖТ    const arr = [...pairs.map((p) => p.right)];
   557тЖТ   557тЖТ    for (let i = arr.length - 1; i > 0; i--) {
   558тЖТ   558тЖТ      const j = Math.floor(Math.random() * (i + 1));
   559тЖТ   559тЖТ      [arr[i], arr[j]] = [arr[j], arr[i]];
   560тЖТ   560тЖТ    }
   561тЖТ   561тЖТ    return arr;
   562тЖТ   562тЖТ  }, [pairs]);
   563тЖТ   563тЖТ
   564тЖТ   564тЖТ  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   565тЖТ   565тЖТ  const [matches, setMatches] = useState<Record<string, string>>({});
   566тЖТ   566тЖТ  const [dragItem, setDragItem] = useState<string | null>(null);
   567тЖТ   567тЖТ  const [dragOverTarget, setDragOverTarget] = useState<string | null>(null);
   568тЖТ   568тЖТ  const [linePositions, setLinePositions] = useState<Array<{ id: string; x1: number; y1: number; x2: number; y2: number }>>([]);
   569тЖТ   569тЖТ  const containerRef = useRef<HTMLDivElement>(null);
   570тЖТ   570тЖТ
   571тЖТ   571тЖТ  // Build index maps for safe DOM queries
   572тЖТ   572тЖТ  const leftIdxMap = useMemo(() => {
   573тЖТ   573тЖТ    const map: Record<string, number> = {};
   574тЖТ   574тЖТ    pairs.forEach((p, i) => { map[p.left] = i; });
   575тЖТ   575тЖТ    return map;
   576тЖТ   576тЖТ  }, [pairs]);
   577тЖТ   577тЖТ
   578тЖТ   578тЖТ  const rightIdxMap = useMemo(() => {
   579тЖТ   579тЖТ    const map: Record<string, number> = {};
   580тЖТ   580тЖТ    shuffledRight.forEach((r, i) => { map[r] = i; });
   581тЖТ   581тЖТ    return map;
   582тЖТ   582тЖТ  }, [shuffledRight]);
   583тЖТ   583тЖТ
   584тЖТ   584тЖТ  // Compute SVG connector line positions

   587тЖТ   587тЖТ    if (!container) return;
   588тЖТ   588тЖТ
   589тЖТ   589тЖТ    requestAnimationFrame(() => {
   590тЖТ   590тЖТ      if (!container) return;
   591тЖТ   591тЖТ      const containerRect = container.getBoundingClientRect();
   592тЖТ   592тЖТ      const positions: Array<{ id: string; x1: number; y1: number; x2: number; y2: number }> = [];
   593тЖТ   593тЖТ
   594тЖТ   594тЖТ      for (const [left, right] of Object.entries(matches)) {
   595тЖТ   595тЖТ        const lIdx = leftIdxMap[left];
   596тЖТ   596тЖТ        const rIdx = rightIdxMap[right];
   597тЖТ   597тЖТ        if (lIdx === undefined || rIdx === undefined) continue;
   598тЖТ   598тЖТ
   599тЖТ   599тЖТ        const leftEl = container.querySelector(`[data-left-idx="${lIdx}"]`);
   600тЖТ   600тЖТ        const rightEl = container.querySelector(`[data-right-idx="${rIdx}"]`);
   601тЖТ   601тЖТ
   602тЖТ   602тЖТ        if (leftEl && rightEl) {

   604тЖТ   604тЖТ          const rightRect = rightEl.getBoundingClientRect();
   605тЖТ   605тЖТ
   606тЖТ   606тЖТ          positions.push({
   607тЖТ   607тЖТ            id: `${left}-${right}`,
   608тЖТ   608тЖТ            x1: leftRect.right - containerRect.left,
   609тЖТ   609тЖТ            y1: leftRect.top + leftRect.height / 2 - containerRect.top,
   610тЖТ   610тЖТ            x2: rightRect.left - containerRect.left,
   611тЖТ   611тЖТ            y2: rightRect.top + rightRect.height / 2 - containerRect.top,
   612тЖТ   612тЖТ          });
   613тЖТ   613тЖТ        }
   614тЖТ   614тЖТ      }
   615тЖТ   615тЖТ
   616тЖТ   616тЖТ      setLinePositions(positions);
   617тЖТ   617тЖТ    });
   618тЖТ   618тЖТ  }, [matches, leftIdxMap, rightIdxMap]);
   619тЖТ   619тЖТ
   620тЖТ   620тЖТ  // Recalculate on matches change
   621тЖТ   621тЖТ  useEffect(() => {
   622тЖТ   622тЖТ    updateLinePositions();
   623тЖТ   623тЖТ  }, [updateLinePositions]);
   624тЖТ   624тЖТ
   625тЖТ   625тЖТ  // Recalculate on resize
   626тЖТ   626тЖТ  useEffect(() => {
   627тЖТ   627тЖТ    const container = containerRef.current;
   628тЖТ   628тЖТ    if (!container) return;
   629тЖТ   629тЖТ
   630тЖТ   630тЖТ    const observer = new ResizeObserver(updateLinePositions);
   631тЖТ   631тЖТ    observer.observe(container);
   632тЖТ   632тЖТ
   633тЖТ   633тЖТ    return () => observer.disconnect();
   634тЖТ   634тЖТ  }, [updateLinePositions]);
   635тЖТ   635тЖТ
   636тЖТ   636тЖТ  const handleLeftClick = (left: string) => {
   637тЖТ   637тЖТ    if (matches[left]) {
   638тЖТ   638тЖТ      // Remove match by clicking connected left item
   639тЖТ   639тЖТ      setMatches((prev) => {
   640тЖТ   640тЖТ        const next = { ...prev };
   641тЖТ   641тЖТ        delete next[left];
   642тЖТ   642тЖТ        return next;
   643тЖТ   643тЖТ      });
   644тЖТ   644тЖТ      return;
   645тЖТ   645тЖТ    }
   646тЖТ   646тЖТ    setSelectedLeft(left);
   647тЖТ   647тЖТ  };
   648тЖТ   648тЖТ
   649тЖТ   649тЖТ  const handleRightClick = (right: string) => {
   650тЖТ   650тЖТ    if (!selectedLeft) return;
   651тЖТ   651тЖТ    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   652тЖТ   652тЖТ    setSelectedLeft(null);
   653тЖТ   653тЖТ  };
   654тЖТ   654тЖТ
   655тЖТ   655тЖТ  const handleDragStart = (e: React.DragEvent, left: string) => {
   656тЖТ   656тЖТ    setDragItem(left);
   657тЖТ   657тЖТ    e.dataTransfer.setData('text/plain', left);
   658тЖТ   658тЖТ    e.dataTransfer.effectAllowed = 'link';
   659тЖТ   659тЖТ  };
   660тЖТ   660тЖТ
   661тЖТ   661тЖТ  const handleDragOver = (e: React.DragEvent, right: string) => {
   662тЖТ   662тЖТ    e.preventDefault();
   663тЖТ   663тЖТ    e.dataTransfer.dropEffect = 'link';
   664тЖТ   664тЖТ    const isUsedByOther = Object.entries(matches).some(
   665тЖТ   665тЖТ      ([k, v]) => v === right && k !== dragItem,
   666тЖТ   666тЖТ    );
   667тЖТ   667тЖТ    if (dragItem && !isUsedByOther) {
   668тЖТ   668тЖТ      setDragOverTarget(right);
   669тЖТ   669тЖТ    }
   670тЖТ   670тЖТ  };
   671тЖТ   671тЖТ
   672тЖТ   672тЖТ  const handleDragLeave = () => {
   673тЖТ   673тЖТ    setDragOverTarget(null);

   676тЖТ   676тЖТ  const handleDrop = (e: React.DragEvent, right: string) => {
   677тЖТ   677тЖТ    e.preventDefault();
   678тЖТ   678тЖТ    const left = e.dataTransfer.getData('text/plain') || dragItem;
