    48→// ---------- Levenshtein distance (standard DP) ----------
    49→function levenshtein(a: string, b: string): number {
    50→  const m = a.length;
    51→  const n = b.length;
    52→  const dp: number[][] = Array.from({ length: n + 1 }, () => new Array(m + 1).fill(0) as number[]);
    53→  for (let i = 0; i <= n; i++) dp[i][0] = i;
    54→  for (let j = 0; j <= m; j++) dp[0][j] = j;
    55→  for (let i = 1; i <= n; i++) {
    56→    for (let j = 1; j <= m; j++) {
    57→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    58→      dp[i][j] = Math.min(
    59→        dp[i - 1][j] + 1,
    60→        dp[i][j - 1] + 1,
    61→        dp[i - 1][j - 1] + cost,
    62→      );
    63→    }
    64→  }
    65→  return dp[n][m];
    66→}

    68→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    69→  const ua = userAnswer.trim().toLowerCase();
    70→  const ca = correctAnswer.trim().toLowerCase();
    71→  if (ua === ca) return true;
    72→  if (ua.includes(ca) || ca.includes(ua)) return true;
    73→  // Levenshtein distance for typo tolerance
    74→  const dist = levenshtein(ua, ca);
    75→  // Allow 1 typo for short answers, 2 for longer ones
    76→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    77→  if (dist <= threshold) return true;
    78→  // Word-level check: all words of correct answer present in user answer (in any order)
    79→  const correctWords = ca.split(/\s+/);
    80→  const userWords = ua.split(/\s+/);
    81→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    82→  return false;
    83→}

    85→// ---------- Fill-in-blank grading with Levenshtein tolerance ----------
    86→// Case-insensitive, trimmed comparison with partial credit for typos:
    87→//   Exact match → 100% (or 75% if hint used)
    88→//   Levenshtein ≤ 2 → 80% (× 0.75 if hint)
    89→//   Levenshtein ≤ 3 → 50% (× 0.75 if hint)
    90→//   Otherwise → 0
    91→type FillBlankGrade = { status: 'correct' | 'close' | 'wrong'; points: number; message: string };
    92→
    93→function gradeFillBlank(userAnswer: string, correctAnswer: string, hintUsed: boolean = false): FillBlankGrade {
    94→  const ua = userAnswer.trim().toLowerCase();
    95→  const ca = correctAnswer.trim().toLowerCase();
    96→  if (ua === ca) {
    97→    return { status: 'correct', points: hintUsed ? 0.75 : 1, message: '' };
    98→  }
    99→  const dist = levenshtein(ua, ca);
   100→  let basePoints = 0;
   101→  if (dist <= 2) basePoints = 0.8;
   102→  else if (dist <= 3) basePoints = 0.5;
   103→  if (basePoints > 0) {
   104→    return { status: 'close', points: basePoints * (hintUsed ? 0.75 : 1), message: `Close! Did you mean: "${correctAnswer}"?` };
   105→  }
   106→  return { status: 'wrong', points: 0, message: '' };
   107→}

   275→  const streak = loadDailyStreak();
   276→  const today = getTodayStr();
   277→
   278→  if (streak.lastDate === today) return streak;
   279→
   280→  const yesterday = new Date();
   281→  yesterday.setDate(yesterday.getDate() - 1);
   282→  const yesterdayStr = yesterday.toISOString().split('T')[0];
   283→
   284→  let newCurrent: number;
   285→  if (streak.lastDate === yesterdayStr) {
   286→    newCurrent = streak.current + 1;
   287→  } else {
   288→    newCurrent = 1;
   289→  }
   290→
   291→  const newBest = Math.max(streak.best, newCurrent);
   292→  const updated = { current: newCurrent, best: newBest, lastDate: today };
   293→  saveDailyStreak(updated);
   294→  return updated;
   295→}

   297→function getTimeUntilMidnight(): { hours: number; minutes: number; seconds: number } {
   298→  const now = new Date();
   299→  const midnight = new Date(now);
   300→  midnight.setHours(24, 0, 0, 0);
   301→  const diff = midnight.getTime() - now.getTime();
   302→  return {
   303→    hours: Math.floor(diff / (1000 * 60 * 60)),
   304→    minutes: Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60)),
   305→    seconds: Math.floor((diff % (1000 * 60)) / 1000),
   306→  };
   307→}

   359→function scoreQuestionForAdaptive(input: AdaptiveScoringInput): ScoredQuestion {
   360→  const { question, masteryMap, adaptiveResults, dueConcepts } = input;
   361→  let score = 0;
   362→  const reasons: string[] = [];
   363→  const concept = question.concept || 'Unknown';
   364→  const mastery = masteryMap[concept];
   365→
   366→  // 1. Low mastery bonus (level 1-2)
   367→  if (mastery && mastery.level < 3) {
   368→    score += 3;
   369→    reasons.push('low mastery');
   370→  }
   371→
   372→  // 2. Spaced repetition due for review
   373→  if (dueConcepts.has(concept)) {
   374→    score += 2;
   375→    reasons.push('needs review');
   376→  }
   377→
   378→  // 3. Recent performance: last 5 results on concept
   379→  const conceptResults = adaptiveResults
   380→    .filter((r) => r.concept === concept)
   381→    .slice(-5);
   382→  const last3 = conceptResults.slice(-3);
   383→  const wrongCount = last3.filter((r) => !r.correct).length;
   384→  const correctCount = conceptResults.filter((r) => r.correct).length;
   385→
   386→  if (wrongCount >= 2) {
   387→    score += 2;
   388→    reasons.push('recent struggles');
   389→  }
   390→
   391→  // 4. Difficulty match bonus
   392→  const targetLevel = mastery ? Math.min(mastery.level + 1, 3) : 1;
   393→  const targetDifficulty = targetLevel === 1 ? 'easy' : targetLevel === 2 ? 'medium' : 'hard';
   394→  if (question.difficulty === targetDifficulty) {
   395→    score += 1;
   396→  } else if (
   397→    (targetDifficulty === 'medium' && question.difficulty === 'easy') ||
   398→    (targetDifficulty === 'hard' && question.difficulty === 'medium')
   399→  ) {
   400→    score += 0.5;
   401→  }
   402→
   403→  // 5. Boost for concepts with many wrong answers (weaker areas)
   404→  if (correctCount >= 3) {
   405→    score += 1;
   406→    reasons.push('ready for harder questions');
   407→  }
   408→
   409→  return { question, score, reasons };
   410→}

   412→function getAdaptiveReasoning(scored: ScoredQuestion[]): string {
   413→  if (scored.length === 0) return '';
   414→
   415→  const conceptReasons = new Map<string, string[]>();
   416→  for (const sq of scored.slice(0, 5)) {
   417→    const concept = sq.question.concept || 'Unknown';
   418→    const existing = conceptReasons.get(concept) || [];
   419→    conceptReasons.set(concept, [...existing, ...sq.reasons]);
   420→  }
   421→
   422→  const parts: string[] = [];
   423→  for (const [concept, reasons] of conceptReasons) {
   424→    const uniqueReasons = [...new Set(reasons)];
   425→    const desc = uniqueReasons.map((r) => {
   426→      switch (r) {
   427→        case 'low mastery': return 'low mastery';
   428→        case 'needs review': return 'needs review';
   429→        case 'recent struggles': return 'struggling recently';
   430→        case 'ready for harder questions': return 'ready for challenge';
   431→        default: return r;
   432→      }
   433→    }).join(', ');
   434→    parts.push(`${concept} (${desc})`);
   435→  }
   436→
   437→  return 'Focusing on: ' + parts.join(', ');
   438→}

   504→// ---------- Animated counter hook ----------
   505→function useAnimatedCounter(target: number, duration: number = 1200) {
   506→  const [count, setCount] = useState(0);
   507→  const rafRef = useRef<number>(0);
   508→  useEffect(() => {
   509→    const start = performance.now();
   510→    const tick = (now: number) => {
   511→      const elapsed = now - start;
   512→      const progress = Math.min(elapsed / duration, 1);
   513→      const eased = 1 - Math.pow(1 - progress, 3);
   514→      setCount(Math.round(eased * target));
   515→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   516→    };
   517→    rafRef.current = requestAnimationFrame(tick);
   518→    return () => cancelAnimationFrame(rafRef.current);
   519→  }, [target, duration]);
   520→  return count;
   521→}

   574→// ---------- Matching question component (drag-and-drop with SVG connectors) ----------
   575→function MatchingInput({
   576→  pairs,
   577→  onAnswer,
   578→}: {
   579→  pairs: Array<{ left: string; right: string }>;
   580→  onAnswer: (answer: string) => void;
   581→}) {
   582→  const shuffledRight = useMemo(() => {
   583→    const arr = [...pairs.map((p) => p.right)];
   584→    for (let i = arr.length - 1; i > 0; i--) {
   585→      const j = Math.floor(Math.random() * (i + 1));
   586→      [arr[i], arr[j]] = [arr[j], arr[i]];
   587→    }
   588→    return arr;
   589→  }, [pairs]);
   590→
   591→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   592→  const [matches, setMatches] = useState<Record<string, string>>({});
   593→  const [dragItem, setDragItem] = useState<string | null>(null);
   594→  const [dragOverTarget, setDragOverTarget] = useState<string | null>(null);
   595→  const [linePositions, setLinePositions] = useState<Array<{ id: string; x1: number; y1: number; x2: number; y2: number }>>([]);
   596→  const containerRef = useRef<HTMLDivElement>(null);
   597→
   598→  // Build index maps for safe DOM queries
   599→  const leftIdxMap = useMemo(() => {
   600→    const map: Record<string, number> = {};
   601→    pairs.forEach((p, i) => { map[p.left] = i; });
   602→    return map;
   603→  }, [pairs]);
   604→
   605→  const rightIdxMap = useMemo(() => {
   606→    const map: Record<string, number> = {};
   607→    shuffledRight.forEach((r, i) => { map[r] = i; });
   608→    return map;
   609→  }, [shuffledRight]);
   610→
   611→  // Compute SVG connector line positions
   612→  const updateLinePositions = useCallback(() => {
   613→    const container = containerRef.current;
   614→    if (!container) return;
   615→
   616→    requestAnimationFrame(() => {
   617→      if (!container) return;
   618→      const containerRect = container.getBoundingClientRect();
   619→      const positions: Array<{ id: string; x1: number; y1: number; x2: number; y2: number }> = [];
   620→
   621→      for (const [left, right] of Object.entries(matches)) {
   622→        const lIdx = leftIdxMap[left];
   623→        const rIdx = rightIdxMap[right];
   624→        if (lIdx === undefined || rIdx === undefined) continue;
   625→
   626→        const leftEl = container.querySelector(`[data-left-idx="${lIdx}"]`);
   627→        const rightEl = container.querySelector(`[data-right-idx="${rIdx}"]`);
   628→
   629→        if (leftEl && rightEl) {
   630→          const leftRect = leftEl.getBoundingClientRect();
   631→          const rightRect = rightEl.getBoundingClientRect();
   632→
   633→          positions.push({
   634→            id: `${left}-${right}`,
   635→            x1: leftRect.right - containerRect.left,
   636→            y1: leftRect.top + leftRect.height / 2 - containerRect.top,
   637→            x2: rightRect.left - containerRect.left,
   638→            y2: rightRect.top + rightRect.height / 2 - containerRect.top,
   639→          });
   640→        }
   641→      }
   642→
   643→      setLinePositions(positions);
   644→    });
   645→  }, [matches, leftIdxMap, rightIdxMap]);
   646→
   647→  // Recalculate on matches change
   648→  useEffect(() => {
   649→    updateLinePositions();
   650→  }, [updateLinePositions]);
   651→
   652→  // Recalculate on resize
   653→  useEffect(() => {
   654→    const container = containerRef.current;
   655→    if (!container) return;
   656→
   657→    const observer = new ResizeObserver(updateLinePositions);
   658→    observer.observe(container);
   659→
   660→    return () => observer.disconnect();
   661→  }, [updateLinePositions]);
   662→
   663→  const handleLeftClick = (left: string) => {
   664→    if (matches[left]) {
   665→      // Remove match by clicking connected left item
   666→      setMatches((prev) => {
   667→        const next = { ...prev };
   668→        delete next[left];
   669→        return next;
   670→      });
   671→      return;
   672→    }
   673→    setSelectedLeft(left);
   674→  };
   675→
   676→  const handleRightClick = (right: string) => {
   677→    if (!selectedLeft) return;
   678→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   679→    setSelectedLeft(null);
   680→  };
   681→
   682→  const handleDragStart = (e: React.DragEvent, left: string) => {
   683→    setDragItem(left);
   684→    e.dataTransfer.setData('text/plain', left);
   685→    e.dataTransfer.effectAllowed = 'link';
   686→  };
   687→
   688→  const handleDragOver = (e: React.DragEvent, right: string) => {
   689→    e.preventDefault();
   690→    e.dataTransfer.dropEffect = 'link';
   691→    const isUsedByOther = Object.entries(matches).some(
   692→      ([k, v]) => v === right && k !== dragItem,
   693→    );
   694→    if (dragItem && !isUsedByOther) {
   695→      setDragOverTarget(right);
   696→    }
   697→  };
   698→
   699→  const handleDragLeave = () => {
   700→    setDragOverTarget(null);
   701→  };
   702→
   703→  const handleDrop = (e: React.DragEvent, right: string) => {
   704→    e.preventDefault();
   705→    const left = e.dataTransfer.getData('text/plain') || dragItem;
   706→    if (left) {
   707→      // Remove any existing match for this left item first
   708→      setMatches((prev) => {
   709→        const next = { ...prev };
   710→        delete next[left];
   711→        next[left] = right;
   712→        return next;
   713→      });
   714→    }
   715→    setDragItem(null);
   716→    setDragOverTarget(null);
   717→  };
   718→
   719→  const handleDragEnd = () => {
   720→    setDragItem(null);
   721→    setDragOverTarget(null);
   722→  };
   723→
   724→  const handleSubmit = () => {
   725→    const answer = Object.entries(matches)
   726→      .map(([l, r]) => `${l}-${r}`)
   727→      .join(', ');
   728→    onAnswer(answer);
   729→  };
   730→
   731→  const allMatched = pairs.every((p) => matches[p.left]);
   732→  const usedRights = new Set(Object.values(matches));
   733→
   734→  return (
   735→    <div className="space-y-4">
   736→      <p className="text-sm text-muted-foreground">
   737→        Drag left items to their match on the right, or click to select. Click a connected left item to remove it.
   738→      </p>
   739→      <div ref={containerRef} className="relative grid grid-cols-2 gap-4">
   740→        {/* SVG Connector Overlay */}
   741→        <svg className="absolute inset-0 pointer-events-none z-10" style={{ width: '100%', height: '100%', overflow: 'visible' }}>
   742→          <defs>
   743→            <linearGradient id="connectorGrad" x1="0%" y1="0%" x2="100%" y2="0%">
   744→              <stop offset="0%" stopColor="#10b981" />
   745→              <stop offset="50%" stopColor="#14b8a6" />
   746→              <stop offset="100%" stopColor="#0d9488" />
   747→            </linearGradient>
   748→          </defs>
   749→          <AnimatePresence>
   750→            {linePositions.map((pos) => {
   751→              const midX = (pos.x1 + pos.x2) / 2;
   752→              return (
   753→                <motion.path
   754→                  key={pos.id}
   755→                  d={`M ${pos.x1} ${pos.y1} C ${midX} ${pos.y1}, ${midX} ${pos.y2}, ${pos.x2} ${pos.y2}`}
   756→                  fill="none"
   757→                  stroke="url(#connectorGrad)"
   758→                  strokeWidth="2.5"
   759→                  strokeLinecap="round"
   760→                  initial={{ pathLength: 0, opacity: 0 }}
   761→                  animate={{ pathLength: 1, opacity: 0.85 }}
   762→                  exit={{ pathLength: 0, opacity: 0 }}
   763→                  transition={{ duration: 0.5, ease: 'easeOut' }}
   764→                />
   765→              );
   766→            })}
   767→          </AnimatePresence>
   768→        </svg>
   769→
   770→        {/* Left column */}
   771→        <div className="flex flex-col gap-2">
   772→          {pairs.map((p) => {
   773→            const isMatched = !!matches[p.left];
   774→            const isDragging = dragItem === p.left;
   775→            return (
   776→              <motion.button
   777→                key={p.left}
   778→                data-left-idx={leftIdxMap[p.left]}
   779→                draggable={!isMatched}
   780→                whileTap={{ scale: 0.98 }}
   781→                onClick={() => handleLeftClick(p.left)}
   782→                onDragStart={(e) => handleDragStart(e, p.left)}
   783→                onDragEnd={handleDragEnd}
   784→                className={`rounded-lg border p-3 text-left text-sm font-medium transition-all ${
   785→                  isDragging
   786→                    ? 'border-primary/50 bg-primary/5 opacity-50 scale-95'
   787→                    : selectedLeft === p.left
   788→                      ? 'border-primary bg-primary/10 text-primary ring-1 ring-primary/30 shadow-sm shadow-primary/10'
   789→                      : isMatched
   790→                        ? 'border-emerald-500/50 bg-emerald-500/5 text-emerald-700 dark:text-emerald-400 shadow-sm shadow-emerald-500/5'
   791→                        : 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-grab active:cursor-grabbing'
   792→                }`}
   793→              >
   794→                <div className="flex items-center gap-2">
   795→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40 shrink-0" />
   796→                  <span className="flex-1">{p.left}</span>
   797→                  {isMatched && (
   798→                    <motion.span
   799→                      initial={{ opacity: 0, x: 10 }}
   800→                      animate={{ opacity: 1, x: 0 }}
   801→                      className="text-xs opacity-70 truncate max-w-[120px] text-emerald-600 dark:text-emerald-400"
   802→                    >
   803→                      {matches[p.left]}
   804→                    </motion.span>
   805→                  )}
   806→                </div>
   807→              </motion.button>
   808→            );
   809→          })}
   810→        </div>
   811→        {/* Right column */}
   812→        <div className="flex flex-col gap-2">
   813→          {shuffledRight.map((r) => {
   814→            const isUsed = usedRights.has(r);
   815→            const isDragOver = dragOverTarget === r && !isUsed;
   816→            return (
   817→              <motion.button
   818→                key={r}
   819→                data-right-idx={rightIdxMap[r]}
   820→                whileTap={{ scale: 0.98 }}
   821→                onClick={() => !isUsed && handleRightClick(r)}
   822→                onDragOver={(e) => handleDragOver(e, r)}
   823→                onDragLeave={handleDragLeave}
   824→                onDrop={(e) => handleDrop(e, r)}
   825→                disabled={isUsed}
   826→                className={`rounded-lg border p-3 text-left text-sm transition-all ${
   827→                  isDragOver
   828→                    ? 'border-emerald-500/60 bg-emerald-500/10 ring-2 ring-emerald-500/30 shadow-lg shadow-emerald-500/20 scale-[1.02]'
   829→                    : isUsed
   830→                      ? 'border-border bg-muted/50 text-muted-foreground line-through opacity-50'
   831→                      : selectedLeft
   832→                        ? 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-pointer hover:shadow-sm hover:shadow-primary/5'
   833→                        : 'border-border opacity-60'
   834→                }`}
   835→              >
   836→                <div className="flex items-center gap-2">
   837→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40 shrink-0" />
   838→                  <span className="flex-1">{r}</span>
   839→                </div>
   840→              </motion.button>
   841→            );
   842→          })}
   843→        </div>
   844→      </div>
   845→      <div className="flex items-center justify-between">
   846→        <div className="flex items-center gap-2">
   847→          <div className="relative h-1.5 w-24 rounded-full bg-muted/50 overflow-hidden">
   848→            <motion.div
   849→              className="absolute inset-y-0 left-0 rounded-full"
   850→              style={{ background: 'linear-gradient(90deg, oklch(0.627 0.194 149.214), oklch(0.687 0.159 177.89))' }}
   851→              animate={{ width: `${(Object.keys(matches).length / pairs.length) * 100}%` }}
   852→              transition={{ duration: 0.3 }}
   853→            />
   854→          </div>
   855→          <span className="text-xs text-muted-foreground">
   856→            {Object.keys(matches).length}/{pairs.length} matched
   857→          </span>
   858→        </div>
   859→        <motion.div
   860→          whileHover={{ scale: 1.05 }}
   861→          whileTap={{ scale: 0.95 }}
   862→        >
   863→          <Button onClick={handleSubmit} disabled={!allMatched} size="sm" className={allMatched ? 'glow-emerald' : ''}>
   864→            Submit Matches
   865→          </Button>
   866→        </motion.div>
   867→      </div>
   868→    </div>
   869→  );
