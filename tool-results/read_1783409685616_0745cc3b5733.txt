   107тЖТ   107тЖТfunction formatRelativeTime(timestamp: number): string {
   108тЖТ   108тЖТ  const now = Date.now();
   109тЖТ   109тЖТ  const diff = now - timestamp;
   110тЖТ   110тЖТ  const minutes = Math.floor(diff / 60000);
   111тЖТ   111тЖТ  const hours = Math.floor(diff / 3600000);
   112тЖТ   112тЖТ  const days = Math.floor(diff / 86400000);
   113тЖТ   113тЖТ
   114тЖТ   114тЖТ  if (minutes < 1) return 'Just now';
   115тЖТ   115тЖТ  if (minutes < 60) return `${minutes}m ago`;
   116тЖТ   116тЖТ  if (hours < 24) return `${hours}h ago`;
   117тЖТ   117тЖТ  if (days === 1) return 'Yesterday';
   118тЖТ   118тЖТ  if (days < 7) return `${days}d ago`;
   119тЖТ   119тЖТ  return `${Math.floor(days / 7)}w ago`;
   120тЖТ   120тЖТ}

   165тЖТ   165тЖТfunction TypewriterText({ text, speed = 40, className = '' }: { text: string; speed?: number; className?: string }) {
   166тЖТ   166тЖТ  const [displayed, setDisplayed] = useState('');
   167тЖТ   167тЖТ  const [done, setDone] = useState(false);
   168тЖТ   168тЖТ
   169тЖТ   169тЖТ  useEffect(() => {
   170тЖТ   170тЖТ    setDisplayed('');
   171тЖТ   171тЖТ    setDone(false);
   172тЖТ   172тЖТ    let i = 0;
   173тЖТ   173тЖТ    const timer = setInterval(() => {
   174тЖТ   174тЖТ      i++;
   175тЖТ   175тЖТ      setDisplayed(text.slice(0, i));
   176тЖТ   176тЖТ      if (i >= text.length) {
   177тЖТ   177тЖТ        setDone(true);
   178тЖТ   178тЖТ        clearInterval(timer);
   179тЖТ   179тЖТ      }
   180тЖТ   180тЖТ    }, speed);
   181тЖТ   181тЖТ    return () => clearInterval(timer);
   182тЖТ   182тЖТ  }, [text, speed]);
   183тЖТ   183тЖТ
   184тЖТ   184тЖТ  return (
   185тЖТ   185тЖТ    <span className={className}>
   186тЖТ   186тЖТ      {displayed}
   187тЖТ   187тЖТ      {!done && (
   188тЖТ   188тЖТ        <motion.span
   189тЖТ   189тЖТ          animate={{ opacity: [1, 0] }}
   190тЖТ   190тЖТ          transition={{ duration: 0.4, repeat: Infinity, repeatType: 'reverse' }}
   191тЖТ   191тЖТ          className="inline-block w-[2px] h-[0.85em] bg-primary ml-0.5 align-middle rounded-full"
   192тЖТ   192тЖТ        />

   319тЖТ   319тЖТ// Hover tilt wrapper for StatsCards
   320тЖТ   320тЖТfunction TiltCard({ children, className = '' }: { children: React.ReactNode; className?: string }) {
   321тЖТ   321тЖТ  const ref = useRef<HTMLDivElement>(null);
   322тЖТ   322тЖТ
   323тЖТ   323тЖТ  const handleMouseMove = useCallback((e: React.MouseEvent<HTMLDivElement>) => {
   324тЖТ   324тЖТ    if (!ref.current) return;
   325тЖТ   325тЖТ    const rect = ref.current.getBoundingClientRect();
   326тЖТ   326тЖТ    const x = (e.clientX - rect.left) / rect.width - 0.5;
   327тЖТ   327тЖТ    const y = (e.clientY - rect.top) / rect.height - 0.5;
   328тЖТ   328тЖТ    ref.current.style.transform = `perspective(600px) rotateY(${x * 4}deg) rotateX(${-y * 4}deg)`;
   329тЖТ   329тЖТ  }, []);
   330тЖТ   330тЖТ
   331тЖТ   331тЖТ  const handleMouseLeave = useCallback(() => {
   332тЖТ   332тЖТ    if (ref.current) {
   333тЖТ   333тЖТ      ref.current.style.transform = 'perspective(600px) rotateY(0deg) rotateX(0deg)';
   334тЖТ   334тЖТ    }
   335тЖТ   335тЖТ  }, []);
   336тЖТ   336тЖТ
   337тЖТ   337тЖТ  return (
   338тЖТ   338тЖТ    <div
   339тЖТ   339тЖТ      ref={ref}
   340тЖТ   340тЖТ      className={className}
   341тЖТ   341тЖТ      onMouseMove={handleMouseMove}
   342тЖТ   342тЖТ      onMouseLeave={handleMouseLeave}
   343тЖТ   343тЖТ      style={{ transition: 'transform 0.2s ease-out' }}
   344тЖТ   344тЖТ    >
