    73→function formatRelativeTime(timestamp: number): string {
    74→  const now = Date.now();
    75→  const diff = now - timestamp;
    76→  const minutes = Math.floor(diff / 60000);
    77→  const hours = Math.floor(diff / 3600000);
    78→  const days = Math.floor(diff / 86400000);
    79→
    80→  if (minutes < 1) return 'Just now';
    81→  if (minutes < 60) return `${minutes}m ago`;
    82→  if (hours < 24) return `${hours}h ago`;
    83→  if (days === 1) return 'Yesterday';
    84→  if (days < 7) return `${days}d ago`;
    85→  return `${Math.floor(days / 7)}w ago`;
    86→}

   130→// Typewriter text component for greeting animation
   131→function TypewriterText({ text, speed = 40, className = '' }: { text: string; speed?: number; className?: string }) {
   132→  const [displayed, setDisplayed] = useState('');
   133→  const [done, setDone] = useState(false);
   134→
   135→  useEffect(() => {
   136→    setDisplayed('');
   137→    setDone(false);
   138→    let i = 0;
   139→    const timer = setInterval(() => {
   140→      i++;
   141→      setDisplayed(text.slice(0, i));
   142→      if (i >= text.length) {
   143→        setDone(true);
   144→        clearInterval(timer);
   145→      }
   146→    }, speed);
   147→    return () => clearInterval(timer);
   148→  }, [text, speed]);
   149→
   150→  return (
   151→    <span className={className}>
   152→      {displayed}
   153→      {!done && (
   154→        <motion.span
   155→          animate={{ opacity: [1, 0] }}
   156→          transition={{ duration: 0.4, repeat: Infinity, repeatType: 'reverse' }}
   157→          className="inline-block w-[2px] h-[0.85em] bg-primary ml-0.5 align-middle rounded-full"
   158→        />
   159→      )}
   160→    </span>
   161→  );
   162→}
