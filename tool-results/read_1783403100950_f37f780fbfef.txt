   495→function KeyboardShortcuts() {
   496→  const { navigate } = useAppStore();
   497→  const [open, setOpen] = useState(false);
   498→
   499→  // Navigation shortcuts + Cmd+, to open this dialog
   500→  useEffect(() => {
   501→    const handler = (e: KeyboardEvent) => {
   502→      if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) return;
   503→
   504→      // ? key (Shift+/) without Cmd/Ctrl/Alt opens shortcuts dialog
   505→      if (e.key === '?' && !e.metaKey && !e.ctrlKey && !e.altKey) {
   506→        e.preventDefault();
   507→        setOpen((v) => !v);
   508→        return;
   509→      }
   510→
   511→      // Cmd+, opens shortcuts dialog
   512→      if ((e.metaKey || e.ctrlKey) && e.key === ',') {
   513→        e.preventDefault();
   514→        setOpen((v) => !v);
   515→        return;
   516→      }
   517→
   518→      if ((e.metaKey || e.ctrlKey) && e.key === '1') { e.preventDefault(); navigate('dashboard'); }
   519→      if ((e.metaKey || e.ctrlKey) && e.key === '2') { e.preventDefault(); navigate('tutor'); }
   520→      if ((e.metaKey || e.ctrlKey) && e.key === '3') { e.preventDefault(); navigate('upload'); }
   521→      if ((e.metaKey || e.ctrlKey) && e.key === '4') { e.preventDefault(); navigate('quiz'); }
   522→      if ((e.metaKey || e.ctrlKey) && e.key === '5') { e.preventDefault(); navigate('notes'); }
   523→      if ((e.metaKey || e.ctrlKey) && e.key === '6') { e.preventDefault(); navigate('profile'); }
   524→      if ((e.metaKey || e.ctrlKey) && e.key === '7') { e.preventDefault(); navigate('settings'); }
   525→      if ((e.metaKey || e.ctrlKey) && e.key === '8') { e.preventDefault(); navigate('focus-timer'); }
   526→      if ((e.metaKey || e.ctrlKey) && e.key === '9') { e.preventDefault(); navigate('notes'); }
   527→    };
   528→    window.addEventListener('keydown', handler);
   529→    return () => window.removeEventListener('keydown', handler);
   530→  }, [navigate]);
   531→
   532→  return (
   533→    <>
   534→      {/* Floating ? button */}
   535→      <motion.button
   536→        onClick={() => setOpen(true)}
   537→        whileHover={{ scale: 1.1 }}
   538→        whileTap={{ scale: 0.95 }}

   596→const fullViewportViews: Array<string> = ['landing', 'onboarding'];
   597→
   598→// Spring transition config for view changes
   599→const springTransition = { type: 'spring' as const, stiffness: 300, damping: 30 };
   600→
   601→export function AppShell() {
   602→  const { currentView, navigate } = useAppStore();
   603→  const isFullViewport = fullViewportViews.includes(currentView);
   604→  const [transitioning, setTransitioning] = useState(false);
   605→  const prevViewRef = useRef(currentView);
   606→
   607→  // Detect view changes to trigger transition indicator
   608→  useEffect(() => {
   609→    if (prevViewRef.current !== currentView) {
   610→      prevViewRef.current = currentView;
   611→      const timer = setTimeout(() => setTransitioning(false), 400);
   612→      // Use a microtask to avoid synchronous setState in effect
   613→      const raf = requestAnimationFrame(() => setTransitioning(true));
   614→      return () => {
   615→        clearTimeout(timer);
   616→        cancelAnimationFrame(raf);
   617→      };
   618→    }
   619→  }, [currentView]);
   620→
   621→  const handleGoDashboard = () => navigate('dashboard');
   622→
   623→  const renderView = () => {
   624→    switch (currentView) {
   625→      case 'landing':
   626→        return <LandingPage />;
   627→      case 'onboarding':
   628→        return <OnboardingFlow />;
   629→      case 'dashboard':
   630→        return <Dashboard />;
   631→      case 'tutor':
   632→        return <TutorView />;
   633→      case 'upload':
   634→        return <UploadView />;
   635→      case 'quiz':
   636→        return <QuizView />;
   637→      case 'profile':
   638→        return <ProfileView />;
   639→      case 'course-detail':
   640→        return <CourseDetail />;
   641→      case 'notes':
   642→        return <NotesView />;
   643→      case 'focus-timer':
   644→        return <FocusTimerView />;
   645→      case 'settings':
   646→        return <SettingsView />;
   647→      default:
   648→        return <Dashboard />;
   649→    }
   650→  };
   651→
   652→  const viewVariant = viewTransitions[currentView] || defaultTransition;
   653→
   654→  return (
   655→    <>
   656→      <StoreInitializer />
   657→      <SearchModal />
   658→      <KeyboardShortcuts />
   659→
   660→      <AnimatePresence mode="wait">
   661→        {isFullViewport ? (
   662→          <motion.div
   663→            key={currentView}
   664→            variants={viewVariant}
   665→            initial="initial"
   666→            animate="animate"
   667→            exit="exit"
   668→            transition={springTransition}
   669→            className="min-h-screen mesh-gradient"
   670→          >
   671→            <TransitionIndicator show={transitioning} />
   672→            <Suspense fallback={<ViewLoader />}>
   673→              <ErrorBoundary onGoDashboard={handleGoDashboard}>
   674→                {renderView()}
   675→              </ErrorBoundary>
   676→            </Suspense>
   677→          </motion.div>
   678→        ) : (
   679→          <motion.div
   680→            key={`app-${currentView}`}
   681→            variants={viewVariant}
   682→            initial="initial"
   683→            animate="animate"
   684→            exit="exit"
   685→            transition={springTransition}
   686→            className="flex min-h-screen"
   687→          >
   688→            <AppSidebar />
   689→            <main className={`relative flex-1 min-h-screen overflow-y-auto overflow-x-hidden ${currentView === 'tutor' ? '!p-0 !max-w-none' : ''}`}>
   690→              {/* Animated mesh-gradient background */}
   691→              <div
   692→                className="pointer-events-none fixed inset-0 -z-10 opacity-40 dark:opacity-20"
   693→                aria-hidden="true"
   694→              >
   695→                <div
   696→                  className="absolute inset-0 blur-3xl"
   697→                  style={{

   699→                    backgroundSize: '400% 400%',
   700→                    animation: 'meshBgShift 20s ease-in-out infinite',
   701→                  }}
   702→                />
   703→              </div>
