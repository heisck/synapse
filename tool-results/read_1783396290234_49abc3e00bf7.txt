    33→// ---------- Typo-tolerance helper ----------
    34→function levenshteinDistance(a: string, b: string): number {
    35→  const matrix: number[][] = [];
    36→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    37→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    38→  for (let i = 1; i <= b.length; i++) {
    39→    for (let j = 1; j <= a.length; j++) {
    40→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    41→      matrix[i][j] = Math.min(
    42→        matrix[i - 1][j] + 1,
    43→        matrix[i][j - 1] + 1,
    44→        matrix[i - 1][j - 1] + cost,
    45→      );
    46→    }
    47→  }
    48→  return matrix[b.length][a.length];
    49→}

    51→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    52→  const ua = userAnswer.trim().toLowerCase();
    53→  const ca = correctAnswer.trim().toLowerCase();
    54→  if (ua === ca) return true;
    55→  if (ua.includes(ca) || ca.includes(ua)) return true;
    56→  // Levenshtein distance for typo tolerance
    57→  const dist = levenshteinDistance(ua, ca);
    58→  // Allow 1 typo for short answers, 2 for longer ones
    59→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    60→  if (dist <= threshold) return true;
    61→  // Word-level check: all words of correct answer present in user answer (in any order)
    62→  const correctWords = ca.split(/\s+/);
    63→  const userWords = ua.split(/\s+/);
    64→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    65→  return false;
    66→}

   204→// ---------- Animated counter hook ----------
   205→function useAnimatedCounter(target: number, duration: number = 1200) {
   206→  const [count, setCount] = useState(0);
   207→  const rafRef = useRef<number>(0);
   208→  useEffect(() => {
   209→    const start = performance.now();
   210→    const tick = (now: number) => {
   211→      const elapsed = now - start;
   212→      const progress = Math.min(elapsed / duration, 1);
   213→      const eased = 1 - Math.pow(1 - progress, 3);
   214→      setCount(Math.round(eased * target));
   215→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   216→    };
   217→    rafRef.current = requestAnimationFrame(tick);
   218→    return () => cancelAnimationFrame(rafRef.current);
   219→  }, [target, duration]);
   220→  return count;
   221→}

   274→// ---------- Matching question component (drag-and-drop) ----------
   275→function MatchingInput({
   276→  pairs,
   277→  onAnswer,
   278→}: {
   279→  pairs: Array<{ left: string; right: string }>;
   280→  onAnswer: (answer: string) => void;
   281→}) {
   282→  const shuffledRight = useMemo(() => {
   283→    const arr = [...pairs.map((p) => p.right)];
   284→    for (let i = arr.length - 1; i > 0; i--) {
   285→      const j = Math.floor(Math.random() * (i + 1));
   286→      [arr[i], arr[j]] = [arr[j], arr[i]];
   287→    }
   288→    return arr;
   289→  }, [pairs]);
   290→
   291→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   292→  const [matches, setMatches] = useState<Record<string, string>>({});
   293→
   294→  const handleLeftClick = (left: string) => {
   295→    setSelectedLeft(left);
   296→  };
   297→
   298→  const handleRightClick = (right: string) => {
   299→    if (!selectedLeft) return;
   300→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   301→    setSelectedLeft(null);
   302→  };
   303→
   304→  const handleRemoveMatch = (left: string) => {
   305→    setMatches((prev) => {
   306→      const next = { ...prev };
   307→      delete next[left];
   308→      return next;
   309→    });
   310→  };
   311→
   312→  const handleSubmit = () => {
   313→    const answer = Object.entries(matches)
   314→      .map(([l, r]) => `${l}-${r}`)
   315→      .join(', ');
   316→    onAnswer(answer);
   317→  };
   318→
   319→  const allMatched = pairs.every((p) => matches[p.left]);
   320→  const usedRights = new Set(Object.values(matches));
   321→
   322→  return (
   323→    <div className="space-y-4">
   324→      <p className="text-sm text-muted-foreground">
   325→        Click a left item, then click its match on the right. Click a matched item to remove it.
   326→      </p>
   327→      <div className="grid grid-cols-2 gap-4">
   328→        {/* Left column */}
   329→        <div className="flex flex-col gap-2">
   330→          {pairs.map((p) => (
   331→            <motion.button
   332→              key={p.left}
   333→              whileTap={{ scale: 0.98 }}
   334→              onClick={() =>
   335→                matches[p.left]
   336→                  ? handleRemoveMatch(p.left)
   337→                  : handleLeftClick(p.left)
   338→              }
   339→              className={`rounded-lg border p-3 text-left text-sm font-medium transition-all ${
   340→                selectedLeft === p.left
   341→                  ? 'border-primary bg-primary/10 text-primary ring-1 ring-primary/30 shadow-sm shadow-primary/10'
   342→                  : matches[p.left]
   343→                    ? 'border-emerald-500/50 bg-emerald-500/5 text-emerald-700 dark:text-emerald-400 shadow-sm shadow-emerald-500/5'
   344→                    : 'border-border hover:border-primary/30 hover:bg-primary/5'
   345→              }`}
   346→            >
   347→              <div className="flex items-center gap-2">
   348→                <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40" />
   349→                <span className="flex-1">{p.left}</span>
   350→                {matches[p.left] && (
   351→                  <motion.span
   352→                    initial={{ opacity: 0, x: 10 }}
   353→                    animate={{ opacity: 1, x: 0 }}
   354→                    className="text-xs opacity-70 truncate max-w-[120px] text-emerald-600 dark:text-emerald-400"
   355→                  >
   356→                    {matches[p.left]}
   357→                  </motion.span>
   358→                )}
   359→              </div>
   360→            </motion.button>
   361→          ))}
   362→        </div>
   363→        {/* Right column */}
   364→        <div className="flex flex-col gap-2">
   365→          {shuffledRight.map((r) => {
   366→            const isUsed = usedRights.has(r);
   367→            return (
   368→              <motion.button
   369→                key={r}
   370→                whileTap={{ scale: 0.98 }}
   371→                onClick={() => !isUsed && handleRightClick(r)}
   372→                disabled={isUsed || !selectedLeft}
   373→                className={`rounded-lg border p-3 text-left text-sm transition-all ${
   374→                  isUsed
   375→                    ? 'border-border bg-muted/50 text-muted-foreground line-through opacity-50'
   376→                    : selectedLeft
   377→                      ? 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-pointer hover:shadow-sm hover:shadow-primary/5'
   378→                      : 'border-border opacity-60'
   379→                }`}
   380→              >
   381→                <div className="flex items-center gap-2">
   382→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40" />
   383→                  {r}
   384→                </div>
   385→              </motion.button>
   386→            );
   387→          })}
   388→        </div>
   389→      </div>
   390→      <div className="flex items-center justify-between">
   391→        <div className="flex items-center gap-2">
   392→          <div className="relative h-1.5 w-24 rounded-full bg-muted/50 overflow-hidden">
   393→            <motion.div
   394→              className="absolute inset-y-0 left-0 rounded-full"
   395→              style={{ background: 'linear-gradient(90deg, oklch(0.627 0.194 149.214), oklch(0.687 0.159 177.89))' }}
   396→              animate={{ width: `${(Object.keys(matches).length / pairs.length) * 100}%` }}
   397→              transition={{ duration: 0.3 }}
   398→            />
   399→          </div>
   400→          <span className="text-xs text-muted-foreground">
   401→            {Object.keys(matches).length}/{pairs.length} matched
   402→          </span>
   403→        </div>
   404→        <motion.div
   405→          whileHover={{ scale: 1.05 }}
   406→          whileTap={{ scale: 0.95 }}
   407→        >
   408→          <Button onClick={handleSubmit} disabled={!allMatched} size="sm" className={allMatched ? 'glow-emerald' : ''}>
   409→            Submit Matches
   410→          </Button>
   411→        </motion.div>
   412→      </div>
   413→    </div>
   414→  );

   830→            <div className="inline-flex items-center gap-2 px-5 py-2 rounded-full text-sm font-bold shadow-lg bg-gradient-to-r from-emerald-400 to-teal-500 text-white shadow-emerald-400/30">
