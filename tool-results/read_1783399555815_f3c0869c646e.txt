    35тЖТ    35тЖТfunction levenshteinDistance(a: string, b: string): number {
    36тЖТ    36тЖТ  const matrix: number[][] = [];
    37тЖТ    37тЖТ  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    38тЖТ    38тЖТ  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    39тЖТ    39тЖТ  for (let i = 1; i <= b.length; i++) {
    40тЖТ    40тЖТ    for (let j = 1; j <= a.length; j++) {
    41тЖТ    41тЖТ      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    42тЖТ    42тЖТ      matrix[i][j] = Math.min(
    43тЖТ    43тЖТ        matrix[i - 1][j] + 1,
    44тЖТ    44тЖТ        matrix[i][j - 1] + 1,
    45тЖТ    45тЖТ        matrix[i - 1][j - 1] + cost,
    46тЖТ    46тЖТ      );
    47тЖТ    47тЖТ    }
    48тЖТ    48тЖТ  }
    49тЖТ    49тЖТ  return matrix[b.length][a.length];
    50тЖТ    50тЖТ}

    52тЖТ    52тЖТfunction isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    53тЖТ    53тЖТ  const ua = userAnswer.trim().toLowerCase();
    54тЖТ    54тЖТ  const ca = correctAnswer.trim().toLowerCase();
    55тЖТ    55тЖТ  if (ua === ca) return true;
    56тЖТ    56тЖТ  if (ua.includes(ca) || ca.includes(ua)) return true;
    57тЖТ    57тЖТ  // Levenshtein distance for typo tolerance
    58тЖТ    58тЖТ  const dist = levenshteinDistance(ua, ca);
    59тЖТ    59тЖТ  // Allow 1 typo for short answers, 2 for longer ones
    60тЖТ    60тЖТ  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    61тЖТ    61тЖТ  if (dist <= threshold) return true;
    62тЖТ    62тЖТ  // Word-level check: all words of correct answer present in user answer (in any order)
    63тЖТ    63тЖТ  const correctWords = ca.split(/\s+/);
    64тЖТ    64тЖТ  const userWords = ua.split(/\s+/);
    65тЖТ    65тЖТ  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    66тЖТ    66тЖТ  return false;
    67тЖТ    67тЖТ}

   219тЖТ   219тЖТ      const elapsed = now - start;
   220тЖТ   220тЖТ      const progress = Math.min(elapsed / duration, 1);
   221тЖТ   221тЖТ      const eased = 1 - Math.pow(1 - progress, 3);
   222тЖТ   222тЖТ      setCount(Math.round(eased * target));
   223тЖТ   223тЖТ      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   224тЖТ   224тЖТ    };
   225тЖТ   225тЖТ    rafRef.current = requestAnimationFrame(tick);
   226тЖТ   226тЖТ    return () => cancelAnimationFrame(rafRef.current);
   227тЖТ   227тЖТ  }, [target, duration]);
   228тЖТ   228тЖТ  return count;
