    40→// ---------- Typo-tolerance helper ----------
    41→function levenshteinDistance(a: string, b: string): number {
    42→  const matrix: number[][] = [];
    43→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    44→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    45→  for (let i = 1; i <= b.length; i++) {
    46→    for (let j = 1; j <= a.length; j++) {
    47→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    48→      matrix[i][j] = Math.min(
    49→        matrix[i - 1][j] + 1,
    50→        matrix[i][j - 1] + 1,
    51→        matrix[i - 1][j - 1] + cost,
    52→      );
    53→    }
    54→  }
    55→  return matrix[b.length][a.length];
    56→}

    58→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    59→  const ua = userAnswer.trim().toLowerCase();
    60→  const ca = correctAnswer.trim().toLowerCase();
    61→  if (ua === ca) return true;
    62→  if (ua.includes(ca) || ca.includes(ua)) return true;
    63→  // Levenshtein distance for typo tolerance
    64→  const dist = levenshteinDistance(ua, ca);
    65→  // Allow 1 typo for short answers, 2 for longer ones
    66→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    67→  if (dist <= threshold) return true;
    68→  // Word-level check: all words of correct answer present in user answer (in any order)
    69→  const correctWords = ca.split(/\s+/);
    70→  const userWords = ua.split(/\s+/);
    71→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    72→  return false;
    73→}

   240→function updateDailyStreak(): { current: number; best: number; lastDate: string } {
   241→  const streak = loadDailyStreak();
   242→  const today = getTodayStr();
   243→
   244→  if (streak.lastDate === today) return streak;
   245→
   246→  const yesterday = new Date();
   247→  yesterday.setDate(yesterday.getDate() - 1);
   248→  const yesterdayStr = yesterday.toISOString().split('T')[0];
   249→
   250→  let newCurrent: number;
   251→  if (streak.lastDate === yesterdayStr) {
   252→    newCurrent = streak.current + 1;
   253→  } else {
   254→    newCurrent = 1;
   255→  }
   256→
   257→  const newBest = Math.max(streak.best, newCurrent);
   258→  const updated = { current: newCurrent, best: newBest, lastDate: today };
   259→  saveDailyStreak(updated);
   260→  return updated;
   261→}

   263→function getTimeUntilMidnight(): { hours: number; minutes: number; seconds: number } {
   264→  const now = new Date();
   265→  const midnight = new Date(now);
   266→  midnight.setHours(24, 0, 0, 0);
   267→  const diff = midnight.getTime() - now.getTime();
   268→  return {
   269→    hours: Math.floor(diff / (1000 * 60 * 60)),
   270→    minutes: Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60)),
   271→    seconds: Math.floor((diff % (1000 * 60)) / 1000),
   272→  };
   273→}

   312→// ---------- Animated counter hook ----------
   313→function useAnimatedCounter(target: number, duration: number = 1200) {
   314→  const [count, setCount] = useState(0);
   315→  const rafRef = useRef<number>(0);
   316→  useEffect(() => {
   317→    const start = performance.now();
   318→    const tick = (now: number) => {
   319→      const elapsed = now - start;
   320→      const progress = Math.min(elapsed / duration, 1);
   321→      const eased = 1 - Math.pow(1 - progress, 3);
   322→      setCount(Math.round(eased * target));
   323→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   324→    };
   325→    rafRef.current = requestAnimationFrame(tick);
   326→    return () => cancelAnimationFrame(rafRef.current);
   327→  }, [target, duration]);
   328→  return count;
   329→}

   384→  pairs,
   385→  onAnswer,
   386→}: {
   387→  pairs: Array<{ left: string; right: string }>;
   388→  onAnswer: (answer: string) => void;
   389→}) {
   390→  const shuffledRight = useMemo(() => {
   391→    const arr = [...pairs.map((p) => p.right)];
   392→    for (let i = arr.length - 1; i > 0; i--) {
   393→      const j = Math.floor(Math.random() * (i + 1));
   394→      [arr[i], arr[j]] = [arr[j], arr[i]];
   395→    }
   396→    return arr;
   397→  }, [pairs]);
   398→
   399→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);

   404→  const containerRef = useRef<HTMLDivElement>(null);
   405→
   406→  // Build index maps for safe DOM queries
   407→  const leftIdxMap = useMemo(() => {
   408→    const map: Record<string, number> = {};
   409→    pairs.forEach((p, i) => { map[p.left] = i; });
   410→    return map;
   411→  }, [pairs]);
   412→
   413→  const rightIdxMap = useMemo(() => {
   414→    const map: Record<string, number> = {};
   415→    shuffledRight.forEach((r, i) => { map[r] = i; });
   416→    return map;
   417→  }, [shuffledRight]);
   418→
   419→  // Compute SVG connector line positions
   420→  const updateLinePositions = useCallback(() => {
   421→    const container = containerRef.current;
   422→    if (!container) return;
   423→
   424→    requestAnimationFrame(() => {
   425→      if (!container) return;
   426→      const containerRect = container.getBoundingClientRect();
   427→      const positions: Array<{ id: string; x1: number; y1: number; x2: number; y2: number }> = [];
   428→
   429→      for (const [left, right] of Object.entries(matches)) {
   430→        const lIdx = leftIdxMap[left];
   431→        const rIdx = rightIdxMap[right];
   432→        if (lIdx === undefined || rIdx === undefined) continue;
   433→
   434→        const leftEl = container.querySelector(`[data-left-idx="${lIdx}"]`);
   435→        const rightEl = container.querySelector(`[data-right-idx="${rIdx}"]`);
   436→
   437→        if (leftEl && rightEl) {
   438→          const leftRect = leftEl.getBoundingClientRect();
   439→          const rightRect = rightEl.getBoundingClientRect();
   440→
   441→          positions.push({
   442→            id: `${left}-${right}`,
   443→            x1: leftRect.right - containerRect.left,
   444→            y1: leftRect.top + leftRect.height / 2 - containerRect.top,
   445→            x2: rightRect.left - containerRect.left,
   446→            y2: rightRect.top + rightRect.height / 2 - containerRect.top,
   447→          });
   448→        }
   449→      }
   450→
   451→      setLinePositions(positions);
   452→    });
   453→  }, [matches, leftIdxMap, rightIdxMap]);
   454→
   455→  // Recalculate on matches change
   456→  useEffect(() => {
   457→    updateLinePositions();
   458→  }, [updateLinePositions]);
   459→
   460→  // Recalculate on resize
   461→  useEffect(() => {
   462→    const container = containerRef.current;
   463→    if (!container) return;
   464→
   465→    const observer = new ResizeObserver(updateLinePositions);
   466→    observer.observe(container);
   467→
   468→    return () => observer.disconnect();
   469→  }, [updateLinePositions]);
   470→
   471→  const handleLeftClick = (left: string) => {
   472→    if (matches[left]) {
   473→      // Remove match by clicking connected left item
   474→      setMatches((prev) => {
   475→        const next = { ...prev };
   476→        delete next[left];
   477→        return next;
   478→      });
   479→      return;
   480→    }
   481→    setSelectedLeft(left);
   482→  };
   483→
   484→  const handleRightClick = (right: string) => {
   485→    if (!selectedLeft) return;
   486→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   487→    setSelectedLeft(null);
   488→  };
   489→
   490→  const handleDragStart = (e: React.DragEvent, left: string) => {
   491→    setDragItem(left);
   492→    e.dataTransfer.setData('text/plain', left);
   493→    e.dataTransfer.effectAllowed = 'link';

   497→    e.preventDefault();

   499→    const isUsedByOther = Object.entries(matches).some(

   503→      setDragOverTarget(right);
   504→    }
   505→  };

   508→    setDragOverTarget(null);
   509→  };
   510→
   511→  const handleDrop = (e: React.DragEvent, right: string) => {
   512→    e.preventDefault();
   513→    const left = e.dataTransfer.getData('text/plain') || dragItem;
   514→    if (left) {
   515→      // Remove any existing match for this left item first
   516→      setMatches((prev) => {
   517→        const next = { ...prev };
   518→        delete next[left];
   519→        next[left] = right;
   520→        return next;
   521→      });
   522→    }
   523→    setDragItem(null);
   524→    setDragOverTarget(null);
   525→  };
   526→
   527→  const handleDragEnd = () => {
   528→    setDragItem(null);
   529→    setDragOverTarget(null);
   530→  };

   537→  };
   538→
   539→  const allMatched = pairs.every((p) => matches[p.left]);
   540→  const usedRights = new Set(Object.values(matches));

   543→    <div className="space-y-4">
   544→      <p className="text-sm text-muted-foreground">
   545→        Drag left items to their match on the right, or click to select. Click a connected left item to remove it.
   546→      </p>
