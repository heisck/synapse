   107→   107→function formatRelativeTime(timestamp: number): string {
   108→   108→  const now = Date.now();
   109→   109→  const diff = now - timestamp;
   110→   110→  const minutes = Math.floor(diff / 60000);
   111→   111→  const hours = Math.floor(diff / 3600000);
   112→   112→  const days = Math.floor(diff / 86400000);
   113→   113→
   114→   114→  if (minutes < 1) return 'Just now';
   115→   115→  if (minutes < 60) return `${minutes}m ago`;
   116→   116→  if (hours < 24) return `${hours}h ago`;
   117→   117→  if (days === 1) return 'Yesterday';
   118→   118→  if (days < 7) return `${days}d ago`;
   119→   119→  return `${Math.floor(days / 7)}w ago`;
   120→   120→}

   164→   164→// Typewriter text component for greeting animation
   165→   165→function TypewriterText({ text, speed = 40, className = '' }: { text: string; speed?: number; className?: string }) {
   166→   166→  const [displayed, setDisplayed] = useState('');
   167→   167→  const [done, setDone] = useState(false);
   168→   168→
   169→   169→  useEffect(() => {
   170→   170→    setDisplayed('');
   171→   171→    setDone(false);
   172→   172→    let i = 0;
   173→   173→    const timer = setInterval(() => {
   174→   174→      i++;
   175→   175→      setDisplayed(text.slice(0, i));
   176→   176→      if (i >= text.length) {
   177→   177→        setDone(true);
   178→   178→        clearInterval(timer);
   179→   179→      }
   180→   180→    }, speed);
   181→   181→    return () => clearInterval(timer);
   182→   182→  }, [text, speed]);
   183→   183→
   184→   184→  return (
   185→   185→    <span className={className}>
   186→   186→      {displayed}
   187→   187→      {!done && (
   188→   188→        <motion.span
   189→   189→          animate={{ opacity: [1, 0] }}
   190→   190→          transition={{ duration: 0.4, repeat: Infinity, repeatType: 'reverse' }}
   191→   191→          className="inline-block w-[2px] h-[0.85em] bg-primary ml-0.5 align-middle rounded-full"
   192→   192→        />
   193→   193→      )}
   194→   194→    </span>
   195→   195→  );
   196→   196→}

   319→   319→// Hover tilt wrapper for StatsCards
   320→   320→function TiltCard({ children, className = '' }: { children: React.ReactNode; className?: string }) {
   321→   321→  const ref = useRef<HTMLDivElement>(null);
   322→   322→
   323→   323→  const handleMouseMove = useCallback((e: React.MouseEvent<HTMLDivElement>) => {
   324→   324→    if (!ref.current) return;
   325→   325→    const rect = ref.current.getBoundingClientRect();
   326→   326→    const x = (e.clientX - rect.left) / rect.width - 0.5;
   327→   327→    const y = (e.clientY - rect.top) / rect.height - 0.5;
   328→   328→    ref.current.style.transform = `perspective(600px) rotateY(${x * 4}deg) rotateX(${-y * 4}deg)`;
   329→   329→  }, []);
   330→   330→
   331→   331→  const handleMouseLeave = useCallback(() => {
   332→   332→    if (ref.current) {
   333→   333→      ref.current.style.transform = 'perspective(600px) rotateY(0deg) rotateX(0deg)';
   334→   334→    }
   335→   335→  }, []);
   336→   336→
   337→   337→  return (
   338→   338→    <div
   339→   339→      ref={ref}
   340→   340→      className={className}
   341→   341→      onMouseMove={handleMouseMove}
   342→   342→      onMouseLeave={handleMouseLeave}
   343→   343→      style={{ transition: 'transform 0.2s ease-out' }}
   344→   344→    >
   345→   345→      {children}
   346→   346→    </div>
   347→   347→  );
   348→   348→}
