    85→    85→function formatRelativeTime(timestamp: number): string {
    86→    86→  const now = Date.now();
    87→    87→  const diff = now - timestamp;
    88→    88→  const minutes = Math.floor(diff / 60000);
    89→    89→  const hours = Math.floor(diff / 3600000);
    90→    90→  const days = Math.floor(diff / 86400000);
    91→    91→
    92→    92→  if (minutes < 1) return 'Just now';
    93→    93→  if (minutes < 60) return `${minutes}m ago`;
    94→    94→  if (hours < 24) return `${hours}h ago`;
    95→    95→  if (days === 1) return 'Yesterday';
    96→    96→  if (days < 7) return `${days}d ago`;
    97→    97→  return `${Math.floor(days / 7)}w ago`;
    98→    98→}

   142→   142→// Typewriter text component for greeting animation
   143→   143→function TypewriterText({ text, speed = 40, className = '' }: { text: string; speed?: number; className?: string }) {
   144→   144→  const [displayed, setDisplayed] = useState('');
   145→   145→  const [done, setDone] = useState(false);
   146→   146→
   147→   147→  useEffect(() => {
   148→   148→    setDisplayed('');
   149→   149→    setDone(false);
   150→   150→    let i = 0;
   151→   151→    const timer = setInterval(() => {
   152→   152→      i++;
   153→   153→      setDisplayed(text.slice(0, i));
   154→   154→      if (i >= text.length) {
   155→   155→        setDone(true);
   156→   156→        clearInterval(timer);
   157→   157→      }
   158→   158→    }, speed);
   159→   159→    return () => clearInterval(timer);
   160→   160→  }, [text, speed]);
   161→   161→
   162→   162→  return (
   163→   163→    <span className={className}>
   164→   164→      {displayed}
   165→   165→      {!done && (
   166→   166→        <motion.span
   167→   167→          animate={{ opacity: [1, 0] }}
   168→   168→          transition={{ duration: 0.4, repeat: Infinity, repeatType: 'reverse' }}
   169→   169→          className="inline-block w-[2px] h-[0.85em] bg-primary ml-0.5 align-middle rounded-full"
   170→   170→        />
   171→   171→      )}
   172→   172→    </span>
   173→   173→  );
   174→   174→}

   297→   297→// Hover tilt wrapper for StatsCards
   298→   298→function TiltCard({ children, className = '' }: { children: React.ReactNode; className?: string }) {
   299→   299→  const ref = useRef<HTMLDivElement>(null);
   300→   300→
   301→   301→  const handleMouseMove = useCallback((e: React.MouseEvent<HTMLDivElement>) => {
   302→   302→    if (!ref.current) return;
   303→   303→    const rect = ref.current.getBoundingClientRect();
   304→   304→    const x = (e.clientX - rect.left) / rect.width - 0.5;
   305→   305→    const y = (e.clientY - rect.top) / rect.height - 0.5;
   306→   306→    ref.current.style.transform = `perspective(600px) rotateY(${x * 4}deg) rotateX(${-y * 4}deg)`;
   307→   307→  }, []);
   308→   308→
   309→   309→  const handleMouseLeave = useCallback(() => {
   310→   310→    if (ref.current) {
   311→   311→      ref.current.style.transform = 'perspective(600px) rotateY(0deg) rotateX(0deg)';
   312→   312→    }
   313→   313→  }, []);
   314→   314→
   315→   315→  return (
   316→   316→    <div
   317→   317→      ref={ref}
   318→   318→      className={className}
   319→   319→      onMouseMove={handleMouseMove}
   320→   320→      onMouseLeave={handleMouseLeave}
   321→   321→      style={{ transition: 'transform 0.2s ease-out' }}
   322→   322→    >
   323→   323→      {children}
   324→   324→    </div>
   325→   325→  );
   326→   326→}
