    98тЖТ    98тЖТ  private ctx: AudioContext | null = null;
    99тЖТ    99тЖТ  private nodes: Map<string, { source: AudioNode; gain: GainNode; cleanup?: () => void }> = new Map();
   100тЖТ   100тЖТ
   101тЖТ   101тЖТ  private getCtx(): AudioContext {
   102тЖТ   102тЖТ    if (!this.ctx) {
   103тЖТ   103тЖТ      this.ctx = new AudioContext();
   104тЖТ   104тЖТ    }
   105тЖТ   105тЖТ    if (this.ctx.state === 'suspended') {
   106тЖТ   106тЖТ      this.ctx.resume();
   107тЖТ   107тЖТ    }
   108тЖТ   108тЖТ    return this.ctx;
   109тЖТ   109тЖТ  }
   110тЖТ   110тЖТ
   111тЖТ   111тЖТ  startRain(volume: number) {
   112тЖТ   112тЖТ    this.stop('rain');
   113тЖТ   113тЖТ    const ctx = this.getCtx();
   114тЖТ   114тЖТ    const bufferSize = 2 * ctx.sampleRate;
   115тЖТ   115тЖТ    const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
   116тЖТ   116тЖТ    const data = buffer.getChannelData(0);
   117тЖТ   117тЖТ    for (let i = 0; i < bufferSize; i++) {
   118тЖТ   118тЖТ      data[i] = Math.random() * 2 - 1;
   119тЖТ   119тЖТ    }
   120тЖТ   120тЖТ    const source = ctx.createBufferSource();
   121тЖТ   121тЖТ    source.buffer = buffer;
   122тЖТ   122тЖТ    source.loop = true;
   123тЖТ   123тЖТ
   124тЖТ   124тЖТ    const lowpass = ctx.createBiquadFilter();
   125тЖТ   125тЖТ    lowpass.type = 'lowpass';
   126тЖТ   126тЖТ    lowpass.frequency.value = 800;
   127тЖТ   127тЖТ    lowpass.Q.value = 0.7;
   128тЖТ   128тЖТ
   129тЖТ   129тЖТ    const gain = ctx.createGain();
   130тЖТ   130тЖТ    gain.gain.value = volume / 100 * 0.5;
   131тЖТ   131тЖТ
   132тЖТ   132тЖТ    source.connect(lowpass);
   133тЖТ   133тЖТ    lowpass.connect(gain);
   134тЖТ   134тЖТ    gain.connect(ctx.destination);
   135тЖТ   135тЖТ    source.start();
   136тЖТ   136тЖТ
   137тЖТ   137тЖТ    this.nodes.set('rain', { source, gain });
   138тЖТ   138тЖТ  }

   140тЖТ   140тЖТ  startForest(volume: number) {
   141тЖТ   141тЖТ    this.stop('forest');
   142тЖТ   142тЖТ    const ctx = this.getCtx();
   143тЖТ   143тЖТ    const bufferSize = 4 * ctx.sampleRate;
   144тЖТ   144тЖТ    const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
   145тЖТ   145тЖТ    const data = buffer.getChannelData(0);
   146тЖТ   146тЖТ    // Brown noise generation
   147тЖТ   147тЖТ    let lastOut = 0;
   148тЖТ   148тЖТ    for (let i = 0; i < bufferSize; i++) {
   149тЖТ   149тЖТ      const white = Math.random() * 2 - 1;
   150тЖТ   150тЖТ      data[i] = (lastOut + 0.02 * white) / 1.02;
   151тЖТ   151тЖТ      lastOut = data[i];
   152тЖТ   152тЖТ      data[i] *= 3.5;
   153тЖТ   153тЖТ    }
   154тЖТ   154тЖТ    const source = ctx.createBufferSource();
   155тЖТ   155тЖТ    source.buffer = buffer;
   156тЖТ   156тЖТ    source.loop = true;
   157тЖТ   157тЖТ
   158тЖТ   158тЖТ    const bandpass = ctx.createBiquadFilter();
   159тЖТ   159тЖТ    bandpass.type = 'bandpass';
   160тЖТ   160тЖТ    bandpass.frequency.value = 400;
   161тЖТ   161тЖТ    bandpass.Q.value = 0.5;
   162тЖТ   162тЖТ
   163тЖТ   163тЖТ    const gain = ctx.createGain();
   164тЖТ   164тЖТ    gain.gain.value = volume / 100 * 0.4;
   165тЖТ   165тЖТ
   166тЖТ   166тЖТ    source.connect(bandpass);
   167тЖТ   167тЖТ    bandpass.connect(gain);
   168тЖТ   168тЖТ    gain.connect(ctx.destination);
   169тЖТ   169тЖТ    source.start();
   170тЖТ   170тЖТ
   171тЖТ   171тЖТ    this.nodes.set('forest', { source, gain });
   172тЖТ   172тЖТ  }

   176тЖТ   176тЖТ    const ctx = this.getCtx();
   177тЖТ   177тЖТ    const osc = ctx.createOscillator();
   178тЖТ   178тЖТ    osc.type = 'sine';
   179тЖТ   179тЖТ    osc.frequency.value = 0.15;
   180тЖТ   180тЖТ
   181тЖТ   181тЖТ    const lfoGain = ctx.createGain();
   182тЖТ   182тЖТ    lfoGain.gain.value = 200;
   183тЖТ   183тЖТ
   184тЖТ   184тЖТ    const lfo = ctx.createOscillator();
   185тЖТ   185тЖТ    lfo.type = 'sine';
   186тЖТ   186тЖТ    lfo.frequency.value = 0.08;
   187тЖТ   187тЖТ
   188тЖТ   188тЖТ    const mainOsc = ctx.createOscillator();
   189тЖТ   189тЖТ    mainOsc.type = 'sine';
   190тЖТ   190тЖТ    mainOsc.frequency.value = 100;
   191тЖТ   191тЖТ
   192тЖТ   192тЖТ    const mainGain = ctx.createGain();
   193тЖТ   193тЖТ    mainGain.gain.value = volume / 100 * 0.3;
   194тЖТ   194тЖТ
   195тЖТ   195тЖТ    lfo.connect(lfoGain);
   196тЖТ   196тЖТ    lfoGain.connect(mainOsc.frequency);
   197тЖТ   197тЖТ    mainOsc.connect(mainGain);
   198тЖТ   198тЖТ    mainGain.connect(ctx.destination);
   199тЖТ   199тЖТ
   200тЖТ   200тЖТ    lfo.start();
   201тЖТ   201тЖТ    mainOsc.start();
   202тЖТ   202тЖТ
   203тЖТ   203тЖТ    this.nodes.set('waves', {
   204тЖТ   204тЖТ      source: mainOsc,
   205тЖТ   205тЖТ      gain: mainGain,
   206тЖТ   206тЖТ      cleanup: () => {
   207тЖТ   207тЖТ        try { lfo.stop(); } catch { /* ignore */ }
   208тЖТ   208тЖТ      },
   209тЖТ   209тЖТ    });
   210тЖТ   210тЖТ  }
   211тЖТ   211тЖТ
   212тЖТ   212тЖТ  setVolume(id: string, volume: number) {
   213тЖТ   213тЖТ    const entry = this.nodes.get(id);
   214тЖТ   214тЖТ    if (entry) {
   215тЖТ   215тЖТ      entry.gain.gain.setTargetAtTime(volume / 100 * 0.5, this.ctx?.currentTime ?? 0, 0.1);
   216тЖТ   216тЖТ    }
   217тЖТ   217тЖТ  }
   218тЖТ   218тЖТ
   219тЖТ   219тЖТ  stop(id: string) {
   220тЖТ   220тЖТ    const entry = this.nodes.get(id);
   221тЖТ   221тЖТ    if (entry) {
   222тЖТ   222тЖТ      entry.cleanup?.();
   223тЖТ   223тЖТ      try {
   224тЖТ   224тЖТ        if (entry.source instanceof AudioBufferSourceNode || entry.source instanceof OscillatorNode) {
   225тЖТ   225тЖТ          entry.source.stop();
   226тЖТ   226тЖТ        }
   227тЖТ   227тЖТ      } catch { /* ignore */ }
   228тЖТ   228тЖТ      this.nodes.delete(id);
   229тЖТ   229тЖТ    }
   230тЖТ   230тЖТ  }
   231тЖТ   231тЖТ
   232тЖТ   232тЖТ  stopAll() {
   233тЖТ   233тЖТ    this.nodes.forEach((_, id) => this.stop(id));
   234тЖТ   234тЖТ  }
   235тЖТ   235тЖТ
   236тЖТ   236тЖТ  dispose() {
   237тЖТ   237тЖТ    this.stopAll();
   238тЖТ   238тЖТ    if (this.ctx) {
   239тЖТ   239тЖТ      this.ctx.close();
   240тЖТ   240тЖТ      this.ctx = null;
   241тЖТ   241тЖТ    }
   242тЖТ   242тЖТ  }
   243тЖТ   243тЖТ}
