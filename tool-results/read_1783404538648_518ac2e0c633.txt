    98тЖТ    98тЖТclass AmbientSoundEngine {
    99тЖТ    99тЖТ  private ctx: AudioContext | null = null;
   100тЖТ   100тЖТ  private nodes: Map<string, { source: AudioNode; gain: GainNode; cleanup?: () => void }> = new Map();
   101тЖТ   101тЖТ
   102тЖТ   102тЖТ  private getCtx(): AudioContext {
   103тЖТ   103тЖТ    if (!this.ctx) {
   104тЖТ   104тЖТ      this.ctx = new AudioContext();
   105тЖТ   105тЖТ    }
   106тЖТ   106тЖТ    if (this.ctx.state === 'suspended') {
   107тЖТ   107тЖТ      this.ctx.resume();
   108тЖТ   108тЖТ    }
   109тЖТ   109тЖТ    return this.ctx;
   110тЖТ   110тЖТ  }
   111тЖТ   111тЖТ
   112тЖТ   112тЖТ  startRain(volume: number) {
   113тЖТ   113тЖТ    this.stop('rain');
   114тЖТ   114тЖТ    const ctx = this.getCtx();
   115тЖТ   115тЖТ    const bufferSize = 2 * ctx.sampleRate;
   116тЖТ   116тЖТ    const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
   117тЖТ   117тЖТ    const data = buffer.getChannelData(0);
   118тЖТ   118тЖТ    for (let i = 0; i < bufferSize; i++) {
   119тЖТ   119тЖТ      data[i] = Math.random() * 2 - 1;
   120тЖТ   120тЖТ    }
   121тЖТ   121тЖТ    const source = ctx.createBufferSource();
   122тЖТ   122тЖТ    source.buffer = buffer;
   123тЖТ   123тЖТ    source.loop = true;
   124тЖТ   124тЖТ
   125тЖТ   125тЖТ    const lowpass = ctx.createBiquadFilter();
   126тЖТ   126тЖТ    lowpass.type = 'lowpass';
   127тЖТ   127тЖТ    lowpass.frequency.value = 800;
   128тЖТ   128тЖТ    lowpass.Q.value = 0.7;
   129тЖТ   129тЖТ
   130тЖТ   130тЖТ    const gain = ctx.createGain();
   131тЖТ   131тЖТ    gain.gain.value = volume / 100 * 0.5;
   132тЖТ   132тЖТ
   133тЖТ   133тЖТ    source.connect(lowpass);
   134тЖТ   134тЖТ    lowpass.connect(gain);
   135тЖТ   135тЖТ    gain.connect(ctx.destination);
   136тЖТ   136тЖТ    source.start();
   137тЖТ   137тЖТ
   138тЖТ   138тЖТ    this.nodes.set('rain', { source, gain });
   139тЖТ   139тЖТ  }
   140тЖТ   140тЖТ
   141тЖТ   141тЖТ  startForest(volume: number) {
   142тЖТ   142тЖТ    this.stop('forest');
   143тЖТ   143тЖТ    const ctx = this.getCtx();
   144тЖТ   144тЖТ    const bufferSize = 4 * ctx.sampleRate;
   145тЖТ   145тЖТ    const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
   146тЖТ   146тЖТ    const data = buffer.getChannelData(0);
   147тЖТ   147тЖТ    // Brown noise generation
   148тЖТ   148тЖТ    let lastOut = 0;
   149тЖТ   149тЖТ    for (let i = 0; i < bufferSize; i++) {
   150тЖТ   150тЖТ      const white = Math.random() * 2 - 1;
   151тЖТ   151тЖТ      data[i] = (lastOut + 0.02 * white) / 1.02;
   152тЖТ   152тЖТ      lastOut = data[i];
   153тЖТ   153тЖТ      data[i] *= 3.5;
   154тЖТ   154тЖТ    }
   155тЖТ   155тЖТ    const source = ctx.createBufferSource();
   156тЖТ   156тЖТ    source.buffer = buffer;
   157тЖТ   157тЖТ    source.loop = true;
   158тЖТ   158тЖТ
   159тЖТ   159тЖТ    const bandpass = ctx.createBiquadFilter();
   160тЖТ   160тЖТ    bandpass.type = 'bandpass';
   161тЖТ   161тЖТ    bandpass.frequency.value = 400;
   162тЖТ   162тЖТ    bandpass.Q.value = 0.5;
   163тЖТ   163тЖТ
   164тЖТ   164тЖТ    const gain = ctx.createGain();
   165тЖТ   165тЖТ    gain.gain.value = volume / 100 * 0.4;
   166тЖТ   166тЖТ
   167тЖТ   167тЖТ    source.connect(bandpass);
   168тЖТ   168тЖТ    bandpass.connect(gain);
   169тЖТ   169тЖТ    gain.connect(ctx.destination);
   170тЖТ   170тЖТ    source.start();
   171тЖТ   171тЖТ
   172тЖТ   172тЖТ    this.nodes.set('forest', { source, gain });
   173тЖТ   173тЖТ  }

   175тЖТ   175тЖТ  startWaves(volume: number) {
   176тЖТ   176тЖТ    this.stop('waves');
   177тЖТ   177тЖТ    const ctx = this.getCtx();
   178тЖТ   178тЖТ    const osc = ctx.createOscillator();
   179тЖТ   179тЖТ    osc.type = 'sine';
   180тЖТ   180тЖТ    osc.frequency.value = 0.15;
   181тЖТ   181тЖТ
   182тЖТ   182тЖТ    const lfoGain = ctx.createGain();
   183тЖТ   183тЖТ    lfoGain.gain.value = 200;
   184тЖТ   184тЖТ
   185тЖТ   185тЖТ    const lfo = ctx.createOscillator();
   186тЖТ   186тЖТ    lfo.type = 'sine';
   187тЖТ   187тЖТ    lfo.frequency.value = 0.08;
   188тЖТ   188тЖТ
   189тЖТ   189тЖТ    const mainOsc = ctx.createOscillator();
   190тЖТ   190тЖТ    mainOsc.type = 'sine';
   191тЖТ   191тЖТ    mainOsc.frequency.value = 100;
   192тЖТ   192тЖТ
   193тЖТ   193тЖТ    const mainGain = ctx.createGain();
   194тЖТ   194тЖТ    mainGain.gain.value = volume / 100 * 0.3;
   195тЖТ   195тЖТ
   196тЖТ   196тЖТ    lfo.connect(lfoGain);
   197тЖТ   197тЖТ    lfoGain.connect(mainOsc.frequency);
   198тЖТ   198тЖТ    mainOsc.connect(mainGain);
   199тЖТ   199тЖТ    mainGain.connect(ctx.destination);
   200тЖТ   200тЖТ
   201тЖТ   201тЖТ    lfo.start();
   202тЖТ   202тЖТ    mainOsc.start();
   203тЖТ   203тЖТ
   204тЖТ   204тЖТ    this.nodes.set('waves', {
   205тЖТ   205тЖТ      source: mainOsc,
   206тЖТ   206тЖТ      gain: mainGain,
   207тЖТ   207тЖТ      cleanup: () => {
   208тЖТ   208тЖТ        try { lfo.stop(); } catch { /* ignore */ }
   209тЖТ   209тЖТ      },
   210тЖТ   210тЖТ    });
   211тЖТ   211тЖТ  }
   212тЖТ   212тЖТ
   213тЖТ   213тЖТ  setVolume(id: string, volume: number) {
   214тЖТ   214тЖТ    const entry = this.nodes.get(id);
   215тЖТ   215тЖТ    if (entry) {
   216тЖТ   216тЖТ      entry.gain.gain.setTargetAtTime(volume / 100 * 0.5, this.ctx?.currentTime ?? 0, 0.1);
   217тЖТ   217тЖТ    }
   218тЖТ   218тЖТ  }
   219тЖТ   219тЖТ
   220тЖТ   220тЖТ  stop(id: string) {
   221тЖТ   221тЖТ    const entry = this.nodes.get(id);
   222тЖТ   222тЖТ    if (entry) {
   223тЖТ   223тЖТ      entry.cleanup?.();
   224тЖТ   224тЖТ      try {
   225тЖТ   225тЖТ        if (entry.source instanceof AudioBufferSourceNode || entry.source instanceof OscillatorNode) {
   226тЖТ   226тЖТ          entry.source.stop();
   227тЖТ   227тЖТ        }
   228тЖТ   228тЖТ      } catch { /* ignore */ }
   229тЖТ   229тЖТ      this.nodes.delete(id);
   230тЖТ   230тЖТ    }
   231тЖТ   231тЖТ  }
   232тЖТ   232тЖТ
   233тЖТ   233тЖТ  stopAll() {
   234тЖТ   234тЖТ    this.nodes.forEach((_, id) => this.stop(id));
   235тЖТ   235тЖТ  }
   236тЖТ   236тЖТ
   237тЖТ   237тЖТ  dispose() {
   238тЖТ   238тЖТ    this.stopAll();
   239тЖТ   239тЖТ    if (this.ctx) {
   240тЖТ   240тЖТ      this.ctx.close();
   241тЖТ   241тЖТ      this.ctx = null;
   242тЖТ   242тЖТ    }
   243тЖТ   243тЖТ  }
   244тЖТ   244тЖТ}
