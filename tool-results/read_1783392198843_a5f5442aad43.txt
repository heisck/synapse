    24→// ---------- Typo-tolerance helper ----------
    25→function levenshteinDistance(a: string, b: string): number {
    26→  const matrix: number[][] = [];
    27→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    28→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    29→  for (let i = 1; i <= b.length; i++) {
    30→    for (let j = 1; j <= a.length; j++) {
    31→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    32→      matrix[i][j] = Math.min(
    33→        matrix[i - 1][j] + 1,
    34→        matrix[i][j - 1] + 1,
    35→        matrix[i - 1][j - 1] + cost,
    36→      );
    37→    }
    38→  }
    39→  return matrix[b.length][a.length];
    40→}

    42→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    43→  const ua = userAnswer.trim().toLowerCase();
    44→  const ca = correctAnswer.trim().toLowerCase();
    45→  if (ua === ca) return true;
    46→  if (ua.includes(ca) || ca.includes(ua)) return true;
    47→  // Levenshtein distance for typo tolerance
    48→  const dist = levenshteinDistance(ua, ca);
    49→  // Allow 1 typo for short answers, 2 for longer ones
    50→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    51→  if (dist <= threshold) return true;
    52→  // Word-level check: all words of correct answer present in user answer (in any order)
    53→  const correctWords = ca.split(/\s+/);
    54→  const userWords = ua.split(/\s+/);
    55→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    56→  return false;
    57→}

   236→// ---------- Matching question component (drag-and-drop) ----------
   237→function MatchingInput({
   238→  pairs,
   239→  onAnswer,
   240→}: {
   241→  pairs: Array<{ left: string; right: string }>;
   242→  onAnswer: (answer: string) => void;
   243→}) {
   244→  const shuffledRight = useMemo(() => {
   245→    const arr = [...pairs.map((p) => p.right)];
   246→    for (let i = arr.length - 1; i > 0; i--) {
   247→      const j = Math.floor(Math.random() * (i + 1));
   248→      [arr[i], arr[j]] = [arr[j], arr[i]];
   249→    }
   250→    return arr;
   251→  }, [pairs]);
   252→
   253→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   254→  const [matches, setMatches] = useState<Record<string, string>>({});
   255→
   256→  const handleLeftClick = (left: string) => {
   257→    setSelectedLeft(left);
   258→  };
   259→
   260→  const handleRightClick = (right: string) => {
   261→    if (!selectedLeft) return;
   262→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   263→    setSelectedLeft(null);
   264→  };
   265→
   266→  const handleRemoveMatch = (left: string) => {
   267→    setMatches((prev) => {
   268→      const next = { ...prev };
   269→      delete next[left];
   270→      return next;
   271→    });
   272→  };
   273→
   274→  const handleSubmit = () => {
   275→    const answer = Object.entries(matches)
   276→      .map(([l, r]) => `${l}-${r}`)
   277→      .join(', ');
   278→    onAnswer(answer);
   279→  };
   280→
   281→  const allMatched = pairs.every((p) => matches[p.left]);
   282→  const usedRights = new Set(Object.values(matches));
   283→
   284→  return (
   285→    <div className="space-y-4">
   286→      <p className="text-sm text-muted-foreground">
   287→        Click a left item, then click its match on the right. Click a matched item to remove it.
   288→      </p>
   289→      <div className="grid grid-cols-2 gap-4">
   290→        {/* Left column */}
   291→        <div className="flex flex-col gap-2">
   292→          {pairs.map((p) => (
   293→            <motion.button
   294→              key={p.left}
   295→              whileTap={{ scale: 0.98 }}
   296→              onClick={() =>
   297→                matches[p.left]
   298→                  ? handleRemoveMatch(p.left)
   299→                  : handleLeftClick(p.left)
   300→              }
   301→              className={`rounded-lg border p-3 text-left text-sm font-medium transition-all ${
   302→                selectedLeft === p.left
   303→                  ? 'border-primary bg-primary/10 text-primary ring-1 ring-primary/30'
   304→                  : matches[p.left]
   305→                    ? 'border-emerald-500/50 bg-emerald-500/5 text-emerald-700 dark:text-emerald-400'
   306→                    : 'border-border hover:border-primary/30 hover:bg-primary/5'
   307→              }`}
   308→            >
   309→              <div className="flex items-center gap-2">
   310→                <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40" />
   311→                <span className="flex-1">{p.left}</span>
   312→                {matches[p.left] && (
   313→                  <span className="text-xs opacity-60 truncate max-w-[120px]">
   314→                    → {matches[p.left]}
   315→                  </span>
   316→                )}
   317→              </div>
   318→            </motion.button>
   319→          ))}
   320→        </div>
   321→        {/* Right column */}
   322→        <div className="flex flex-col gap-2">
   323→          {shuffledRight.map((r) => {
   324→            const isUsed = usedRights.has(r);
   325→            return (
   326→              <motion.button
   327→                key={r}
   328→                whileTap={{ scale: 0.98 }}
   329→                onClick={() => !isUsed && handleRightClick(r)}
   330→                disabled={isUsed || !selectedLeft}
   331→                className={`rounded-lg border p-3 text-left text-sm transition-all ${
   332→                  isUsed
   333→                    ? 'border-border bg-muted/50 text-muted-foreground line-through opacity-50'
   334→                    : selectedLeft
   335→                      ? 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-pointer'
   336→                      : 'border-border opacity-60'
   337→                }`}
   338→              >
   339→                <div className="flex items-center gap-2">
   340→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40" />
   341→                  {r}
   342→                </div>
   343→              </motion.button>
   344→            );
   345→          })}
   346→        </div>
   347→      </div>
   348→      <div className="flex items-center justify-between">
   349→        <span className="text-xs text-muted-foreground">
   350→          {Object.keys(matches).length} of {pairs.length} matched
   351→        </span>
   352→        <Button onClick={handleSubmit} disabled={!allMatched} size="sm">
   353→          Submit Matches
   354→        </Button>
   355→      </div>
   356→    </div>
   357→  );
   358→}

   550→                const wasCorrect = answered[q.id] && isCorrect(q, answers[q.id] || '');
   551→                return (
   552→                  <div key={q.id} className="flex items-center gap-2 text-xs px-2 py-1.5 rounded-md bg-muted/50">
   553→                    <span className={`flex h-5 w-5 items-center justify-center rounded-full text-[10px] font-bold ${wasCorrect ? 'bg-emerald-500 text-white' : 'bg-destructive/20 text-destructive'}`}>
   554→                      {wasCorrect ? <CheckCircle2 className="h-3 w-3" /> : <XCircle className="h-3 w-3" />}
   555→                    </span>
   556→                    <span className="flex-1 truncate">{q.question.slice(0, 60)}{q.question.length > 60 ? '...' : ''}</span>
   557→                    <Badge variant="outline" className="text-[9px] px-1 py-0">{q.type.replace('_', ' ')}</Badge>
   558→                  </div>
   559→                );
   560→              })}
   561→            </div>
   562→          </div>
   563→
   564→          <div className="flex justify-center gap-3">
   565→            <Button onClick={handleRetry} variant="outline">
   566→              <RotateCcw className="h-4 w-4 mr-2" />
   567→              Try Again
   568→            </Button>
   569→            <Button onClick={() => navigate('dashboard')}>
   570→              <ArrowLeft className="h-4 w-4 mr-2" />
   571→              Back to Dashboard
   572→            </Button>
   573→          </div>
   574→        </div>
   575→      </motion.div>
   576→    );
   577→  }
   578→
   579→  // ---------- Question Screen ----------
   580→  return (
   581→    <div className="pl-14 lg:pl-0">
   582→      <div className="relative">
   583→        {/* Confetti */}
   584→        <AnimatePresence>
   585→          {showConfetti && (
   586→            <div className="fixed inset-0 pointer-events-none z-50">
   587→              {Array.from({ length: 16 }).map((_, i) => (

   589→                  key={i}
   590→                  delay={i * 0.03}
   591→                  color={CONFETTI_COLORS[i % CONFETTI_COLORS.length]}

   594→            </div>
   595→          )}

   597→
   598→        {/* Course filter + Progress bar */}
   599→        <div className="mb-6 space-y-3">
   600→          {/* Course filter tabs */}
   601→          <div className="flex items-center gap-2 overflow-x-auto pb-1">
   602→            {COURSE_QUIZ_GROUPS.map((group) => (
   603→              <button
   604→                key={group.id}
   605→                onClick={() => handleCourseChange(group.id)}
   606→                className={`flex items-center gap-1.5 rounded-full px-3 py-1.5 text-xs font-medium whitespace-nowrap transition-all ${
   607→                  selectedCourse === group.id
   608→                    ? 'bg-primary text-primary-foreground shadow-sm'
   609→                    : 'border border-border text-muted-foreground hover:text-foreground hover:border-primary/30'

   611→              >
   612→                <group.icon className="h-3.5 w-3.5" />

   614→                <span className={`ml-0.5 text-[10px] ${selectedCourse === group.id ? 'bg-white/20 px-1 rounded' : 'text-muted-foreground'}`}>

   617→                    : allQuestions.filter((q) => q.courseId === group.id).length}
   618→                </span>
   619→              </button>
   620→            ))}
   621→          </div>
   622→          <div className="flex items-center justify-between mb-2">
   623→            <span className="text-sm font-medium">
   624→              Question {currentIndex + 1} of {questions.length}
   625→            </span>
   626→            <div className="flex items-center gap-2">
   627→              <Badge variant="outline" className="text-[10px] px-1.5">
   628→                {currentQ?.difficulty}
   629→              </Badge>
   630→              <span className="text-xs text-muted-foreground bg-primary/10 px-2 py-0.5 rounded-full font-medium">
   631→                {currentQ?.type.replace('_', ' ')}
   632→              </span>
   633→            </div>
   634→          </div>
   635→          <Progress value={progress} className="h-2" />
   636→        </div>
   637→
   638→        {/* Question card */}
   639→        <AnimatePresence mode="wait">
   640→          {currentQ && (
   641→            <motion.div
   642→              key={currentQ.id}
   643→              initial={{ opacity: 0, x: 30 }}
   644→              animate={{ opacity: 1, x: 0 }}
   645→              exit={{ opacity: 0, x: -30 }}
   646→              transition={{ duration: 0.3 }}
   647→              className="glass rounded-xl p-6 space-y-6"

   649→              {/* Concept tag */}
   650→              {currentQ.concept && (
   651→                <div className="flex items-center gap-2">
   652→                  <Badge variant="secondary" className="text-xs">
   653→                    {currentQ.concept}
   654→                  </Badge>
   655→                </div>
   656→              )}
   657→
   658→              {/* Question text */}
   659→              <h2 className="text-lg font-semibold leading-relaxed">{currentQ.question}</h2>
   660→
   661→              {/* Question type renderer */}
   662→              {currentQ.type === 'multiple_choice' && currentQ.options && (
   663→                <div className="grid gap-2">
   664→                  {currentQ.options.map((opt, i) => {
   665→                    const letter = String.fromCharCode(65 + i);
   666→                    const isAnswered = answered[currentQ.id];
   667→                    const isSelected = answers[currentQ.id] === opt;
   668→                    const isCorrectOpt = opt === currentQ.answer;
   669→
   670→                    return (
   671→                      <motion.button
   672→                        key={opt}
   673→                        whileHover={!isAnswered ? { scale: 1.01 } : {}}
   674→                        whileTap={!isAnswered ? { scale: 0.99 } : {}}
   675→                        onClick={() => !isAnswered && handleAnswer(opt)}
   676→                        disabled={isAnswered}
   677→                        className={`flex items-center gap-3 rounded-lg border p-4 text-left text-sm transition-all ${
   678→                          isAnswered
   679→                            ? isCorrectOpt
   680→                              ? 'border-emerald-500 bg-emerald-500/10'
   681→                              : isSelected
   682→                                ? 'border-destructive bg-destructive/10'
   683→                                : 'border-border opacity-60'
   684→                            : 'border-border hover:border-primary/50 hover:bg-primary/5 cursor-pointer'

   686→                      >
   687→                        <span
   688→                          className={`flex h-7 w-7 shrink-0 items-center justify-center rounded-md text-xs font-bold ${
   689→                            isAnswered && isCorrectOpt
   690→                              ? 'bg-emerald-500 text-white'
   691→                              : isAnswered && isSelected
   692→                                ? 'bg-destructive text-white'
   693→                                : 'bg-muted text-muted-foreground'
   694→                          }`}
   695→                        >
   696→                          {isAnswered && isCorrectOpt ? (
   697→                            <CheckCircle2 className="h-4 w-4" />
   698→                          ) : isAnswered && isSelected ? (
   699→                            <XCircle className="h-4 w-4" />
   700→                          ) : (
   701→                            letter
   702→                          )}
   703→                        </span>
   704→                        <span>{opt}</span>
   705→                      </motion.button>
   706→                    );
   707→                  })}
   708→                </div>
   709→              )}
   710→
   711→              {currentQ.type === 'true_false' && (
   712→                <div className="flex gap-3">
   713→                  {['True', 'False'].map((opt) => {
   714→                    const isAnswered = answered[currentQ.id];
   715→                    const isSelected = answers[currentQ.id] === opt;
   716→                    const isCorrectOpt = opt === currentQ.answer;
   717→
   718→                    return (
   719→                      <motion.button
   720→                        key={opt}
   721→                        whileHover={!isAnswered ? { scale: 1.02 } : {}}
   722→                        whileTap={!isAnswered ? { scale: 0.98 } : {}}
   723→                        onClick={() => !isAnswered && handleAnswer(opt)}
   724→                        disabled={isAnswered}
   725→                        className={`flex-1 rounded-lg border p-4 text-center font-semibold transition-all ${
   726→                          isAnswered
   727→                            ? isCorrectOpt
   728→                              ? 'border-emerald-500 bg-emerald-500/10 text-emerald-700 dark:text-emerald-400'
   729→                              : isSelected
   730→                                ? 'border-destructive bg-destructive/10 text-destructive'

   732→                            : 'border-border hover:border-primary/50 hover:bg-primary/5 cursor-pointer'
   733→                        }`}
   734→                      >
   735→                        {opt}
   736→                      </motion.button>
   737→                    );
   738→                  })}
   739→                </div>
   740→              )}
   741→
   742→              {currentQ.type === 'short_answer' && !answered[currentQ.id] && (
   743→                <div className="flex gap-2">
   744→                  <Input
   745→                    placeholder="Type your answer..."
   746→                    onKeyDown={(e) => {
   747→                      if (e.key === 'Enter' && (e.target as HTMLInputElement).value.trim()) {
   748→                        handleAnswer((e.target as HTMLInputElement).value);

   750→                    }}
   751→                    id="short-answer-input"
   752→                  />
   753→                  <Button
   754→                    onClick={() => {
   755→                      const input = document.getElementById('short-answer-input') as HTMLInputElement;

   757→                    }}
   758→                  >
   759→                    Submit
   760→                  </Button>
   761→                </div>
   762→              )}
   763→
   764→              {currentQ.type === 'short_answer' && answered[currentQ.id] && (
   765→                <div className="rounded-lg border p-4 text-sm space-y-1">
   766→                  <p className="text-muted-foreground">Your answer:</p>
   767→                  <p className="font-medium">{answers[currentQ.id]}</p>
   768→                </div>
   769→              )}
   770→
   771→              {currentQ.type === 'fill_blank' && !answered[currentQ.id] && (
   772→                <div className="flex gap-2 items-center">
   773→                  <Input
   774→                    placeholder="Fill in the blank..."
   775→                    onKeyDown={(e) => {
   776→                      if (e.key === 'Enter' && (e.target as HTMLInputElement).value.trim()) {
   777→                        handleAnswer((e.target as HTMLInputElement).value);
   778→                      }
   779→                    }}
   780→                    id="fill-blank-input"
   781→                  />
   782→                  <Button
   783→                    onClick={() => {
   784→                      const input = document.getElementById('fill-blank-input') as HTMLInputElement;
   785→                      if (input?.value.trim()) handleAnswer(input.value);
   786→                    }}
   787→                  >
   788→                    Submit
   789→                  </Button>
   790→                </div>
   791→              )}
   792→
   793→              {currentQ.type === 'fill_blank' && answered[currentQ.id] && (
   794→                <div className="rounded-lg border p-4 text-sm space-y-1">
   795→                  <p className="text-muted-foreground">Your answer:</p>
   796→                  <p className="font-medium">{answers[currentQ.id]}</p>
   797→                </div>
   798→              )}
   799→
   800→              {currentQ.type === 'matching' && currentQ.matchingPairs && !answered[currentQ.id] && (
   801→                <MatchingInput
   802→                  pairs={currentQ.matchingPairs}
   803→                  onAnswer={(answer) => handleAnswer(answer)}
   804→                />
   805→              )}
   806→
   807→              {currentQ.type === 'error_correction' && currentQ.errorText && !answered[currentQ.id] && (
   808→                <ErrorCorrectionInput
   809→                  errorText={currentQ.errorText}
   810→                  onAnswer={(answer) => handleAnswer(answer)}
   811→                />
   812→              )}
   813→
   814→              {currentQ.type === 'error_correction' && answered[currentQ.id] && (
   815→                <div className="rounded-lg border p-4 text-sm space-y-1">
   816→                  <p className="text-muted-foreground">Your correction:</p>
   817→                  <p className="font-medium">{answers[currentQ.id] || '(empty)'}</p>
   818→                </div>
   819→              )}
   820→
   821→              {/* Explanation */}
   822→              <AnimatePresence>
   823→                {showExplanation && currentQ.explanation && (
   824→                  <motion.div
   825→                    initial={{ opacity: 0, y: 10, height: 0 }}
   826→                    animate={{ opacity: 1, y: 0, height: 'auto' }}
   827→                    exit={{ opacity: 0, y: -10, height: 0 }}
   828→                    className="overflow-hidden"
   829→                  >
   830→                    <div className="rounded-lg border border-primary/20 bg-primary/5 p-4 space-y-2">
   831→                      <div className="flex items-center gap-2">
   832→                        <Sparkles className="h-4 w-4 text-primary" />
   833→                        <span className="text-sm font-semibold text-primary">Explanation</span>
   834→                      </div>
   835→                      <p className="text-sm text-muted-foreground leading-relaxed">
   836→                        {currentQ.explanation}
   837→                      </p>
   838→                      {currentQ.type === 'short_answer' || currentQ.type === 'error_correction' || currentQ.type === 'fill_blank' ? (
   839→                        <div className="mt-2">
   840→                          <span className="text-xs text-muted-foreground">Expected answer: </span>
   841→                          <span className="text-xs font-medium">{currentQ.answer}</span>
   842→                        </div>
   843→                      ) : null}
   844→                    </div>
   845→                  </motion.div>
   846→                )}
   847→              </AnimatePresence>
   848→
   849→              {/* Navigation buttons */}
   850→              <div className="flex items-center justify-between">
   851→                <Button
   852→                  variant="ghost"
   853→                  size="sm"
   854→                  onClick={() => setCurrentIndex(Math.max(0, currentIndex - 1))}
   855→                  disabled={currentIndex === 0}
   856→                  className="text-muted-foreground"
   857→                >
   858→                  Previous
   859→                </Button>
   860→                {answered[currentQ.id] && (
   861→                  <motion.div
   862→                    initial={{ opacity: 0, y: 10 }}
   863→                    animate={{ opacity: 1, y: 0 }}
   864→                  >
   865→                    <Button onClick={handleNext}>
   866→                      {currentIndex < questions.length - 1 ? (
   867→                        <>
   868→                          Next <ChevronRight className="h-4 w-4 ml-1" />
   869→                        </>
   870→                      ) : (
   871→                        <>
   872→                          See Results <Trophy className="h-4 w-4 ml-1" />
   873→                        </>
   874→                      )}
   875→                    </Button>
   876→                  </motion.div>
   877→                )}
   878→              </div>
   879→            </motion.div>
   880→          )}
   881→        </AnimatePresence>
   882→
   883→        {/* Question navigator dots */}
   884→        {questions.length > 1 && (
   885→          <div className="flex items-center justify-center gap-1 mt-6 flex-wrap">
   886→            {questions.map((q, i) => (
   887→              <button
   888→                key={q.id}
   889→                onClick={() => setCurrentIndex(i)}
   890→                className={`h-2.5 w-2.5 rounded-full transition-all ${
   891→                  i === currentIndex
   892→                    ? 'bg-primary w-6'
   893→                    : answered[q.id]
   894→                      ? isCorrect(q, answers[q.id] || '')
   895→                        ? 'bg-emerald-500'
   896→                        : 'bg-destructive/60'
   897→                      : 'bg-muted hover:bg-primary/40'
   898→                }`}
   899→                aria-label={`Go to question ${i + 1}`}
   900→              />
   901→            ))}
   902→          </div>
   903→        )}
   904→      </div>
   905→    </div>
   906→  );
