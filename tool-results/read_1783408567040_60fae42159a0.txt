    48→// ---------- Typo-tolerance helper ----------
    49→function levenshteinDistance(a: string, b: string): number {
    50→  const matrix: number[][] = [];
    51→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    52→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    53→  for (let i = 1; i <= b.length; i++) {
    54→    for (let j = 1; j <= a.length; j++) {
    55→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    56→      matrix[i][j] = Math.min(
    57→        matrix[i - 1][j] + 1,
    58→        matrix[i][j - 1] + 1,
    59→        matrix[i - 1][j - 1] + cost,
    60→      );
    61→    }
    62→  }
    63→  return matrix[b.length][a.length];
    64→}

    66→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    67→  const ua = userAnswer.trim().toLowerCase();
    68→  const ca = correctAnswer.trim().toLowerCase();
    69→  if (ua === ca) return true;
    70→  if (ua.includes(ca) || ca.includes(ua)) return true;
    71→  // Levenshtein distance for typo tolerance
    72→  const dist = levenshteinDistance(ua, ca);
    73→  // Allow 1 typo for short answers, 2 for longer ones
    74→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    75→  if (dist <= threshold) return true;
    76→  // Word-level check: all words of correct answer present in user answer (in any order)
    77→  const correctWords = ca.split(/\s+/);
    78→  const userWords = ua.split(/\s+/);
    79→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    80→  return false;
    81→}

    83→// ---------- Fill-in-blank grading with Levenshtein tolerance ----------
    84→type FillBlankGrade = { status: 'correct' | 'close' | 'wrong'; points: number; message: string };
    85→
    86→function gradeFillBlank(userAnswer: string, correctAnswer: string, hintUsed: boolean = false): FillBlankGrade {
    87→  const ua = userAnswer.trim().toLowerCase();
    88→  const ca = correctAnswer.trim().toLowerCase();
    89→  if (ua === ca) {
    90→    return { status: 'correct', points: hintUsed ? 0.75 : 1, message: '' };
    91→  }
    92→  const dist = levenshteinDistance(ua, ca);
    93→  let basePoints = 0;
    94→  if (dist <= 2) basePoints = 0.8;
    95→  else if (dist <= 3) basePoints = 0.5;
    96→  if (basePoints > 0) {
    97→    return { status: 'close', points: basePoints * (hintUsed ? 0.75 : 1), message: `Close! Did you mean: "${correctAnswer}"?` };
    98→  }
    99→  return { status: 'wrong', points: 0, message: '' };
   100→}

   268→  const streak = loadDailyStreak();

   270→

   276→
   277→  let newCurrent: number;
   278→  if (streak.lastDate === yesterdayStr) {
   279→    newCurrent = streak.current + 1;
   280→  } else {
   281→    newCurrent = 1;
   282→  }
   283→
   284→  const newBest = Math.max(streak.best, newCurrent);
   285→  const updated = { current: newCurrent, best: newBest, lastDate: today };
   286→  saveDailyStreak(updated);
   287→  return updated;
   288→}

   290→function getTimeUntilMidnight(): { hours: number; minutes: number; seconds: number } {
   291→  const now = new Date();
   292→  const midnight = new Date(now);
   293→  midnight.setHours(24, 0, 0, 0);
   294→  const diff = midnight.getTime() - now.getTime();
   295→  return {
   296→    hours: Math.floor(diff / (1000 * 60 * 60)),
   297→    minutes: Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60)),
   298→    seconds: Math.floor((diff % (1000 * 60)) / 1000),
   299→  };
   300→}

   352→function scoreQuestionForAdaptive(input: AdaptiveScoringInput): ScoredQuestion {
   353→  const { question, masteryMap, adaptiveResults, dueConcepts } = input;
   354→  let score = 0;
   355→  const reasons: string[] = [];
   356→  const concept = question.concept || 'Unknown';
   357→  const mastery = masteryMap[concept];
   358→
   359→  // 1. Low mastery bonus (level 1-2)
   360→  if (mastery && mastery.level < 3) {
   361→    score += 3;
   362→    reasons.push('low mastery');
   363→  }
   364→
   365→  // 2. Spaced repetition due for review
   366→  if (dueConcepts.has(concept)) {
   367→    score += 2;
   368→    reasons.push('needs review');
   369→  }
   370→
   371→  // 3. Recent performance: last 5 results on concept
   372→  const conceptResults = adaptiveResults
   373→    .filter((r) => r.concept === concept)
   374→    .slice(-5);
   375→  const last3 = conceptResults.slice(-3);
   376→  const wrongCount = last3.filter((r) => !r.correct).length;
   377→  const correctCount = conceptResults.filter((r) => r.correct).length;
   378→
   379→  if (wrongCount >= 2) {
   380→    score += 2;
   381→    reasons.push('recent struggles');
   382→  }
   383→
   384→  // 4. Difficulty match bonus
   385→  const targetLevel = mastery ? Math.min(mastery.level + 1, 3) : 1;
   386→  const targetDifficulty = targetLevel === 1 ? 'easy' : targetLevel === 2 ? 'medium' : 'hard';
   387→  if (question.difficulty === targetDifficulty) {
   388→    score += 1;
   389→  } else if (
   390→    (targetDifficulty === 'medium' && question.difficulty === 'easy') ||
   391→    (targetDifficulty === 'hard' && question.difficulty === 'medium')
   392→  ) {
   393→    score += 0.5;
   394→  }
   395→
   396→  // 5. Boost for concepts with many wrong answers (weaker areas)
   397→  if (correctCount >= 3) {
   398→    score += 1;
   399→    reasons.push('ready for harder questions');
   400→  }
   401→
   402→  return { question, score, reasons };
   403→}

   405→function getAdaptiveReasoning(scored: ScoredQuestion[]): string {
   406→  if (scored.length === 0) return '';
   407→
   408→  const conceptReasons = new Map<string, string[]>();
   409→  for (const sq of scored.slice(0, 5)) {
   410→    const concept = sq.question.concept || 'Unknown';
   411→    const existing = conceptReasons.get(concept) || [];
   412→    conceptReasons.set(concept, [...existing, ...sq.reasons]);
   413→  }
   414→
   415→  const parts: string[] = [];
   416→  for (const [concept, reasons] of conceptReasons) {
   417→    const uniqueReasons = [...new Set(reasons)];
   418→    const desc = uniqueReasons.map((r) => {
   419→      switch (r) {
   420→        case 'low mastery': return 'low mastery';
   421→        case 'needs review': return 'needs review';
   422→        case 'recent struggles': return 'struggling recently';
   423→        case 'ready for harder questions': return 'ready for challenge';
   424→        default: return r;
   425→      }
   426→    }).join(', ');
   427→    parts.push(`${concept} (${desc})`);
   428→  }
   429→
   430→  return 'Focusing on: ' + parts.join(', ');
   431→}

   497→// ---------- Animated counter hook ----------
   498→function useAnimatedCounter(target: number, duration: number = 1200) {
   499→  const [count, setCount] = useState(0);
   500→  const rafRef = useRef<number>(0);
   501→  useEffect(() => {
   502→    const start = performance.now();
   503→    const tick = (now: number) => {
   504→      const elapsed = now - start;
   505→      const progress = Math.min(elapsed / duration, 1);
   506→      const eased = 1 - Math.pow(1 - progress, 3);
   507→      setCount(Math.round(eased * target));
   508→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   509→    };
   510→    rafRef.current = requestAnimationFrame(tick);
   511→    return () => cancelAnimationFrame(rafRef.current);
   512→  }, [target, duration]);
   513→  return count;
   514→}

   567→// ---------- Matching question component (drag-and-drop with SVG connectors) ----------
   568→function MatchingInput({
   569→  pairs,
   570→  onAnswer,
   571→}: {
   572→  pairs: Array<{ left: string; right: string }>;
   573→  onAnswer: (answer: string) => void;
   574→}) {
   575→  const shuffledRight = useMemo(() => {
   576→    const arr = [...pairs.map((p) => p.right)];
   577→    for (let i = arr.length - 1; i > 0; i--) {
   578→      const j = Math.floor(Math.random() * (i + 1));
   579→      [arr[i], arr[j]] = [arr[j], arr[i]];
   580→    }
   581→    return arr;
   582→  }, [pairs]);
   583→
   584→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   585→  const [matches, setMatches] = useState<Record<string, string>>({});
   586→  const [dragItem, setDragItem] = useState<string | null>(null);
   587→  const [dragOverTarget, setDragOverTarget] = useState<string | null>(null);
   588→  const [linePositions, setLinePositions] = useState<Array<{ id: string; x1: number; y1: number; x2: number; y2: number }>>([]);
   589→  const containerRef = useRef<HTMLDivElement>(null);
   590→
   591→  // Build index maps for safe DOM queries
   592→  const leftIdxMap = useMemo(() => {
   593→    const map: Record<string, number> = {};
   594→    pairs.forEach((p, i) => { map[p.left] = i; });
   595→    return map;
   596→  }, [pairs]);
   597→
   598→  const rightIdxMap = useMemo(() => {
   599→    const map: Record<string, number> = {};
   600→    shuffledRight.forEach((r, i) => { map[r] = i; });
   601→    return map;
   602→  }, [shuffledRight]);
   603→
   604→  // Compute SVG connector line positions
   605→  const updateLinePositions = useCallback(() => {
   606→    const container = containerRef.current;
   607→    if (!container) return;
   608→
   609→    requestAnimationFrame(() => {
   610→      if (!container) return;
   611→      const containerRect = container.getBoundingClientRect();
   612→      const positions: Array<{ id: string; x1: number; y1: number; x2: number; y2: number }> = [];
   613→
   614→      for (const [left, right] of Object.entries(matches)) {
   615→        const lIdx = leftIdxMap[left];
   616→        const rIdx = rightIdxMap[right];
   617→        if (lIdx === undefined || rIdx === undefined) continue;
   618→
   619→        const leftEl = container.querySelector(`[data-left-idx="${lIdx}"]`);
   620→        const rightEl = container.querySelector(`[data-right-idx="${rIdx}"]`);
   621→
   622→        if (leftEl && rightEl) {
   623→          const leftRect = leftEl.getBoundingClientRect();
   624→          const rightRect = rightEl.getBoundingClientRect();
   625→
   626→          positions.push({
   627→            id: `${left}-${right}`,
   628→            x1: leftRect.right - containerRect.left,
   629→            y1: leftRect.top + leftRect.height / 2 - containerRect.top,
   630→            x2: rightRect.left - containerRect.left,
   631→            y2: rightRect.top + rightRect.height / 2 - containerRect.top,
   632→          });
   633→        }
   634→      }
   635→
   636→      setLinePositions(positions);
   637→    });
   638→  }, [matches, leftIdxMap, rightIdxMap]);
   639→
   640→  // Recalculate on matches change
   641→  useEffect(() => {
   642→    updateLinePositions();
   643→  }, [updateLinePositions]);
   644→
   645→  // Recalculate on resize
   646→  useEffect(() => {
   647→    const container = containerRef.current;
   648→    if (!container) return;
   649→
   650→    const observer = new ResizeObserver(updateLinePositions);
   651→    observer.observe(container);
   652→
   653→    return () => observer.disconnect();
   654→  }, [updateLinePositions]);
   655→
   656→  const handleLeftClick = (left: string) => {
   657→    if (matches[left]) {
   658→      // Remove match by clicking connected left item
   659→      setMatches((prev) => {
   660→        const next = { ...prev };
   661→        delete next[left];
   662→        return next;
   663→      });
   664→      return;
   665→    }
   666→    setSelectedLeft(left);
   667→  };
   668→
   669→  const handleRightClick = (right: string) => {
   670→    if (!selectedLeft) return;
   671→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   672→    setSelectedLeft(null);
   673→  };
   674→
   675→  const handleDragStart = (e: React.DragEvent, left: string) => {
   676→    setDragItem(left);
   677→    e.dataTransfer.setData('text/plain', left);
   678→    e.dataTransfer.effectAllowed = 'link';
   679→  };
   680→
   681→  const handleDragOver = (e: React.DragEvent, right: string) => {
   682→    e.preventDefault();
   683→    e.dataTransfer.dropEffect = 'link';
   684→    const isUsedByOther = Object.entries(matches).some(
   685→      ([k, v]) => v === right && k !== dragItem,
   686→    );
   687→    if (dragItem && !isUsedByOther) {
   688→      setDragOverTarget(right);
   689→    }
   690→  };
   691→
   692→  const handleDragLeave = () => {
   693→    setDragOverTarget(null);
   694→  };
   695→
   696→  const handleDrop = (e: React.DragEvent, right: string) => {
   697→    e.preventDefault();
   698→    const left = e.dataTransfer.getData('text/plain') || dragItem;
   699→    if (left) {
   700→      // Remove any existing match for this left item first
   701→      setMatches((prev) => {
   702→        const next = { ...prev };
   703→        delete next[left];
   704→        next[left] = right;
   705→        return next;
   706→      });
   707→    }
   708→    setDragItem(null);
   709→    setDragOverTarget(null);
   710→  };
   711→
   712→  const handleDragEnd = () => {
   713→    setDragItem(null);
   714→    setDragOverTarget(null);
   715→  };
   716→
   717→  const handleSubmit = () => {
   718→    const answer = Object.entries(matches)
   719→      .map(([l, r]) => `${l}-${r}`)
   720→      .join(', ');
   721→    onAnswer(answer);
   722→  };
   723→
   724→  const allMatched = pairs.every((p) => matches[p.left]);
   725→  const usedRights = new Set(Object.values(matches));
   726→
   727→  return (
   728→    <div className="space-y-4">
   729→      <p className="text-sm text-muted-foreground">
   730→        Drag left items to their match on the right, or click to select. Click a connected left item to remove it.
   731→      </p>
   732→      <div ref={containerRef} className="relative grid grid-cols-2 gap-4">
   733→        {/* SVG Connector Overlay */}
   734→        <svg className="absolute inset-0 pointer-events-none z-10" style={{ width: '100%', height: '100%', overflow: 'visible' }}>
   735→          <defs>
   736→            <linearGradient id="connectorGrad" x1="0%" y1="0%" x2="100%" y2="0%">
   737→              <stop offset="0%" stopColor="#10b981" />
   738→              <stop offset="50%" stopColor="#14b8a6" />
   739→              <stop offset="100%" stopColor="#0d9488" />
   740→            </linearGradient>
   741→          </defs>
   742→          <AnimatePresence>
   743→            {linePositions.map((pos) => {
   744→              const midX = (pos.x1 + pos.x2) / 2;
   745→              return (
   746→                <motion.path
   747→                  key={pos.id}
   748→                  d={`M ${pos.x1} ${pos.y1} C ${midX} ${pos.y1}, ${midX} ${pos.y2}, ${pos.x2} ${pos.y2}`}
   749→                  fill="none"
   750→                  stroke="url(#connectorGrad)"
   751→                  strokeWidth="2.5"
   752→                  strokeLinecap="round"
   753→                  initial={{ pathLength: 0, opacity: 0 }}
   754→                  animate={{ pathLength: 1, opacity: 0.85 }}
   755→                  exit={{ pathLength: 0, opacity: 0 }}
   756→                  transition={{ duration: 0.5, ease: 'easeOut' }}
   757→                />
   758→              );
   759→            })}
   760→          </AnimatePresence>
   761→        </svg>
   762→
   763→        {/* Left column */}
   764→        <div className="flex flex-col gap-2">
   765→          {pairs.map((p) => {
   766→            const isMatched = !!matches[p.left];
   767→            const isDragging = dragItem === p.left;
   768→            return (
   769→              <motion.button
   770→                key={p.left}
   771→                data-left-idx={leftIdxMap[p.left]}
   772→                draggable={!isMatched}
   773→                whileTap={{ scale: 0.98 }}
   774→                onClick={() => handleLeftClick(p.left)}
   775→                onDragStart={(e) => handleDragStart(e, p.left)}
   776→                onDragEnd={handleDragEnd}
   777→                className={`rounded-lg border p-3 text-left text-sm font-medium transition-all ${
   778→                  isDragging
   779→                    ? 'border-primary/50 bg-primary/5 opacity-50 scale-95'
   780→                    : selectedLeft === p.left
   781→                      ? 'border-primary bg-primary/10 text-primary ring-1 ring-primary/30 shadow-sm shadow-primary/10'
   782→                      : isMatched
   783→                        ? 'border-emerald-500/50 bg-emerald-500/5 text-emerald-700 dark:text-emerald-400 shadow-sm shadow-emerald-500/5'
   784→                        : 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-grab active:cursor-grabbing'
   785→                }`}
   786→              >
   787→                <div className="flex items-center gap-2">
   788→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40 shrink-0" />
   789→                  <span className="flex-1">{p.left}</span>
   790→                  {isMatched && (
   791→                    <motion.span
   792→                      initial={{ opacity: 0, x: 10 }}
   793→                      animate={{ opacity: 1, x: 0 }}
   794→                      className="text-xs opacity-70 truncate max-w-[120px] text-emerald-600 dark:text-emerald-400"
   795→                    >
   796→                      {matches[p.left]}
   797→                    </motion.span>
   798→                  )}
   799→                </div>
   800→              </motion.button>
   801→            );
   802→          })}
   803→        </div>
   804→        {/* Right column */}
   805→        <div className="flex flex-col gap-2">
   806→          {shuffledRight.map((r) => {
   807→            const isUsed = usedRights.has(r);
   808→            const isDragOver = dragOverTarget === r && !isUsed;
   809→            return (
   810→              <motion.button
   811→                key={r}
   812→                data-right-idx={rightIdxMap[r]}
   813→                whileTap={{ scale: 0.98 }}
   814→                onClick={() => !isUsed && handleRightClick(r)}
   815→                onDragOver={(e) => handleDragOver(e, r)}
   816→                onDragLeave={handleDragLeave}
   817→                onDrop={(e) => handleDrop(e, r)}
   818→                disabled={isUsed}
   819→                className={`rounded-lg border p-3 text-left text-sm transition-all ${
   820→                  isDragOver
   821→                    ? 'border-emerald-500/60 bg-emerald-500/10 ring-2 ring-emerald-500/30 shadow-lg shadow-emerald-500/20 scale-[1.02]'
   822→                    : isUsed
   823→                      ? 'border-border bg-muted/50 text-muted-foreground line-through opacity-50'
   824→                      : selectedLeft
   825→                        ? 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-pointer hover:shadow-sm hover:shadow-primary/5'
   826→                        : 'border-border opacity-60'
   827→                }`}
   828→              >
   829→                <div className="flex items-center gap-2">
   830→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40 shrink-0" />
   831→                  <span className="flex-1">{r}</span>
   832→                </div>
   833→              </motion.button>
   834→            );
   835→          })}
   836→        </div>
   837→      </div>
   838→      <div className="flex items-center justify-between">
   839→        <div className="flex items-center gap-2">
   840→          <div className="relative h-1.5 w-24 rounded-full bg-muted/50 overflow-hidden">
   841→            <motion.div
   842→              className="absolute inset-y-0 left-0 rounded-full"
   843→              style={{ background: 'linear-gradient(90deg, oklch(0.627 0.194 149.214), oklch(0.687 0.159 177.89))' }}
   844→              animate={{ width: `${(Object.keys(matches).length / pairs.length) * 100}%` }}
   845→              transition={{ duration: 0.3 }}
   846→            />
   847→          </div>
   848→          <span className="text-xs text-muted-foreground">
   849→            {Object.keys(matches).length}/{pairs.length} matched
   850→          </span>
   851→        </div>
   852→        <motion.div
   853→          whileHover={{ scale: 1.05 }}
   854→          whileTap={{ scale: 0.95 }}
   855→        >
   856→          <Button onClick={handleSubmit} disabled={!allMatched} size="sm" className={allMatched ? 'glow-emerald' : ''}>
   857→            Submit Matches
   858→          </Button>
   859→        </motion.div>
   860→      </div>
   861→    </div>
   862→  );
