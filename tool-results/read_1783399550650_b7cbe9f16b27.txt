    34→// ---------- Typo-tolerance helper ----------
    35→function levenshteinDistance(a: string, b: string): number {
    36→  const matrix: number[][] = [];
    37→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    38→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    39→  for (let i = 1; i <= b.length; i++) {
    40→    for (let j = 1; j <= a.length; j++) {
    41→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    42→      matrix[i][j] = Math.min(
    43→        matrix[i - 1][j] + 1,
    44→        matrix[i][j - 1] + 1,
    45→        matrix[i - 1][j - 1] + cost,
    46→      );
    47→    }
    48→  }
    49→  return matrix[b.length][a.length];
    50→}

    52→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    53→  const ua = userAnswer.trim().toLowerCase();
    54→  const ca = correctAnswer.trim().toLowerCase();
    55→  if (ua === ca) return true;
    56→  if (ua.includes(ca) || ca.includes(ua)) return true;
    57→  // Levenshtein distance for typo tolerance
    58→  const dist = levenshteinDistance(ua, ca);
    59→  // Allow 1 typo for short answers, 2 for longer ones
    60→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    61→  if (dist <= threshold) return true;
    62→  // Word-level check: all words of correct answer present in user answer (in any order)
    63→  const correctWords = ca.split(/\s+/);
    64→  const userWords = ua.split(/\s+/);
    65→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    66→  return false;
    67→}

   212→// ---------- Animated counter hook ----------
   213→function useAnimatedCounter(target: number, duration: number = 1200) {
   214→  const [count, setCount] = useState(0);
   215→  const rafRef = useRef<number>(0);
   216→  useEffect(() => {
   217→    const start = performance.now();
   218→    const tick = (now: number) => {
   219→      const elapsed = now - start;
   220→      const progress = Math.min(elapsed / duration, 1);
   221→      const eased = 1 - Math.pow(1 - progress, 3);
   222→      setCount(Math.round(eased * target));
   223→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   224→    };
   225→    rafRef.current = requestAnimationFrame(tick);
   226→    return () => cancelAnimationFrame(rafRef.current);
   227→  }, [target, duration]);
   228→  return count;
   229→}

   282→// ---------- Matching question component (drag-and-drop with SVG connectors) ----------
   283→function MatchingInput({
   284→  pairs,
   285→  onAnswer,
   286→}: {
   287→  pairs: Array<{ left: string; right: string }>;
   288→  onAnswer: (answer: string) => void;
   289→}) {
   290→  const shuffledRight = useMemo(() => {
   291→    const arr = [...pairs.map((p) => p.right)];
   292→    for (let i = arr.length - 1; i > 0; i--) {
   293→      const j = Math.floor(Math.random() * (i + 1));
   294→      [arr[i], arr[j]] = [arr[j], arr[i]];
   295→    }
   296→    return arr;
   297→  }, [pairs]);
   298→
   299→  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   300→  const [matches, setMatches] = useState<Record<string, string>>({});
   301→  const [dragItem, setDragItem] = useState<string | null>(null);
   302→  const [dragOverTarget, setDragOverTarget] = useState<string | null>(null);
   303→  const [linePositions, setLinePositions] = useState<Array<{ id: string; x1: number; y1: number; x2: number; y2: number }>>([]);
   304→  const containerRef = useRef<HTMLDivElement>(null);
   305→
   306→  // Build index maps for safe DOM queries
   307→  const leftIdxMap = useMemo(() => {
   308→    const map: Record<string, number> = {};
   309→    pairs.forEach((p, i) => { map[p.left] = i; });
   310→    return map;
   311→  }, [pairs]);
   312→
   313→  const rightIdxMap = useMemo(() => {
   314→    const map: Record<string, number> = {};
   315→    shuffledRight.forEach((r, i) => { map[r] = i; });
   316→    return map;
   317→  }, [shuffledRight]);
   318→
   319→  // Compute SVG connector line positions
   320→  const updateLinePositions = useCallback(() => {
   321→    const container = containerRef.current;
   322→    if (!container) return;
   323→
   324→    requestAnimationFrame(() => {
   325→      if (!container) return;
   326→      const containerRect = container.getBoundingClientRect();
   327→      const positions: Array<{ id: string; x1: number; y1: number; x2: number; y2: number }> = [];
   328→
   329→      for (const [left, right] of Object.entries(matches)) {
   330→        const lIdx = leftIdxMap[left];
   331→        const rIdx = rightIdxMap[right];
   332→        if (lIdx === undefined || rIdx === undefined) continue;
   333→
   334→        const leftEl = container.querySelector(`[data-left-idx="${lIdx}"]`);
   335→        const rightEl = container.querySelector(`[data-right-idx="${rIdx}"]`);
   336→
   337→        if (leftEl && rightEl) {
   338→          const leftRect = leftEl.getBoundingClientRect();
   339→          const rightRect = rightEl.getBoundingClientRect();
   340→
   341→          positions.push({
   342→            id: `${left}-${right}`,
   343→            x1: leftRect.right - containerRect.left,
   344→            y1: leftRect.top + leftRect.height / 2 - containerRect.top,
   345→            x2: rightRect.left - containerRect.left,
   346→            y2: rightRect.top + rightRect.height / 2 - containerRect.top,
   347→          });
   348→        }
   349→      }
   350→
   351→      setLinePositions(positions);
   352→    });
   353→  }, [matches, leftIdxMap, rightIdxMap]);
   354→
   355→  // Recalculate on matches change
   356→  useEffect(() => {
   357→    updateLinePositions();
   358→  }, [updateLinePositions]);
   359→
   360→  // Recalculate on resize
   361→  useEffect(() => {
   362→    const container = containerRef.current;
   363→    if (!container) return;
   364→
   365→    const observer = new ResizeObserver(updateLinePositions);
   366→    observer.observe(container);
   367→
   368→    return () => observer.disconnect();
   369→  }, [updateLinePositions]);
   370→
   371→  const handleLeftClick = (left: string) => {
   372→    if (matches[left]) {
   373→      // Remove match by clicking connected left item
   374→      setMatches((prev) => {
   375→        const next = { ...prev };
   376→        delete next[left];
   377→        return next;
   378→      });
   379→      return;
   380→    }
   381→    setSelectedLeft(left);
   382→  };
   383→
   384→  const handleRightClick = (right: string) => {
   385→    if (!selectedLeft) return;
   386→    setMatches((prev) => ({ ...prev, [selectedLeft]: right }));
   387→    setSelectedLeft(null);
   388→  };
   389→
   390→  const handleDragStart = (e: React.DragEvent, left: string) => {
   391→    setDragItem(left);
   392→    e.dataTransfer.setData('text/plain', left);
   393→    e.dataTransfer.effectAllowed = 'link';
   394→  };
   395→
   396→  const handleDragOver = (e: React.DragEvent, right: string) => {
   397→    e.preventDefault();
   398→    e.dataTransfer.dropEffect = 'link';
   399→    const isUsedByOther = Object.entries(matches).some(
   400→      ([k, v]) => v === right && k !== dragItem,
   401→    );
   402→    if (dragItem && !isUsedByOther) {
   403→      setDragOverTarget(right);
   404→    }
   405→  };
   406→
   407→  const handleDragLeave = () => {
   408→    setDragOverTarget(null);
   409→  };
   410→
   411→  const handleDrop = (e: React.DragEvent, right: string) => {
   412→    e.preventDefault();
   413→    const left = e.dataTransfer.getData('text/plain') || dragItem;
   414→    if (left) {
   415→      // Remove any existing match for this left item first
   416→      setMatches((prev) => {
   417→        const next = { ...prev };
   418→        delete next[left];
   419→        next[left] = right;
   420→        return next;
   421→      });
   422→    }
   423→    setDragItem(null);
   424→    setDragOverTarget(null);
   425→  };
   426→
   427→  const handleDragEnd = () => {
   428→    setDragItem(null);
   429→    setDragOverTarget(null);
   430→  };
   431→
   432→  const handleSubmit = () => {
   433→    const answer = Object.entries(matches)
   434→      .map(([l, r]) => `${l}-${r}`)
   435→      .join(', ');
   436→    onAnswer(answer);
   437→  };
   438→
   439→  const allMatched = pairs.every((p) => matches[p.left]);
   440→  const usedRights = new Set(Object.values(matches));
   441→
   442→  return (
   443→    <div className="space-y-4">
   444→      <p className="text-sm text-muted-foreground">
   445→        Drag left items to their match on the right, or click to select. Click a connected left item to remove it.
   446→      </p>
   447→      <div ref={containerRef} className="relative grid grid-cols-2 gap-4">
   448→        {/* SVG Connector Overlay */}
   449→        <svg className="absolute inset-0 pointer-events-none z-10" style={{ width: '100%', height: '100%', overflow: 'visible' }}>
   450→          <defs>
   451→            <linearGradient id="connectorGrad" x1="0%" y1="0%" x2="100%" y2="0%">
   452→              <stop offset="0%" stopColor="#10b981" />
   453→              <stop offset="50%" stopColor="#14b8a6" />
   454→              <stop offset="100%" stopColor="#0d9488" />
   455→            </linearGradient>
   456→          </defs>
   457→          <AnimatePresence>
   458→            {linePositions.map((pos) => {
   459→              const midX = (pos.x1 + pos.x2) / 2;
   460→              return (
   461→                <motion.path
   462→                  key={pos.id}
   463→                  d={`M ${pos.x1} ${pos.y1} C ${midX} ${pos.y1}, ${midX} ${pos.y2}, ${pos.x2} ${pos.y2}`}
   464→                  fill="none"
   465→                  stroke="url(#connectorGrad)"
   466→                  strokeWidth="2.5"
   467→                  strokeLinecap="round"
   468→                  initial={{ pathLength: 0, opacity: 0 }}
   469→                  animate={{ pathLength: 1, opacity: 0.85 }}
   470→                  exit={{ pathLength: 0, opacity: 0 }}
   471→                  transition={{ duration: 0.5, ease: 'easeOut' }}
   472→                />
   473→              );
   474→            })}
   475→          </AnimatePresence>
   476→        </svg>
   477→
   478→        {/* Left column */}
   479→        <div className="flex flex-col gap-2">
   480→          {pairs.map((p) => {
   481→            const isMatched = !!matches[p.left];
   482→            const isDragging = dragItem === p.left;
   483→            return (
   484→              <motion.button
   485→                key={p.left}
   486→                data-left-idx={leftIdxMap[p.left]}
   487→                draggable={!isMatched}
   488→                whileTap={{ scale: 0.98 }}
   489→                onClick={() => handleLeftClick(p.left)}
   490→                onDragStart={(e) => handleDragStart(e, p.left)}
   491→                onDragEnd={handleDragEnd}
   492→                className={`rounded-lg border p-3 text-left text-sm font-medium transition-all ${
   493→                  isDragging
   494→                    ? 'border-primary/50 bg-primary/5 opacity-50 scale-95'
   495→                    : selectedLeft === p.left
   496→                      ? 'border-primary bg-primary/10 text-primary ring-1 ring-primary/30 shadow-sm shadow-primary/10'
   497→                      : isMatched
   498→                        ? 'border-emerald-500/50 bg-emerald-500/5 text-emerald-700 dark:text-emerald-400 shadow-sm shadow-emerald-500/5'
   499→                        : 'border-border hover:border-primary/30 hover:bg-primary/5 cursor-grab active:cursor-grabbing'
   500→                }`}
   501→              >
   502→                <div className="flex items-center gap-2">
   503→                  <GripVertical className="h-3.5 w-3.5 text-muted-foreground/40 shrink-0" />
   504→                  <span className="flex-1">{p.left}</span>
   505→                  {isMatched && (
   506→                    <motion.span
   507→                      initial={{ opacity: 0, x: 10 }}
   508→                      animate={{ opacity: 1, x: 0 }}
   509→                      className="text-xs opacity-70 truncate max-w-[120px] text-emerald-600 dark:text-emerald-400"
   510→                    >
   511→                      {matches[p.left]}
   512→                    </motion.span>
   513→                  )}
   514→                </div>
   515→              </motion.button>
   516→            );
   517→          })}
   518→        </div>
   519→        {/* Right column */}
   520→        <div className="flex flex-col gap-2">
   521→          {shuffledRight.map((r) => {
   522→            const isUsed = usedRights.has(r);
   523→            const isDragOver = dragOverTarget === r && !isUsed;
   524→            return (
   525→              <motion.button
   526→                key={r}
   527→                data-right-idx={rightIdxMap[r]}
