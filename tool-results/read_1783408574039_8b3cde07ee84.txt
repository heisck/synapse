    49тЖТ    49тЖТfunction levenshteinDistance(a: string, b: string): number {
    50тЖТ    50тЖТ  const matrix: number[][] = [];
    51тЖТ    51тЖТ  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    52тЖТ    52тЖТ  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    53тЖТ    53тЖТ  for (let i = 1; i <= b.length; i++) {
    54тЖТ    54тЖТ    for (let j = 1; j <= a.length; j++) {
    55тЖТ    55тЖТ      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    56тЖТ    56тЖТ      matrix[i][j] = Math.min(
    57тЖТ    57тЖТ        matrix[i - 1][j] + 1,
    58тЖТ    58тЖТ        matrix[i][j - 1] + 1,
    59тЖТ    59тЖТ        matrix[i - 1][j - 1] + cost,
    60тЖТ    60тЖТ      );
    61тЖТ    61тЖТ    }
    62тЖТ    62тЖТ  }
    63тЖТ    63тЖТ  return matrix[b.length][a.length];
    64тЖТ    64тЖТ}

    66тЖТ    66тЖТfunction isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    67тЖТ    67тЖТ  const ua = userAnswer.trim().toLowerCase();
    68тЖТ    68тЖТ  const ca = correctAnswer.trim().toLowerCase();
    69тЖТ    69тЖТ  if (ua === ca) return true;
    70тЖТ    70тЖТ  if (ua.includes(ca) || ca.includes(ua)) return true;
    71тЖТ    71тЖТ  // Levenshtein distance for typo tolerance
    72тЖТ    72тЖТ  const dist = levenshteinDistance(ua, ca);
    73тЖТ    73тЖТ  // Allow 1 typo for short answers, 2 for longer ones
    74тЖТ    74тЖТ  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    75тЖТ    75тЖТ  if (dist <= threshold) return true;
    76тЖТ    76тЖТ  // Word-level check: all words of correct answer present in user answer (in any order)
    77тЖТ    77тЖТ  const correctWords = ca.split(/\s+/);
    78тЖТ    78тЖТ  const userWords = ua.split(/\s+/);
    79тЖТ    79тЖТ  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    80тЖТ    80тЖТ  return false;
    81тЖТ    81тЖТ}

    87тЖТ    87тЖТ  const ua = userAnswer.trim().toLowerCase();
    88тЖТ    88тЖТ  const ca = correctAnswer.trim().toLowerCase();
    89тЖТ    89тЖТ  if (ua === ca) {
    90тЖТ    90тЖТ    return { status: 'correct', points: hintUsed ? 0.75 : 1, message: '' };
    91тЖТ    91тЖТ  }
    92тЖТ    92тЖТ  const dist = levenshteinDistance(ua, ca);
    93тЖТ    93тЖТ  let basePoints = 0;
    94тЖТ    94тЖТ  if (dist <= 2) basePoints = 0.8;
    95тЖТ    95тЖТ  else if (dist <= 3) basePoints = 0.5;
    96тЖТ    96тЖТ  if (basePoints > 0) {
    97тЖТ    97тЖТ    return { status: 'close', points: basePoints * (hintUsed ? 0.75 : 1), message: `Close! Did you mean: "${correctAnswer}"?` };
    98тЖТ    98тЖТ  }
    99тЖТ    99тЖТ  return { status: 'wrong', points: 0, message: '' };
   100тЖТ   100тЖТ}

   352тЖТ   352тЖТfunction scoreQuestionForAdaptive(input: AdaptiveScoringInput): ScoredQuestion {
   353тЖТ   353тЖТ  const { question, masteryMap, adaptiveResults, dueConcepts } = input;
   354тЖТ   354тЖТ  let score = 0;
   355тЖТ   355тЖТ  const reasons: string[] = [];
   356тЖТ   356тЖТ  const concept = question.concept || 'Unknown';
   357тЖТ   357тЖТ  const mastery = masteryMap[concept];
   358тЖТ   358тЖТ
   359тЖТ   359тЖТ  // 1. Low mastery bonus (level 1-2)
   360тЖТ   360тЖТ  if (mastery && mastery.level < 3) {
   361тЖТ   361тЖТ    score += 3;
   362тЖТ   362тЖТ    reasons.push('low mastery');
   363тЖТ   363тЖТ  }
   364тЖТ   364тЖТ
   365тЖТ   365тЖТ  // 2. Spaced repetition due for review
   366тЖТ   366тЖТ  if (dueConcepts.has(concept)) {
   367тЖТ   367тЖТ    score += 2;
   368тЖТ   368тЖТ    reasons.push('needs review');
   369тЖТ   369тЖТ  }
   370тЖТ   370тЖТ
   371тЖТ   371тЖТ  // 3. Recent performance: last 5 results on concept
   372тЖТ   372тЖТ  const conceptResults = adaptiveResults
   373тЖТ   373тЖТ    .filter((r) => r.concept === concept)
   374тЖТ   374тЖТ    .slice(-5);
   375тЖТ   375тЖТ  const last3 = conceptResults.slice(-3);
   376тЖТ   376тЖТ  const wrongCount = last3.filter((r) => !r.correct).length;
   377тЖТ   377тЖТ  const correctCount = conceptResults.filter((r) => r.correct).length;
   378тЖТ   378тЖТ
   379тЖТ   379тЖТ  if (wrongCount >= 2) {
   380тЖТ   380тЖТ    score += 2;
   381тЖТ   381тЖТ    reasons.push('recent struggles');
   382тЖТ   382тЖТ  }
   383тЖТ   383тЖТ
   384тЖТ   384тЖТ  // 4. Difficulty match bonus
   385тЖТ   385тЖТ  const targetLevel = mastery ? Math.min(mastery.level + 1, 3) : 1;
   386тЖТ   386тЖТ  const targetDifficulty = targetLevel === 1 ? 'easy' : targetLevel === 2 ? 'medium' : 'hard';
   387тЖТ   387тЖТ  if (question.difficulty === targetDifficulty) {
   388тЖТ   388тЖТ    score += 1;
   389тЖТ   389тЖТ  } else if (
   390тЖТ   390тЖТ    (targetDifficulty === 'medium' && question.difficulty === 'easy') ||
   391тЖТ   391тЖТ    (targetDifficulty === 'hard' && question.difficulty === 'medium')
   392тЖТ   392тЖТ  ) {
   393тЖТ   393тЖТ    score += 0.5;
   394тЖТ   394тЖТ  }
   395тЖТ   395тЖТ
   396тЖТ   396тЖТ  // 5. Boost for concepts with many wrong answers (weaker areas)
   397тЖТ   397тЖТ  if (correctCount >= 3) {
   398тЖТ   398тЖТ    score += 1;
   399тЖТ   399тЖТ    reasons.push('ready for harder questions');
   400тЖТ   400тЖТ  }
   401тЖТ   401тЖТ
   402тЖТ   402тЖТ  return { question, score, reasons };
   403тЖТ   403тЖТ}

   405тЖТ   405тЖТfunction getAdaptiveReasoning(scored: ScoredQuestion[]): string {
   406тЖТ   406тЖТ  if (scored.length === 0) return '';
   407тЖТ   407тЖТ
   408тЖТ   408тЖТ  const conceptReasons = new Map<string, string[]>();
   409тЖТ   409тЖТ  for (const sq of scored.slice(0, 5)) {
   410тЖТ   410тЖТ    const concept = sq.question.concept || 'Unknown';
   411тЖТ   411тЖТ    const existing = conceptReasons.get(concept) || [];
   412тЖТ   412тЖТ    conceptReasons.set(concept, [...existing, ...sq.reasons]);
   413тЖТ   413тЖТ  }
   414тЖТ   414тЖТ
   415тЖТ   415тЖТ  const parts: string[] = [];
   416тЖТ   416тЖТ  for (const [concept, reasons] of conceptReasons) {
   417тЖТ   417тЖТ    const uniqueReasons = [...new Set(reasons)];
   418тЖТ   418тЖТ    const desc = uniqueReasons.map((r) => {
   419тЖТ   419тЖТ      switch (r) {
   420тЖТ   420тЖТ        case 'low mastery': return 'low mastery';
   421тЖТ   421тЖТ        case 'needs review': return 'needs review';
   422тЖТ   422тЖТ        case 'recent struggles': return 'struggling recently';
   423тЖТ   423тЖТ        case 'ready for harder questions': return 'ready for challenge';
   424тЖТ   424тЖТ        default: return r;
   425тЖТ   425тЖТ      }
   426тЖТ   426тЖТ    }).join(', ');
   427тЖТ   427тЖТ    parts.push(`${concept} (${desc})`);
   428тЖТ   428тЖТ  }
   429тЖТ   429тЖТ
   430тЖТ   430тЖТ  return 'Focusing on: ' + parts.join(', ');
   431тЖТ   431тЖТ}

   498тЖТ   498тЖТfunction useAnimatedCounter(target: number, duration: number = 1200) {
   499тЖТ   499тЖТ  const [count, setCount] = useState(0);
   500тЖТ   500тЖТ  const rafRef = useRef<number>(0);
   501тЖТ   501тЖТ  useEffect(() => {
   502тЖТ   502тЖТ    const start = performance.now();
   503тЖТ   503тЖТ    const tick = (now: number) => {
   504тЖТ   504тЖТ      const elapsed = now - start;
   505тЖТ   505тЖТ      const progress = Math.min(elapsed / duration, 1);
   506тЖТ   506тЖТ      const eased = 1 - Math.pow(1 - progress, 3);
   507тЖТ   507тЖТ      setCount(Math.round(eased * target));
   508тЖТ   508тЖТ      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   509тЖТ   509тЖТ    };
   510тЖТ   510тЖТ    rafRef.current = requestAnimationFrame(tick);
   511тЖТ   511тЖТ    return () => cancelAnimationFrame(rafRef.current);
   512тЖТ   512тЖТ  }, [target, duration]);
   513тЖТ   513тЖТ  return count;
   514тЖТ   514тЖТ}

   568тЖТ   568тЖТfunction MatchingInput({
   569тЖТ   569тЖТ  pairs,
   570тЖТ   570тЖТ  onAnswer,
   571тЖТ   571тЖТ}: {
   572тЖТ   572тЖТ  pairs: Array<{ left: string; right: string }>;
   573тЖТ   573тЖТ  onAnswer: (answer: string) => void;
   574тЖТ   574тЖТ}) {
   575тЖТ   575тЖТ  const shuffledRight = useMemo(() => {
   576тЖТ   576тЖТ    const arr = [...pairs.map((p) => p.right)];
   577тЖТ   577тЖТ    for (let i = arr.length - 1; i > 0; i--) {
   578тЖТ   578тЖТ      const j = Math.floor(Math.random() * (i + 1));
   579тЖТ   579тЖТ      [arr[i], arr[j]] = [arr[j], arr[i]];
   580тЖТ   580тЖТ    }
   581тЖТ   581тЖТ    return arr;
   582тЖТ   582тЖТ  }, [pairs]);
   583тЖТ   583тЖТ
   584тЖТ   584тЖТ  const [selectedLeft, setSelectedLeft] = useState<string | null>(null);
   585тЖТ   585тЖТ  const [matches, setMatches] = useState<Record<string, string>>({});
   586тЖТ   586тЖТ  const [dragItem, setDragItem] = useState<string | null>(null);
   587тЖТ   587тЖТ  const [dragOverTarget, setDragOverTarget] = useState<string | null>(null);

   589тЖТ   589тЖТ  const containerRef = useRef<HTMLDivElement>(null);
   590тЖТ   590тЖТ
   591тЖТ   591тЖТ  // Build index maps for safe DOM queries
   592тЖТ   592тЖТ  const leftIdxMap = useMemo(() => {
   593тЖТ   593тЖТ    const map: Record<string, number> = {};
   594тЖТ   594тЖТ    pairs.forEach((p, i) => { map[p.left] = i; });
   595тЖТ   595тЖТ    return map;
   596тЖТ   596тЖТ  }, [pairs]);
   597тЖТ   597тЖТ
   598тЖТ   598тЖТ  const rightIdxMap = useMemo(() => {
   599тЖТ   599тЖТ    const map: Record<string, number> = {};
   600тЖТ   600тЖТ    shuffledRight.forEach((r, i) => { map[r] = i; });
   601тЖТ   601тЖТ    return map;
   602тЖТ   602тЖТ  }, [shuffledRight]);
   603тЖТ   603тЖТ
   604тЖТ   604тЖТ  // Compute SVG connector line positions
   605тЖТ   605тЖТ  const updateLinePositions = useCallback(() => {
   606тЖТ   606тЖТ    const container = containerRef.current;
   607тЖТ   607тЖТ    if (!container) return;

   613тЖТ   613тЖТ

   616тЖТ   616тЖТ        const rIdx = rightIdxMap[right];

   618тЖТ   618тЖТ

   620тЖТ   620тЖТ        const rightEl = container.querySelector(`[data-right-idx="${rIdx}"]`);
   621тЖТ   621тЖТ

   624тЖТ   624тЖТ          const rightRect = rightEl.getBoundingClientRect();
   625тЖТ   625тЖТ
   626тЖТ   626тЖТ          positions.push({
   627тЖТ   627тЖТ            id: `${left}-${right}`,
   628тЖТ   628тЖТ            x1: leftRect.right - containerRect.left,
   629тЖТ   629тЖТ            y1: leftRect.top + leftRect.height / 2 - containerRect.top,
   630тЖТ   630тЖТ            x2: rightRect.left - containerRect.left,
   631тЖТ   631тЖТ            y2: rightRect.top + rightRect.height / 2 - containerRect.top,

   647тЖТ   647тЖТ    const container = containerRef.current;
   648тЖТ   648тЖТ    if (!container) return;
   649тЖТ   649тЖТ
   650тЖТ   650тЖТ    const observer = new ResizeObserver(updateLinePositions);
   651тЖТ   651тЖТ    observer.observe(container);
   652тЖТ   652тЖТ
   653тЖТ   653тЖТ    return () => observer.disconnect();
   654тЖТ   654тЖТ  }, [updateLinePositions]);
   655тЖТ   655тЖТ
   656тЖТ   656тЖТ  const handleLeftClick = (left: string) => {
   657тЖТ   657тЖТ    if (matches[left]) {
   658тЖТ   658тЖТ      // Remove match by clicking connected left item
   659тЖТ   659тЖТ      setMatches((prev) => {
   660тЖТ   660тЖТ        const next = { ...prev };
   661тЖТ   661тЖТ        delete next[left];
   662тЖТ   662тЖТ        return next;
   663тЖТ   663тЖТ      });
   664тЖТ   664тЖТ      return;
   665тЖТ   665тЖТ    }
   666тЖТ   666тЖТ    setSelectedLeft(left);
   667тЖТ   667тЖТ  };
   668тЖТ   668тЖТ

   672тЖТ   672тЖТ    setSelectedLeft(null);
   673тЖТ   673тЖТ  };
