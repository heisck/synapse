    25тЖТ    25тЖТfunction levenshteinDistance(a: string, b: string): number {
    26тЖТ    26тЖТ  const matrix: number[][] = [];
    27тЖТ    27тЖТ  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    28тЖТ    28тЖТ  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    29тЖТ    29тЖТ  for (let i = 1; i <= b.length; i++) {
    30тЖТ    30тЖТ    for (let j = 1; j <= a.length; j++) {
    31тЖТ    31тЖТ      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    32тЖТ    32тЖТ      matrix[i][j] = Math.min(
    33тЖТ    33тЖТ        matrix[i - 1][j] + 1,
    34тЖТ    34тЖТ        matrix[i][j - 1] + 1,
    35тЖТ    35тЖТ        matrix[i - 1][j - 1] + cost,
    36тЖТ    36тЖТ      );
    37тЖТ    37тЖТ    }
    38тЖТ    38тЖТ  }
    39тЖТ    39тЖТ  return matrix[b.length][a.length];
    40тЖТ    40тЖТ}

    42тЖТ    42тЖТfunction isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    43тЖТ    43тЖТ  const ua = userAnswer.trim().toLowerCase();
    44тЖТ    44тЖТ  const ca = correctAnswer.trim().toLowerCase();
    45тЖТ    45тЖТ  if (ua === ca) return true;
    46тЖТ    46тЖТ  if (ua.includes(ca) || ca.includes(ua)) return true;
    47тЖТ    47тЖТ  // Levenshtein distance for typo tolerance
    48тЖТ    48тЖТ  const dist = levenshteinDistance(ua, ca);
    49тЖТ    49тЖТ  // Allow 1 typo for short answers, 2 for longer ones
    50тЖТ    50тЖТ  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    51тЖТ    51тЖТ  if (dist <= threshold) return true;
    52тЖТ    52тЖТ  // Word-level check: all words of correct answer present in user answer (in any order)
    53тЖТ    53тЖТ  const correctWords = ca.split(/\s+/);
    54тЖТ    54тЖТ  const userWords = ua.split(/\s+/);
    55тЖТ    55тЖТ  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    56тЖТ    56тЖТ  return false;
    57тЖТ    57тЖТ}
