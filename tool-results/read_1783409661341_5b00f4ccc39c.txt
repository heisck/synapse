    84→// Confetti particle component
    85→function ConfettiParticle({ index }: { index: number }) {
    86→  const colors = ['#10b981', '#14b8a6', '#f59e0b', '#3b82f6', '#ec4899', '#8b5cf6'];
    87→  const color = colors[index % colors.length];
    88→  const startX = 50 + (Math.sin(index * 2.4) * 40);
    89→  const startY = -10;
    90→  const endX = startX + (Math.cos(index * 1.7) * 60);
    91→  const endY = 120;
    92→  const rotation = index * 137.5;
    93→  const size = 4 + (index % 3) * 2;
    94→
    95→  return (
    96→    <motion.div
    97→      initial={{ x: `${startX}%`, y: `${startY}%`, rotate: 0, opacity: 1, scale: 1 }}
    98→      animate={{
    99→        y: `${endY}%`,
   100→        x: `${endX}%`,
   101→        rotate: rotation + 720,
   102→        opacity: 0,
   103→        scale: 0.3,
   104→      }}
   105→      transition={{
   106→        duration: 1.5 + (index % 4) * 0.3,
   107→        delay: index * 0.05,
   108→        ease: 'easeIn',
   109→      }}
   110→      className="absolute rounded-sm pointer-events-none"
   111→      style={{
   112→        width: size,
   113→        height: size,
   114→        backgroundColor: color,
   115→        left: 0,
   116→        top: 0,
   117→      }}
   118→    />
   119→  );
   120→}

   498→                      whileTap={{ scale: 0.98 }}
   499→                      onClick={() => handleSlideSelect(i)}
   500→                      className={`flex items-center gap-3 rounded-lg px-3 py-2.5 text-left text-sm whitespace-nowrap lg:whitespace-normal transition-all min-w-[160px] lg:min-w-0 ${
   501→                        i === currentSlideIndex
   502→                          ? 'bg-gradient-to-r from-primary/10 to-secondary/5 text-primary font-medium shadow-sm border border-primary/10'
   503→                          : viewed
   504→                            ? 'text-foreground glass'
   505→                            : 'text-muted-foreground hover:bg-accent'
   506→                      }`}
   507→                    >
   508→                      <span className={`flex h-6 w-6 shrink-0 items-center justify-center rounded text-xs font-bold transition-colors ${
   509→                        i === currentSlideIndex
   510→                          ? 'bg-primary text-primary-foreground'
   511→                          : viewed
   512→                            ? 'bg-emerald-500/15 text-emerald-600 dark:text-emerald-400'
   513→                            : 'bg-muted text-muted-foreground'
   514→                      }`}>
   515→                        {viewed && i !== currentSlideIndex ? (
   516→                          <motion.span initial={{ scale: 0 }} animate={{ scale: 1 }} transition={{ type: 'spring', stiffness: 300 }}>
   517→                            <Check className="h-3.5 w-3.5" />
   518→                          </motion.span>
   519→                        ) : i === currentSlideIndex ? (
   520→                          i + 1
   521→                        ) : (
   522→                          i + 1
   523→                        )}
   524→                      </span>
   525→                      <span className="truncate">{slide.title}</span>
   526→                    </motion.button>
   527→                  );
   528→                })}
   529→              </motion.div>
   530→            </ScrollArea>
   531→          </div>
   532→        </div>
   533→
   534→        {/* Right panel: slide content */}
   535→        <div className="flex-1 min-w-0">
   536→          <AnimatePresence mode="wait" custom={slideDirection}>
   537→          {currentSlide ? (
   538→            <motion.div
   539→              key={currentSlide.id}
   540→              custom={slideDirection}
   541→              initial={{ opacity: 0, x: slideDirection * 40, scale: 0.98 }}
   542→              animate={{ opacity: 1, x: 0, scale: 1 }}
   543→              exit={{ opacity: 0, x: slideDirection * -40, scale: 0.98 }}
   544→              transition={{ type: 'spring', stiffness: 250, damping: 25 }}
   545→              className="glass rounded-xl p-6 space-y-4 glow-emerald"
   546→            >
   547→              <div className="flex items-start justify-between gap-4">
   548→                <div className="space-y-1">
   549→                  <div className="flex items-center gap-2">
   550→                    <motion.div initial={{ rotate: -10, scale: 0.8 }} animate={{ rotate: 0, scale: 1 }} transition={{ type: 'spring' }}>
   551→                      <FileText className="h-4 w-4 text-primary shrink-0" />
   552→                    </motion.div>
   553→                    <h2 className="text-lg font-semibold leading-snug">{currentSlide.title}</h2>
   554→                  </div>
   555→                  <p className="text-xs text-muted-foreground">Slide {currentSlide.order} of {slides.length}</p>
   556→                </div>
   557→                <motion.div whileHover={{ scale: 1.05 }} whileTap={{ scale: 0.95 }}>
   558→                  <Button
   559→                    size="sm"
   560→                    variant="outline"
   561→                    onClick={() => setActiveSlideId(activeSlideId === currentSlide.id ? null : currentSlide.id)}
   562→                    className={activeSlideId === currentSlide.id ? 'glow-emerald border-primary/30' : ''}
   563→                  >
   564→                    <Sparkles className="h-4 w-4 mr-1" />
   565→                    Generate Questions
   566→                  </Button>
   567→                </motion.div>
   568→              </div>
   569→
   570→              <Separator />
   571→
   572→              {/* Slide content with animated bullet points */}
   573→              <div id="slide-content-scroll" className="prose prose-sm dark:prose-invert max-w-none max-h-[50vh] overflow-y-auto pr-2">
   574→                {currentSlide.content.split('\n').map((line, i) => {
   575→                  if (!line.trim()) return <br key={i} />;
   576→                  if (line.startsWith('• ') || line.startsWith('- ')) {
   577→                    return (
   578→                      <motion.div
   579→                        key={i}
   580→                        custom={i}
   581→                        variants={bulletVariants}
   582→                        initial="hidden"
   583→                        animate="visible"
   584→                        className="flex gap-2 ml-2 py-0.5"
   585→                      >
   586→                        <motion.span
   587→                          className="mt-1.5 h-1.5 w-1.5 rounded-full bg-primary shrink-0"
   588→                          initial={{ scale: 0 }}
   589→                          animate={{ scale: 1 }}
   590→                          transition={{ delay: i * 0.06 + 0.1, type: 'spring', stiffness: 400 }}
   591→                        />
   592→                        <span dangerouslySetInnerHTML={{ __html: formatBold(line.slice(2)) }} />
   593→                      </motion.div>
   594→                    );
   595→                  }
   596→                  if (/^\d+\.\s/.test(line)) {
   597→                    return (
   598→                      <motion.div
   599→                        key={i}
   600→                        custom={i}
   601→                        variants={bulletVariants}
   602→                        initial="hidden"
   603→                        animate="visible"
   604→                        className="flex gap-2 ml-2 py-0.5"
   605→                      >
   606→                        <span className="flex h-5 w-5 shrink-0 items-center justify-center rounded bg-primary/10 text-primary text-[10px] font-bold mt-0.5">
   607→                          {line.match(/^(\d+)\./)?.[1]}
   608→                        </span>
   609→                        <span dangerouslySetInnerHTML={{ __html: formatBold(line.replace(/^\d+\.\s/, '')) }} />
   610→                      </motion.div>
   611→                    );
   612→                  }
   613→                  if (line.startsWith('**') && line.endsWith('**')) {
   614→                    return (
   615→                      <motion.h3
   616→                        key={i}
   617→                        initial={{ opacity: 0, y: 5 }}
   618→                        animate={{ opacity: 1, y: 0 }}
   619→                        transition={{ delay: i * 0.04 }}
   620→                        className="font-semibold text-sm mt-3 mb-1 gradient-text"
   621→                      >
   622→                        {line.replace(/\*\*/g, '')}
   623→                      </motion.h3>
   624→                    );
   625→                  }
   626→                  return (
   627→                    <motion.p
   628→                      key={i}
   629→                      initial={{ opacity: 0 }}
   630→                      animate={{ opacity: 1 }}
   631→                      transition={{ delay: i * 0.03 }}
   632→                      className="text-sm leading-relaxed"
   633→                      dangerouslySetInnerHTML={{ __html: formatBold(line) }}
   634→                    />
   635→                  );
   636→                })}
   637→              </div>
   638→
   639→              {/* Slide navigation */}
   640→              <div className="flex items-center justify-between pt-2">
   641→                <Button
   642→                  variant="outline"
   643→                  size="sm"
   644→                  onClick={handlePrev}
   645→                  disabled={currentSlideIndex === 0}
   646→                  className="gap-1"
   647→                >
   648→                  <ChevronLeft className="h-4 w-4" />
   649→                  Previous
   650→                </Button>
   651→                <div className="flex items-center gap-1">
   652→                  {slides.map((slide, i) => (
   653→                    <button
   654→                      key={slide.id}
   655→                      onClick={() => handleSlideSelect(i)}
   656→                      className={`h-1.5 rounded-full transition-all ${
   657→                        i === currentSlideIndex ? 'w-4 bg-primary' : 'w-1.5 bg-muted'
   658→                      }`}
   659→                    />
   660→                  ))}
   661→                </div>
   662→                <Button
   663→                  size="sm"
   664→                  onClick={handleNext}
   665→                  disabled={currentSlideIndex === slides.length - 1}
   666→                  className="gap-1"
   667→                >
   668→                  Next
   669→                  <ChevronRight className="h-4 w-4" />
   670→                </Button>
   671→              </div>
   672→
   673→              {/* Generated questions placeholder */}
   674→              {activeSlideId === currentSlide.id && (
   675→                <motion.div
   676→                  initial={{ opacity: 0, y: 10, height: 0 }}
   677→                  animate={{ opacity: 1, y: 0, height: 'auto' }}
   678→                  exit={{ opacity: 0, y: -10, height: 0 }}
   679→                  className="rounded-lg border border-primary/20 bg-primary/5 p-4 overflow-hidden"
   680→                >
   681→                  <div className="flex items-center gap-2 mb-2">
   682→                    <motion.span
   683→                      animate={{ rotate: [0, 15, -15, 0] }}
   684→                      transition={{ duration: 0.6, repeat: Infinity, repeatDelay: 2 }}
   685→                    >
   686→                      <Sparkles className="h-4 w-4 text-primary" />
   687→                    </motion.span>
   688→                    <span className="text-sm font-semibold text-primary">Questions Generated</span>
   689→                  </div>
   690→                  <p className="text-sm text-muted-foreground">
   691→                    5 questions generated for &ldquo;{currentSlide.title}&rdquo;.
   692→                    <Button variant="link" size="sm" className="px-1 h-auto font-medium" onClick={handleTakeQuiz}>
   693→                      Start Quiz →
   694→                    </Button>
   695→                  </p>
   696→                </motion.div>
   697→              )}
   698→            </motion.div>
   699→          ) : (
   700→            <div className="glass rounded-xl flex items-center justify-center h-64">
   701→              <div className="text-center">
   702→                <BookOpen className="h-10 w-10 text-muted-foreground/30 mx-auto mb-2" />
   703→                <p className="text-muted-foreground">Select a slide to view its content</p>
   704→              </div>
   705→            </div>
   706→          )}
   707→          </AnimatePresence>
   708→        </div>
   709→      </motion.div>
   710→    </motion.div>
   711→  );
   712→}
   713→
   714→/** Simple markdown bold formatter */
   715→function formatBold(text: string): string {
   716→  return text.replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>');
   717→}
