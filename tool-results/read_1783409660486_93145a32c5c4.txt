   107→function formatRelativeTime(timestamp: number): string {
   108→  const now = Date.now();
   109→  const diff = now - timestamp;
   110→  const minutes = Math.floor(diff / 60000);
   111→  const hours = Math.floor(diff / 3600000);
   112→  const days = Math.floor(diff / 86400000);
   113→
   114→  if (minutes < 1) return 'Just now';
   115→  if (minutes < 60) return `${minutes}m ago`;
   116→  if (hours < 24) return `${hours}h ago`;
   117→  if (days === 1) return 'Yesterday';
   118→  if (days < 7) return `${days}d ago`;
   119→  return `${Math.floor(days / 7)}w ago`;
   120→}

   164→// Typewriter text component for greeting animation
   165→function TypewriterText({ text, speed = 40, className = '' }: { text: string; speed?: number; className?: string }) {
   166→  const [displayed, setDisplayed] = useState('');
   167→  const [done, setDone] = useState(false);
   168→
   169→  useEffect(() => {
   170→    setDisplayed('');
   171→    setDone(false);
   172→    let i = 0;
   173→    const timer = setInterval(() => {
   174→      i++;
   175→      setDisplayed(text.slice(0, i));
   176→      if (i >= text.length) {
   177→        setDone(true);
   178→        clearInterval(timer);
   179→      }
   180→    }, speed);
   181→    return () => clearInterval(timer);
   182→  }, [text, speed]);
   183→
   184→  return (
   185→    <span className={className}>
   186→      {displayed}
   187→      {!done && (
   188→        <motion.span
   189→          animate={{ opacity: [1, 0] }}
   190→          transition={{ duration: 0.4, repeat: Infinity, repeatType: 'reverse' }}
   191→          className="inline-block w-[2px] h-[0.85em] bg-primary ml-0.5 align-middle rounded-full"
   192→        />
   193→      )}
   194→    </span>
   195→  );
   196→}

   242→// Circular SVG progress ring with emerald gradient
   243→function GoalProgressRing({ percentage, size = 64, strokeWidth = 5 }: { percentage: number; size?: number; strokeWidth?: number }) {
   244→  const radius = (size - strokeWidth) / 2;
   245→  const circumference = 2 * Math.PI * radius;
   246→  const clampedPct = Math.min(Math.max(percentage, 0), 100);
   247→  const isComplete = clampedPct >= 100;
   248→  const countUpValue = useCountUp(Math.round(clampedPct), { duration: 1000 });
   249→
   250→  return (
   251→    <div className="relative" style={{ width: size, height: size }}>
   252→      <svg width={size} height={size} className="-rotate-90">
   253→        <defs>
   254→          <linearGradient id={`goalRing-${size}`} x1="0%" y1="0%" x2="100%" y2="100%">
   255→            <stop offset="0%" stopColor="#10b981" />
   256→            <stop offset="100%" stopColor="#14b8a6" />
   257→          </linearGradient>
   258→        </defs>
   259→        <circle
   260→          cx={size / 2}
   261→          cy={size / 2}
   262→          r={radius}
   263→          fill="none"
   264→          stroke="currentColor"
   265→          strokeWidth={strokeWidth}
   266→          className="text-muted/20"
   267→        />
   268→        <motion.circle
   269→          cx={size / 2}
   270→          cy={size / 2}
   271→          r={radius}
   272→          fill="none"
   273→          stroke={`url(#goalRing-${size})`}
   274→          strokeWidth={strokeWidth}
   275→          strokeLinecap="round"
   276→          strokeDasharray={circumference}
   277→          initial={{ strokeDashoffset: circumference }}
   278→          animate={{ strokeDashoffset: circumference - (clampedPct / 100) * circumference }}
   279→          transition={{ duration: 1.2, ease: 'easeOut', type: 'spring', stiffness: 300, damping: 25 }}

   319→// Hover tilt wrapper for StatsCards
   320→function TiltCard({ children, className = '' }: { children: React.ReactNode; className?: string }) {
   321→  const ref = useRef<HTMLDivElement>(null);
   322→
   323→  const handleMouseMove = useCallback((e: React.MouseEvent<HTMLDivElement>) => {
   324→    if (!ref.current) return;
   325→    const rect = ref.current.getBoundingClientRect();
   326→    const x = (e.clientX - rect.left) / rect.width - 0.5;
   327→    const y = (e.clientY - rect.top) / rect.height - 0.5;
   328→    ref.current.style.transform = `perspective(600px) rotateY(${x * 4}deg) rotateX(${-y * 4}deg)`;
   329→  }, []);
   330→
   331→  const handleMouseLeave = useCallback(() => {
   332→    if (ref.current) {
   333→      ref.current.style.transform = 'perspective(600px) rotateY(0deg) rotateX(0deg)';
   334→    }
   335→  }, []);
   336→
   337→  return (
   338→    <div
   339→      ref={ref}
   340→      className={className}
   341→      onMouseMove={handleMouseMove}
   342→      onMouseLeave={handleMouseLeave}
   343→      style={{ transition: 'transform 0.2s ease-out' }}
   344→    >
   345→      {children}
   346→    </div>
   347→  );
   348→}

   670→  const [selectedTopics, setSelectedTopics] = useState<string[]>([]);
   671→  const [customTopic, setCustomTopic] = useState('');
   672→  const [hoursPerWeek, setHoursPerWeek] = useState([10]);
   673→  const [level, setLevel] = useState<'beginner' | 'intermediate' | 'advanced'>('intermediate');
   674→  const [goals, setGoals] = useState<string[]>([]);
   675→  const [goalInput, setGoalInput] = useState('');
   676→  const [pace, setPace] = useState('moderate');
   677→  const [style, setStyle] = useState('balanced');
   678→
   679→  // Load persisted plan
   680→  useEffect(() => {
   681→    try {
   682→      const saved = localStorage.getItem('synapse-study-plan');
   683→      if (saved) {
   684→        setPlan(JSON.parse(saved));
   685→      }
   686→    } catch { /* ignore */ }
   687→  }, []);
   688→
   689→  const courseTopics = useMemo(() => {
   690→    return courses.map((c) => c.title);
   691→  }, [courses]);
   692→
   693→  const toggleTopic = (topic: string) => {
   694→    setSelectedTopics((prev) =>
   695→      prev.includes(topic) ? prev.filter((t) => t !== topic) : prev.length < 5 ? [...prev, topic] : prev

   699→  const addCustomTopic = () => {
   700→    const trimmed = customTopic.trim();
   701→    if (trimmed && !selectedTopics.includes(trimmed) && selectedTopics.length < 5) {
   702→      setSelectedTopics((prev) => [...prev, trimmed]);

   722→    setDialogOpen(false);

   724→    try {

   726→        method: 'POST',
   727→        headers: { 'Content-Type': 'application/json' },

   729→          topics: selectedTopics,
   730→          hoursPerWeek: hoursPerWeek[0],
   731→          level,

   733→          preferences: { pace, style },

   739→        setPlan(data);
   740→        setExpandedDay(data.plan.days[0]?.day ?? null);
   741→        setCheckedMilestones(new Set());
