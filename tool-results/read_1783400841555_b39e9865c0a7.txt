    37→    37→function levenshteinDistance(a: string, b: string): number {
    38→    38→  const matrix: number[][] = [];
    39→    39→  for (let i = 0; i <= b.length; i++) matrix[i] = [i];
    40→    40→  for (let j = 0; j <= a.length; j++) matrix[0][j] = j;
    41→    41→  for (let i = 1; i <= b.length; i++) {
    42→    42→    for (let j = 1; j <= a.length; j++) {
    43→    43→      const cost = b[i - 1] === a[j - 1] ? 0 : 1;
    44→    44→      matrix[i][j] = Math.min(
    45→    45→        matrix[i - 1][j] + 1,
    46→    46→        matrix[i][j - 1] + 1,
    47→    47→        matrix[i - 1][j - 1] + cost,
    48→    48→      );
    49→    49→    }
    50→    50→  }
    51→    51→  return matrix[b.length][a.length];
    52→    52→}

    54→    54→function isFuzzyMatch(userAnswer: string, correctAnswer: string, maxDistance: number = 2): boolean {
    55→    55→  const ua = userAnswer.trim().toLowerCase();
    56→    56→  const ca = correctAnswer.trim().toLowerCase();
    57→    57→  if (ua === ca) return true;
    58→    58→  if (ua.includes(ca) || ca.includes(ua)) return true;
    59→    59→  // Levenshtein distance for typo tolerance
    60→    60→  const dist = levenshteinDistance(ua, ca);
    61→    61→  // Allow 1 typo for short answers, 2 for longer ones
    62→    62→  const threshold = maxDistance || (ca.length <= 4 ? 1 : 2);
    63→    63→  if (dist <= threshold) return true;
    64→    64→  // Word-level check: all words of correct answer present in user answer (in any order)
    65→    65→  const correctWords = ca.split(/\s+/);
    66→    66→  const userWords = ua.split(/\s+/);
    67→    67→  if (correctWords.every((w) => userWords.some((uw) => uw.includes(w) || w.includes(uw)))) return true;
    68→    68→  return false;
    69→    69→}

   309→   309→function useAnimatedCounter(target: number, duration: number = 1200) {
   310→   310→  const [count, setCount] = useState(0);
   311→   311→  const rafRef = useRef<number>(0);
   312→   312→  useEffect(() => {
   313→   313→    const start = performance.now();
   314→   314→    const tick = (now: number) => {
   315→   315→      const elapsed = now - start;
   316→   316→      const progress = Math.min(elapsed / duration, 1);
   317→   317→      const eased = 1 - Math.pow(1 - progress, 3);
   318→   318→      setCount(Math.round(eased * target));
   319→   319→      if (progress < 1) rafRef.current = requestAnimationFrame(tick);
   320→   320→    };
   321→   321→    rafRef.current = requestAnimationFrame(tick);
   322→   322→    return () => cancelAnimationFrame(rafRef.current);
   323→   323→  }, [target, duration]);
   324→   324→  return count;
   325→   325→}
